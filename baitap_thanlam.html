Bài Tập Tham Lam

Bộ đề rèn luyện kỹ năng Tham Lam (Greedy) dành cho HSG Tin học

Bài 1: XẾP SỮA
XEPSUA.INP / XEPSUA.OUT

Nông trại có N con bò và M chai sữa. Con bò thứ i chỉ chấp nhận sữa có giá không quá pi đồng. Chai sữa thứ j có giá cj đồng. Mỗi con bò uống tối đa 1 chai, mỗi chai cho tối đa 1 con bò. Tìm số con bò được uống sữa nhiều nhất.

Dữ liệu vào (XEPSUA.INP):
  • Dòng 1: Hai số nguyên N, M (1 ≤ N, M ≤ 105)
  • Dòng 2: N số nguyên p1, p2, ..., pN (1 ≤ pi ≤ 109)
  • Dòng 3: M số nguyên c1, c2, ..., cM (1 ≤ cj ≤ 109)
Kết quả ra (XEPSUA.OUT):
  • Một số nguyên duy nhất — số con bò được uống sữa tối đa

XEPSUA.INP

4 5
3 5 2 4
4 1 6 2 3

XEPSUA.OUT

📝 Code C++ — E1. Sort + Matching 
// Bài 1: XEPSUA — Greedy sort + two-pointer matching
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream fin("XEPSUA.INP");
    ofstream fout("XEPSUA.OUT");
    int n, m;
    fin >> n >> m;
    vector<int> p(n), c(m);
    for (int i = 0; i < n; i++) fin >> p[i];
    for (int i = 0; i < m; i++) fin >> c[i];
    sort(p.begin(), p.end());
    sort(c.begin(), c.end());
    int ans = 0, j = 0;
    for (int i = 0; i < n && j < m; i++) {
        if (c[j] <= p[i]) {
            ans++;
            j++;
        }
    }
    fout << ans;
    return 0;
}
💡 Ý tưởng: Sắp xếp cả hai mảng tăng dần. Duyệt hai con trỏ: nếu sữa rẻ nhất hiện tại ≤ giá chấp nhận của bò, ghép cặp; ngược lại chuyển sang bò có yêu cầu cao hơn. Độ phức tạp: O(N log N + M log M).

📝 Code C++ — E2. Activity Selection 
// Bài 2: CHONDOAN — Greedy sort by end time
#include <bits/stdc++.h>
using namespace std;

bool cmp(pair<int,int> a, pair<int,int> b) {
    return a.second < b.second;
}

int main() {
    ifstream fin("CHONDOAN.INP");
    ofstream fout("CHONDOAN.OUT");
    int n;
    fin >> n;
    vector<pair<int,int>> a(n);
    for (int i = 0; i < n; i++) fin >> a[i].first >> a[i].second;
    sort(a.begin(), a.end(), cmp);
    int ans = 0, last = 0;
    for (int i = 0; i < n; i++) {
        if (a[i].first >= last) {
            ans++;
            last = a[i].second;
        }
    }
    fout << ans;
    return 0;
}
💡 Ý tưởng: Sắp xếp hoạt động theo thời điểm kết thúc tăng dần. Chọn hoạt động kết thúc sớm nhất không chồng chéo với hoạt động đã chọn trước đó. Tham lam dạng Activity Selection kinh điển. Độ phức tạp: O(N log N).

// Bài 3: GHEPSO — Sắp xếp, ghép nhỏ nhất với lớn nhất
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream fin("GHEPSO.INP");
    ofstream fout("GHEPSO.OUT");
    int n;
    fin >> n;
    vector<int> a(n);
    for (int i = 0; i < n; i++) fin >> a[i];
    sort(a.begin(), a.end());
    int ans = 0;
    for (int i = 0; i < n / 2; i++)
        ans = max(ans, a[i] + a[n - 1 - i]);
    fout << ans;
    return 0;
}
💡 Ý tưởng: Sắp xếp dãy tăng dần. Ghép phần tử nhỏ nhất với lớn nhất, nhỏ nhì với lớn nhì, ... Cách ghép này cân bằng tổng các cặp, tối thiểu hóa giá trị lớn nhất. Độ phức tạp: O(N log N).

📝 Code C++ — E4. Two-Pointer 
// Bài 4: THUYENBUOM — Greedy two-pointer
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream fin("THUYENBUOM.INP");
    ofstream fout("THUYENBUOM.OUT");
    int n, C;
    fin >> n >> C;
    vector<int> w(n);
    for (int i = 0; i < n; i++) fin >> w[i];
    sort(w.begin(), w.end());
    int left = 0, right = n - 1, boats = 0;
    while (left <= right) {
        if (left == right) {
            boats++;
            break;
        }
        if (w[left] + w[right] <= C) left++;
        right--;
        boats++;
    }
    fout << boats;
    return 0;
}
💡 Ý tưởng: Sắp xếp trọng lượng tăng dần. Thử ghép người nhẹ nhất (left) với nặng nhất (right). Nếu vừa thuyền thì ghép cặp (left++, right--), nếu không thì người nặng đi riêng (right--). Độ phức tạp: O(N log N).

// Bài 5: SANBAY — Sweep line (quét sự kiện)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream fin("SANBAY.INP");
    ofstream fout("SANBAY.OUT");
    int n;
    fin >> n;
    // Sự kiện: (thời gian, loại). Đến = +1, Đi = -1
    vector<pair<int,int>> ev;
    for (int i = 0; i < n; i++) {
        int a, d;
        fin >> a >> d;
        ev.push_back(make_pair(a, 1));
        ev.push_back(make_pair(d, -1));
    }
    // Sort: cùng thời điểm, xử lý đi (-1) trước đến (+1)
    sort(ev.begin(), ev.end());
    int cur = 0, ans = 0;
    for (int i = 0; i < (int)ev.size(); i++) {
        cur += ev[i].second;
        ans = max(ans, cur);
    }
    fout << ans;
    return 0;
}
💡 Ý tưởng: Tạo sự kiện đến (+1) và đi (−1). Sắp xếp theo thời gian (cùng thời điểm thì xử lý sự kiện đi trước, do pair sắp xếp −1 trước +1). Quét qua và đếm tối đa đồng thời — đó là số cổng cần. Kỹ thuật Sweep Line. Độ phức tạp: O(N log N).

📝 Code C++ — E6. Fractional Knapsack 
// Bài 6: BALOPL — Greedy sort by value/weight ratio
#include <bits/stdc++.h>
using namespace std;

struct Item { int w, v; };

bool cmp(Item a, Item b) {
    return (long long)a.v * b.w > (long long)b.v * a.w;
}

int main() {
    ifstream fin("BALOPL.INP");
    ofstream fout("BALOPL.OUT");
    int n, W;
    fin >> n >> W;
    vector<Item> items(n);
    for (int i = 0; i < n; i++) fin >> items[i].w >> items[i].v;
    sort(items.begin(), items.end(), cmp);
    double ans = 0;
    int rem = W;
    for (int i = 0; i < n && rem > 0; i++) {
        if (items[i].w <= rem) {
            ans += items[i].v;
            rem -= items[i].w;
        } else {
            ans += (double)items[i].v * rem / items[i].w;
            rem = 0;
        }
    }
    fout << fixed << setprecision(2) << ans;
    return 0;
}
💡 Ý tưởng: Sắp xếp vật theo tỉ lệ giá trị/khối lượng giảm dần (dùng nhân chéo tránh sai số thực). Lấy trọn vật nếu còn đủ chỗ, lấy phần phân số nếu không. Đây là bài Fractional Knapsack kinh điển. Độ phức tạp: O(N log N).

// Bài 7: NOITEP — Min-heap, luôn nối 2 tệp nhỏ nhất
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream fin("NOITEP.INP");
    ofstream fout("NOITEP.OUT");
    int n;
    fin >> n;
    priority_queue<long long, vector<long long>, greater<long long>> pq;
    for (int i = 0; i < n; i++) {
        long long x;
        fin >> x;
        pq.push(x);
    }
    long long ans = 0;
    while (pq.size() > 1) {
        long long a = pq.top(); pq.pop();
        long long b = pq.top(); pq.pop();
        ans += a + b;
        pq.push(a + b);
    }
    fout << ans;
    return 0;
}
💡 Ý tưởng: Dùng hàng đợi ưu tiên (min-heap). Luôn lấy 2 tệp nhỏ nhất ra nối, rồi đẩy kết quả lại heap. Chiến lược tương tự mã Huffman. Độ phức tạp: O(N log N).

// Bài 8: NHIENLIEU — Đi xa nhất có thể, đổ khi cần
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream fin("NHIENLIEU.INP");
    ofstream fout("NHIENLIEU.OUT");
    int L, C, n;
    fin >> L >> C >> n;
    vector<int> d(n + 2);
    d[0] = 0;
    for (int i = 1; i <= n; i++) fin >> d[i];
    d[n + 1] = L;
    // Kiểm tra khả thi: mọi khoảng cách liên tiếp ≤ C
    for (int i = 1; i <= n + 1; i++) {
        if (d[i] - d[i - 1] > C) {
            fout << -1;
            return 0;
        }
    }
    int fuel = C, stops = 0;
    for (int i = 1; i <= n + 1; i++) {
        fuel -= (d[i] - d[i - 1]);
        // Tại trạm i: nếu không đủ xăng đến trạm tiếp, đổ đầy
        if (i <= n && fuel < d[i + 1] - d[i]) {
            fuel = C;
            stops++;
        }
    }
    fout << stops;
    return 0;
}
💡 Ý tưởng: Thêm điểm đầu (0) và đích (L) vào danh sách trạm. Đi qua từng trạm, trừ xăng. Tại mỗi trạm xăng, chỉ đổ khi không đủ nhiên liệu đến trạm kế tiếp. Nếu khoảng cách giữa 2 trạm liên tiếp vượt quá C thì không thể. Độ phức tạp: O(N).

// Bài 9: NHAYBAC — Greedy BFS-like (Jump Game II)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream fin("NHAYBAC.INP");
    ofstream fout("NHAYBAC.OUT");
    int n;
    fin >> n;
    vector<int> a(n);
    for (int i = 0; i < n; i++) fin >> a[i];
    if (n == 1) { fout << 0; return 0; }
    // curEnd: biên phải vùng đang xét, farthest: xa nhất có thể tới
    int jumps = 0, curEnd = 0, farthest = 0;
    for (int i = 0; i < n - 1; i++) {
        farthest = max(farthest, i + a[i]);
        if (i == curEnd) {
            if (farthest == curEnd) {
                fout << -1;
                return 0;
            }
            jumps++;
            curEnd = farthest;
            if (curEnd >= n - 1) break;
        }
    }
    fout << jumps;
    return 0;
}
💡 Ý tưởng: Kỹ thuật giống BFS theo tầng: curEnd là biên phải vùng đang xét, farthest là vị trí xa nhất có thể nhảy tới. Khi duyệt đến curEnd, tăng bước nhảy và cập nhật biên mới. Độ phức tạp: O(N).

📝 Code C++ — E10. Interval Cover 
// Bài 10: PHUDOAN — Sort by left, greedy extend
#include <bits/stdc++.h>
using namespace std;

bool cmp(pair<int,int> a, pair<int,int> b) {
    if (a.first != b.first) return a.first < b.first;
    return a.second > b.second;
}

int main() {
    ifstream fin("PHUDOAN.INP");
    ofstream fout("PHUDOAN.OUT");
    int A, B, n;
    fin >> A >> B >> n;
    vector<pair<int,int>> seg(n);
    for (int i = 0; i < n; i++) fin >> seg[i].first >> seg[i].second;
    sort(seg.begin(), seg.end(), cmp);
    int ans = 0, cur = A, idx = 0;
    while (cur < B) {
        int best = cur;
        while (idx < n && seg[idx].first <= cur) {
            best = max(best, seg[idx].second);
            idx++;
        }
        if (best == cur) {
            fout << -1;
            return 0;
        }
        cur = best;
        ans++;
    }
    fout << ans;
    return 0;
}
💡 Ý tưởng: Sắp xếp đoạn theo đầu trái tăng dần. Tham lam: tại mỗi bước, trong tất cả đoạn bắt đầu trước hoặc đúng vùng đã phủ, chọn đoạn kéo dài xa nhất sang phải. Nếu không tiến thêm được → không phủ kín. Độ phức tạp: O(N log N).

N em học sinh xếp hàng, em thứ i có điểm ri. Phát kẹo sao cho: mỗi em ít nhất 1 viên kẹo; em nào có điểm cao hơn bạn ngay bên cạnh (trái hoặc phải) phải được nhiều kẹo hơn bạn đó. Tìm tổng số kẹo ít nhất cần phát.

📝 Code C++ — E11. Two-Pass Greedy 
// Bài 11: PHATKEO — Duyệt 2 lượt trái→phải, phải→trái
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream fin("PHATKEO.INP");
    ofstream fout("PHATKEO.OUT");
    int n;
    fin >> n;
    vector<int> r(n);
    for (int i = 0; i < n; i++) fin >> r[i];
    vector<int> candy(n, 1);
    // Lượt 1: trái → phải
    for (int i = 1; i < n; i++)
        if (r[i] > r[i - 1]) candy[i] = candy[i - 1] + 1;
    // Lượt 2: phải → trái
    for (int i = n - 2; i >= 0; i--)
        if (r[i] > r[i + 1]) candy[i] = max(candy[i], candy[i + 1] + 1);
    long long ans = 0;
    for (int i = 0; i < n; i++) ans += candy[i];
    fout << ans;
    return 0;
}
💡 Ý tưởng: Duyệt trái→phải: nếu điểm cao hơn bạn bên trái thì tăng kẹo. Duyệt phải→trái: nếu điểm cao hơn bạn bên phải thì cập nhật max. Kết hợp hai lượt đảm bảo cả hai hướng. Độ phức tạp: O(N).

📝 Code C++ — E12. Custom Comparator 
// Bài 12: SOTOTHAT — Sắp xếp bằng bộ so sánh tùy chỉnh
#include <bits/stdc++.h>
using namespace std;

bool cmp(string a, string b) {
    return a + b > b + a;
}

int main() {
    ifstream fin("SOTOTHAT.INP");
    ofstream fout("SOTOTHAT.OUT");
    int n;
    fin >> n;
    vector<string> a(n);
    for (int i = 0; i < n; i++) {
        int x;
        fin >> x;
        a[i] = to_string(x);
    }
    sort(a.begin(), a.end(), cmp);
    // Trường hợp toàn số 0
    if (a[0] == "0") {
        fout << "0";
        return 0;
    }
    string result;
    for (int i = 0; i < n; i++) result += a[i];
    fout << result;
    return 0;
}
💡 Ý tưởng: Chuyển các số thành chuỗi. Sắp xếp bằng bộ so sánh: a đứng trước b nếu ghép ab > ba (so sánh chuỗi). Trường hợp đặc biệt: nếu số lớn nhất sau sắp xếp là "0" thì tất cả đều 0, in "0". Độ phức tạp: O(N log N).

📝 Code C++ — E13. Stack Greedy 
// Bài 13: XAUMIN — Ngăn xếp đơn điệu tăng
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream fin("XAUMIN.INP");
    ofstream fout("XAUMIN.OUT");
    string s;
    int k;
    fin >> s >> k;
    string result;
    for (int i = 0; i < (int)s.size(); i++) {
        // Bỏ đỉnh stack nếu nó lớn hơn ký tự hiện tại và còn được xóa
        while (k > 0 && !result.empty() && result.back() > s[i]) {
            result.pop_back();
            k--;
        }
        result.push_back(s[i]);
    }
    // Nếu K còn dư, xóa từ cuối
    while (k > 0) { result.pop_back(); k--; }
    fout << result;
    return 0;
}
💡 Ý tưởng: Dùng ngăn xếp (string) đơn điệu tăng. Duyệt từng ký tự: nếu đỉnh stack lớn hơn ký tự hiện tại và còn quota xóa, bỏ đỉnh stack. Cuối cùng nếu K còn dư thì xóa từ cuối. Độ phức tạp: O(N).

📝 Code C++ — E14. Earliest Deadline First 
// Bài 14: TREHAN — Sắp xếp theo hạn chót (EDF)
#include <bits/stdc++.h>
using namespace std;

bool cmp(pair<int,int> a, pair<int,int> b) {
    return a.second < b.second;
}

int main() {
    ifstream fin("TREHAN.INP");
    ofstream fout("TREHAN.OUT");
    int n;
    fin >> n;
    vector<pair<int,int>> jobs(n); // (duration, deadline)
    for (int i = 0; i < n; i++) fin >> jobs[i].first >> jobs[i].second;
    sort(jobs.begin(), jobs.end(), cmp);
    long long time = 0, maxLate = 0;
    for (int i = 0; i < n; i++) {
        time += jobs[i].first;
        long long late = time - jobs[i].second;
        if (late > maxLate) maxLate = late;
    }
    fout << maxLate;
    return 0;
}
💡 Ý tưởng: Sắp xếp công việc theo hạn chót tăng dần (Earliest Deadline First). Thực hiện tuần tự, tính thời gian trễ tích lũy. Đây là chiến lược tối ưu cho bài toán tối thiểu hóa độ trễ lớn nhất. Độ phức tạp: O(N log N).

📝 Code C++ — E15. DSU Scheduling 
// Bài 15: LICHLAMVIEC — Sort by profit + DSU tìm slot trống
#include <bits/stdc++.h>
using namespace std;

int par[100001];

int find(int x) {
    while (par[x] != x) {
        par[x] = par[par[x]]; // path halving
        x = par[x];
    }
    return x;
}

bool cmp(pair<int,int> a, pair<int,int> b) {
    return a.second > b.second;
}

int main() {
    ifstream fin("LICHLAMVIEC.INP");
    ofstream fout("LICHLAMVIEC.OUT");
    int n;
    fin >> n;
    vector<pair<int,int>> jobs(n); // (deadline, profit)
    int maxD = 0;
    for (int i = 0; i < n; i++) {
        fin >> jobs[i].first >> jobs[i].second;
        maxD = max(maxD, jobs[i].first);
    }
    for (int i = 0; i <= maxD; i++) par[i] = i;
    sort(jobs.begin(), jobs.end(), cmp);
    long long ans = 0;
    for (int i = 0; i < n; i++) {
        int slot = find(jobs[i].first);
        if (slot > 0) {
            ans += jobs[i].second;
            par[slot] = slot - 1;
        }
    }
    fout << ans;
    return 0;
}
💡 Ý tưởng: Sắp xếp công việc theo lợi nhuận giảm dần. Với mỗi việc, dùng DSU (Disjoint Set Union) tìm slot lớn nhất ≤ deadline còn trống. Nếu tìm được (slot > 0), gán việc và hợp slot với slot − 1. Kỹ thuật DSU giúp tìm slot trống trong O(α(N)). Tổng: O(N log N + N · α(N)).
