PHẦN 4: CÁC HÀM TỰ VIẾT HAY DÙNG

4.1. Lũy thừa nhanh (Modular Exponentiation)

ll power(ll a, ll n, ll mod) {
    ll res = 1;
    a %= mod;
    while (n > 0) {
        if (n & 1) res = res * a % mod;
        a = a * a % mod;
        n >>= 1;
    }
    return res;
}
// Dùng: power(2, 30, MOD) = 2^30 % MOD

4.2. Nghịch đảo modular (Modular Inverse)

// Cách 1: Dùng lũy thừa nhanh (mod phải là số nguyên tố)
ll modInverse(ll a, ll mod) {
    return power(a, mod - 2, mod);
}

// Cách 2: Extended GCD (tổng quát hơn)
ll extgcd(ll a, ll b, ll &x, ll &y) {
    if (b == 0) { x = 1; y = 0; return a; }
    ll x1, y1;
    ll g = extgcd(b, a % b, x1, y1);
    x = y1;
    y = x1 - (a / b) * y1;
    return g;
}

4.3. Tính tổ hợp C(n, k)

// Tính trước giai thừa
ll fact[MAXN], inv_fact[MAXN];

void precompute(int n) {
    fact[0] = 1;
    for (int i = 1; i <= n; i++)
        fact[i] = fact[i-1] * i % MOD;
    inv_fact[n] = power(fact[n], MOD - 2, MOD);
    for (int i = n - 1; i >= 0; i--)
        inv_fact[i] = inv_fact[i+1] * (i+1) % MOD;
}

ll C(int n, int k) {
    if (k < 0 || k > n) return 0;
    return fact[n] * inv_fact[k] % MOD * inv_fact[n-k] % MOD;
}

vector<pair<ll, int>> factorize(ll n) {
    vector<pair<ll, int>> factors;
    for (ll i = 2; i * i <= n; i++) {
        if (n % i == 0) {
            int cnt = 0;
            while (n % i == 0) { n /= i; cnt++; }
            factors.push_back({i, cnt});

    return factors;
}

4.8. DSU (Union-Find)

int par[MAXN], rnk[MAXN];

void init(int n) {
    for (int i = 1; i <= n; i++) { par[i] = i; rnk[i] = 0; }
}

int find(int x) {
    if (par[x] != x) par[x] = find(par[x]);  // Path compression
    return par[x];
}

void unite(int x, int y) {
    x = find(x); y = find(y);
    if (x == y) return;
    if (rnk[x] < rnk[y]) swap(x, y);  // Union by rank
    par[y] = x;
    if (rnk[x] == rnk[y]) rnk[x]++;
}

bool same(int x, int y) { return find(x) == find(y); }

4.9. Binary Indexed Tree (BIT / Fenwick Tree)

ll bit[MAXN];

void update(int i, ll delta) {
    for (; i <= n; i += i & (-i)) bit[i] += delta;

    ll sum = 0;

ll rangeQuery(int l, int r) {

4.10. LCA (Lowest Common Ancestor) - Binary Lifting

const int LOG = 20;
int par[MAXN][LOG], depth[MAXN];
vector<int> adj[MAXN];

void dfs(int u, int p, int d) {
    par[u][0] = p;
    depth[u] = d;
    for (int i = 1; i < LOG; i++)
        par[u][i] = par[par[u][i-1]][i-1];
    for (int v : adj[u])
        if (v != p) dfs(v, u, d + 1);
}

int lca(int u, int v) {
    if (depth[u] < depth[v]) swap(u, v);
    int diff = depth[u] - depth[v];
    for (int i = 0; i < LOG; i++)
        if ((diff >> i) & 1) u = par[u][i];
    if (u == v) return u;
    for (int i = LOG - 1; i >= 0; i--)
        if (par[u][i] != par[v][i]) {
            u = par[u][i]; v = par[v][i];
        }
    return par[u][0];
}

PHẦN 5: THUẬT TOÁN ĐỒ THỊ CƠ BẢN

5.1. BFS - Duyệt theo chiều rộng

int dist[MAXN];
bool visited[MAXN];

void bfs(int start) {
    queue<int> q;
    memset(dist, -1, sizeof(dist));
    dist[start] = 0;
    q.push(start);
    
    while (!q.empty()) {
        int u = q.front(); q.pop();
        for (int v : adj[u]) {
            if (dist[v] == -1) {
                dist[v] = dist[u] + 1;
                q.push(v);
            }
        }
    }
}

5.2. DFS - Duyệt theo chiều sâu

bool visited[MAXN];

void dfs(int u) {
    visited[u] = true;
    for (int v : adj[u]) {
        if (!visited[v]) dfs(v);
    }
}

// Đếm thành phần liên thông
int countComponents(int n) {
    int cnt = 0;
    for (int i = 1; i <= n; i++) {
        if (!visited[i]) {
            dfs(i);
            cnt++;
        }
    }
    return cnt;
}

5.3. Dijkstra - Đường đi ngắn nhất

vector<pair<int, ll>> adj[MAXN];  // {đỉnh kề, trọng số}
ll dist[MAXN];

void dijkstra(int start, int n) {
    fill(dist, dist + n + 1, INF_LL);
    priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<pair<ll, int>>> pq;
    
    dist[start] = 0;
    pq.push({0, start});
    
    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d > dist[u]) continue;
        
        for (auto [v, w] : adj[u]) {
            if (dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                pq.push({dist[v], v});
            }
        }
    }
}

5.4. Bellman-Ford - Đường đi ngắn nhất (có cạnh âm)

struct Edge { int u, v; ll w; };
vector<Edge> edges;
ll dist[MAXN];

bool bellmanFord(int start, int n) {
    fill(dist, dist + n + 1, INF_LL);
    dist[start] = 0;
    
    for (int i = 1; i < n; i++) {
        for (auto &e : edges) {
            if (dist[e.u] < INF_LL && dist[e.u] + e.w < dist[e.v]) {
                dist[e.v] = dist[e.u] + e.w;
            }
        }
    }
    
    // Kiểm tra chu trình âm
    for (auto &e : edges) {
        if (dist[e.u] < INF_LL && dist[e.u] + e.w < dist[e.v])
            return false;  // Có chu trình âm
    }
    return true;
}

5.5. Floyd-Warshall - Mọi cặp đỉnh

ll dist[MAXN][MAXN];  // Khởi tạo = INF_LL, dist[i][i] = 0

void floyd(int n) {
    for (int k = 1; k <= n; k++)
        for (int i = 1; i <= n; i++)
            for (int j = 1; j <= n; j++)
                if (dist[i][k] < INF_LL && dist[k][j] < INF_LL)
                    dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j]);
}

    return false;

// Kiểm tra a * b có tràn không  
bool willOverflowMul(ll a, ll b) {
    if (a == 0 || b == 0) return false;
    return abs(a) > LLONG_MAX / abs(b);
}

7.2. Các test case cần thử

  • N = 0, N = 1: Trường hợp nhỏ nhất
  • N = 2: Cặp đầu tiên
  • N = max: Giới hạn lớn nhất (kiểm tra TLE, tràn số)
  • Tất cả bằng nhau: a[i] = 1 với mọi i
  • Tất cả khác nhau: a[i] = i
  • Đã sắp xếp tăng/giảm
  • Số âm: Nếu đề cho phép
  • Không có đáp án: Kiểm tra output -1

7.3. Stress test nhanh

// Sinh test ngẫu nhiên và so sánh với brute force
void stress() {
    srand(time(0));
    for (int t = 0; t < 1000; t++) {
        int n = rand() % 100 + 1;
        vector<int> a(n);
        for (int &x : a) x = rand() % 1000;
        
        int ans1 = solve(a);      // Thuật toán của bạn
        int ans2 = bruteForce(a); // Thuật toán O(n²) chắc chắn đúng
        
        if (ans1 != ans2) {
            cout << "WRONG! n=" << n << endl;
            for (int x : a) cout << x << " ";
            cout << "\nExpected: " << ans2 << ", Got: " << ans1 << endl;
            return;
        }
    }
    cout << "All tests passed!" << endl;
}
