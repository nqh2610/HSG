1.2. TRÀN SỐ (Overflow) — Sai 50-70% test

Giới hạn kiểu số nguyên

Mẹo nhớ: int chứa được ~2 tỷ. Nếu kết quả có thể vượt 2 tỷ → dùng long long.

Modulo 10⁹+7 là gì?

Khi đề bài hỏi "có bao nhiêu cách?", kết quả có thể cực lớn (hàng trăm chữ số), vượt cả long long. Lúc đó đề sẽ yêu cầu: "In ra kết quả mod 10⁹+7".

10⁹+7 = 1,000,000,007 — là một số nguyên tố rất lớn, được chọn vì: hai số mod này nhân nhau vẫn nằm gọn trong long long.

const int MOD = 1e9 + 7;   // = 1000000007

// Cộng mod
ans = (a + b) % MOD;

// Nhân mod (ép long long trước khi nhân!)
ans = (1LL * a * b) % MOD;

// Trừ mod — phải cộng MOD tránh bị âm
// Vì (a - b) có thể ra số âm → % MOD vẫn âm → cộng MOD cho dương → % MOD lần nữa
ans = ((a - b) % MOD + MOD) % MOD;

A1. Prefix Sum — Tổng tiền tố

💡 Hình dung: Bạn có 1 cuốn sổ ghi "tổng tiền từ ngày 1 đến hôm nay" mỗi ngày. Khi sếp hỏi "tổng tiền từ ngày 3 đến ngày 7", thay vì cộng lại 5 ngày, bạn chỉ cần lấy: tổng_đến_ngày_7 − tổng_đến_ngày_2 = ra ngay! Tính trước 1 lần → trả lời bao nhiêu câu hỏi cũng chỉ cần 1 phép trừ.

DẤU HIỆU: "tổng đoạn [L,R]", "trung bình đoạn", "đoạn con có tổng = k" → truy vấn nhiều lần
// ═══ PREFIX SUM 1D ═══
// Build: O(n) | Query: O(1)
long long prefix[100005];
prefix[0] = 0;
for (int i = 1; i <= n; i++)
    prefix[i] = prefix[i-1] + a[i];

// Tổng đoạn [l, r]
long long tong = prefix[r] - prefix[l-1];

A2. Difference Array — Mảng hiệu (Cộng cả đoạn trong O(1))

💡 Hình dung: Bạn có 100 học sinh xếp hàng. Thầy giáo ra 10.000 lệnh kiểu "từ bạn thứ 3 đến bạn thứ 7, mỗi người thêm 5 điểm". Nếu cộng từng bạn một → quá lâu.
Mảng hiệu = "ghi chú thông minh": chỉ ghi "+5 ở vị trí 3" và "−5 ở vị trí 8" (sau đoạn). Cuối cùng Prefix Sum lại 1 lần → ra kết quả tất cả mọi người.

DẤU HIỆU: "Cập nhật cả đoạn [l, r] nhiều lần" rồi "hỏi giá trị cuối cùng". Là phép ngược của Prefix Sum.
Prefix Sum vs Mảng hiệu — Đi cặp với nhau:

Giỏi ởHỏi tổng đoạn [l,r] nhanhCộng/trừ cả đoạn [l,r] nhanh
Thao tácprefix[i] = prefix[i-1] + a[i]d[l] += val; d[r+1] -= val;

Ví dụ chạy tay: Mảng 5 phần tử, ban đầu = [0, 0, 0, 0, 0]

Lệnh 1: Cộng 3 vào đoạn [2, 4]   →  d[2] += 3, d[5] -= 3
Lệnh 2: Cộng 2 vào đoạn [1, 3]   →  d[1] += 2, d[4] -= 2

Mảng d[]:   [0, 2, 3, 0, -2, -3]
              ↓ Prefix Sum ↓
Kết quả:    [0, 2, 5, 5,  3,  0]
                   ↑     ↑
            Đoạn [2,4] được +3, đoạn [1,3] được +2 → vị trí 2,3 được +5

A3. Two Pointers — Hai con trỏ

💡 Hình dung: Hai bạn L và R đứng trên một hàng ghế. R đi sang phải để mở rộng đoạn, nếu đoạn vi phạm điều kiện thì L đi sang phải để thu hẹp. Cả hai chỉ đi về bên phải, không lùi → thay vì thử tất cả các cặp vị trí (rất lâu), ta chỉ cần mỗi người đi qua hàng ghế 1 lần → nhanh gấp nhiều lần.

DẤU HIỆU: "đoạn con dài nhất có tổng ≤ K", "cặp (i,j) thỏa a[i]+a[j] = k" (mảng đã sort), "đoạn ngắn nhất chứa đủ k giá trị"
// ═══ Tìm đoạn con dài nhất có tổng ≤ K ═══
// O(n) — L chỉ tiến, không lùi
int L = 0, maxLen = 0;
long long sum = 0;

for (int R = 0; R < n; R++) {
    sum += a[R];                    // Mở rộng phải
    while (sum > K && L <= R) {    // Thu hẹp trái nếu vi phạm
        sum -= a[L];
        L++;
    }
    maxLen = max(maxLen, R - L + 1);
}

// ═══ Đếm cặp tổng = K (mảng đã sort) ═══
sort(a, a + n);
int L = 0, R = n - 1, cnt = 0;
while (L < R) {
    int s = a[L] + a[R];
    if (s == K)      { cnt++; L++; R--; }
    else if (s < K)  L++;
    else             R--;
}

A4. Sliding Window — Cửa sổ trượt

💡 Hình dung: Bạn nhìn qua cửa sổ xe buýt, chỉ thấy đúng 3 ngôi nhà. Xe chạy tới — mất 1 nhà bên trái, thêm 1 bên phải. Muốn biết "nhà cao nhất đang thấy" mà không so lại cả 3 nhà mỗi lần → dùng deque giữ sẵn "ứng viên cao nhất".

DẤU HIỆU: "max/min trong cửa sổ kích thước k", "đoạn có max - min ≤ K"
Khác Two Pointers: Sliding Window thường duy trì thêm trạng thái (deque, map, tần suất) bên trong cửa sổ.
📝 VÍ DỤ CHẠY TAY: Max trong mỗi cửa sổ k=3, mảng [1, 3, 2, 5, 4] 
               Cửa sổ                Deque (ứng viên max)     Max

Bước 1:   [1   3   2]  5   4        [3, 2]                   → 3

Bước 2:    1  [3   2   5]  4        [5]                      → 5

Bước 3:    1   3  [2   5   4]       [5, 4]                   → 5

Kết quả: 3  5  5

Deque là gì? Là danh sách "ứng viên" đang đứng trong cửa sổ,
xếp từ lớn → nhỏ. Đầu danh sách luôn là MAX hiện tại.

Mỗi lần cửa sổ trượt sang phải 1 ô:
  1. Phần tử mới bước VÀO cửa sổ:
     → Nó lớn hơn ai trong deque → loại hết (thua chắc, không cần giữ)
     → Rồi nó vào cuối deque
  2. Phần tử cũ bước RA khỏi cửa sổ:
     → Nếu nó đang đứng đầu deque → loại (hết hạn)
  3. Đầu deque = MAX → in ra

// ═══ Max trong mỗi cửa sổ kích thước k ═══
// Dùng deque giảm dần — O(n)
deque<int> dq;  // Lưu chỉ số (vị trí), KHÔNG lưu giá trị
for (int i = 0; i < n; i++) {
    //1. Đầu deque ra khỏi cửa sổ rồi → bỏ
    // Cửa sổ kết thúc ở i, bắt đầu ở i-k+1
    // Nên vị trí ≤ i-k là ngoài cửa sổ
    while (!dq.empty() && dq.front() <= i - k)
        dq.pop_front();
    //2. Phần tử mới vào: đá hết ai nhỏ hơn nó
    while (!dq.empty() && a[dq.back()] <= a[i])
        dq.pop_back();
    dq.push_back(i);  // Cho phần tử mới vào cuối deque
    //3. Đầu deque = max cửa sổ hiện tại
    if (i >= k - 1)
        out << a[dq.front()] << " ";
}

A5. Monotonic Stack — Ngăn xếp đơn điệu

💡 Hình dung: Bạn đứng trong hàng người xếp hàng, nhìn về bên phải. Bạn chỉ thấy người đầu tiên CAO HƠN bạn — những người thấp hơn bị "che" mất. Mono Stack giúp tìm "người cao hơn gần nhất" cho mọi người trong hàng, chỉ với O(n).

DẤU HIỆU:
  • "Phần tử lớn hơn / nhỏ hơn gần nhất bên phải (hoặc trái)"
  • "Diện tích hình chữ nhật lớn nhất trong histogram"
  • "Khoảng cách đến ngày nóng hơn gần nhất"

Ví dụ: Cho mảng a = [2, 1, 5, 6, 2, 3]. Tìm phần tử lớn hơn gần nhất bên phải của mỗi phần tử.

Duyệt từ phải sang trái, giữ stack tăng dần (đáy → đỉnh):

  i=5: a[5]=3  stack=[]       → không có → ans=-1, push 3     stack=[3]
  i=4: a[4]=2  stack=[3]      → 3>2      → ans=3,  push 2     stack=[3,2]
  i=3: a[3]=6  pop 2, pop 3   → hết      → ans=-1, push 6     stack=[6]
  i=2: a[2]=5  stack=[6]      → 6>5      → ans=6,  push 5     stack=[6,5]
  i=1: a[1]=1  stack=[6,5]    → 5>1      → ans=5,  push 1     stack=[6,5,1]
  i=0: a[0]=2  pop 1          → 5>2      → ans=5,  push 2     stack=[6,5,2]

Kết quả: [5, 5, 6, -1, 3, -1]

Code mẫu — Phần tử lớn hơn gần nhất bên phải:

// Mono Stack — O(n)
// Mỗi phần tử chỉ vào/ra stack đúng 1 lần → tổng O(n)
stack<int> st;          // Lưu giá trị (hoặc chỉ số tuỳ bài)
int ans[100005];

for (int i = n - 1; i >= 0; i--) {
    // Loại phần tử nhỏ hơn hoặc bằng — chúng bị "che" bởi a[i]
    while (!st.empty() && st.top() <= a[i])
        st.pop();

    // Đỉnh stack = phần tử lớn hơn gần nhất bên phải
    ans[i] = st.empty() ? -1 : st.top();

    st.push(a[i]);
}

A6. Binary Search — Tìm kiếm nhị phân

💡 Hình dung trò chơi "Đoán số": bạn nghĩ 1 số từ 1 đến 1.000.000, bạn đoán và được trả lời "lớn hơn" hoặc "nhỏ hơn". Thay vì đoán từ 1, 2, 3... bạn luôn đoán SỐ GIỮA → mỗi lần loại bỏ NỬA số còn lại. 1 triệu số → chỉ cần ~20 lần đoán!
"Chặt nhị phân đáp án" cùng ý tưởng: thử đáp án ở giữa, kiểm tra có được không, rồi thu hẹp phạm vi.

DẤU HIỆU:
  • "tìm x trong dãy đã sort" → lower_bound, upper_bound
  • "đếm phần tử trong [L, R]" → upper_bound(R) - lower_bound(L)
  • "tìm giá trị nhỏ nhất/lớn nhất thỏa..." → Chặt nhị phân đáp án (RẤT PHỔ BIẾN!)
// ═══ Hàm có sẵn (mảng PHẢI sort trước) ═══
sort(a, a + n);

int pos1 = lower_bound(a, a+n, x) - a;  // Vị trí đầu tiên >= x
int pos2 = upper_bound(a, a+n, x) - a;  // Vị trí đầu tiên > x
int cntX = pos2 - pos1;                  // Số phần tử = x
int cntLR = upper_bound(a,a+n,R) - lower_bound(a,a+n,L);  // Trong [L,R]

// ═══ CHẶT NHỊ PHÂN ĐÁP ÁN ═══
// "Tìm giá trị nhỏ nhất sao cho check(mid) = true"
int lo = MIN_ANS, hi = MAX_ANS, ans = -1;
while (lo <= hi) {
    int mid = lo + (hi - lo) / 2;
    if (check(mid)) {   // Hàm kiểm tra mid có thỏa không
        ans = mid;
        hi = mid - 1;   // Tìm nhỏ hơn
    } else {
        lo = mid + 1;
    }
}

A9. Kadane — Đoạn con tổng lớn nhất

💡 Hình dung: Bạn đi dọc dãy số, tay cầm một cái túi. Mỗi lần gặp số mới, bạn quyết định: "Bỏ vào túi cũ" (cộng thêm) hay "Vứt túi cũ, bắt đầu túi mới từ số này". Nếu túi cũ đang âm (tổng < 0) thì vứt đi cho rồi! Luôn ghi nhớ túi nặng nhất từng có. Chỉ đi 1 lượt là xong.

long long maxSum = a[0], cur = a[0];
for (int i = 1; i < n; i++) {

}

Dạng 4: Quay lui (Backtracking) — Đệ quy + cắt nhánh

💡 Hình dung: Đi trong mê cung — mỗi ngã rẽ bạn chọn 1 hướng. Nếu đi vào ngõ cụt → quay lui, thử hướng khác. Nếu phát hiện sớm "đi tiếp cũng không được" → cắt luôn, không cần thử (= cắt nhánh).

DẤU HIỆU: "liệt kê tất cả cách chọn k phần tử từ n" (tổ hợp), "tìm đường đi", "đặt N quân hậu". Mạnh hơn brute force nhờ cắt nhánh sớm.
// Mẫu quay lui tổng quát
int chon[25];       // Lưu các phần tử đã chọn
bool used[25];      // Đánh dấu đã dùng chưa

void Try(int k) {   // Đang chọn vị trí thứ k
    if (k > m) {     // Đã chọn đủ m phần tử → xử lý
        xuLy();
        return;
    }
    for (int i = 1; i <= n; i++) {
        if (!used[i]) {
            // CẮT NHÁNH: bỏ qua nếu biết chắc không tối ưu
            if (khongTheToiUu()) continue;

            chon[k] = i;
            used[i] = true;
            Try(k + 1);        // Đệ quy tiếp
            used[i] = false;   // QUAY LUI — hoàn tác để thử cách khác
        }
    }
}

Ví dụ: Liệt kê tổ hợp chập k của n (chọn k từ n phần tử, không quan tâm thứ tự):

int chon[25];

void Try(int pos, int start) {  // pos: vị trí đang chọn, start: bắt đầu từ đâu
    if (pos > k) {
        for (int i = 1; i <= k; i++) fout << chon[i] << " ";
        fout << "\n";
        return;
    }
    // CẮT NHÁNH: còn lại không đủ phần tử để chọn
    if (n - start + 1 < k - pos + 1) return;

    for (int i = start; i <= n; i++) {
        chon[pos] = i;
        Try(pos + 1, i + 1);   // i+1 vì tổ hợp, không lấy lại
    }
}
// Gọi: Try(1, 1);
Mẹo cắt nhánh thường gặp:
  • Tổng hiện tại đã vượt K → không cần thử thêm
  • Số phần tử còn lại không đủ để hoàn thành → bỏ
  • Kết quả hiện tại đã tệ hơn đáp án tốt nhất → bỏ

Dạng 5: Meet in the Middle — Gặp nhau ở giữa

💡 Hình dung: Bạn có 36 món đồ, cần tìm cách chọn sao cho tổng = K. Brute force 2³⁶ ≈ 7×10¹⁰ → quá lâu. Nhưng chia đôi thành 2 nửa 18 món → mỗi nửa chỉ 2¹⁸ ≈ 260.000 → duyệt được! Rồi ghép: tìm trong nửa B tổng = K − (tổng nửa A) bằng binary search.

DẤU HIỆU:
  • “Tập con có tổng = K” với N ≈ 30-40 (quá lớn cho bitmask 2ᴺ, nhưng 2^(N/2) được)
  • “Đếm cặp tập con thỏa điều kiện”

3 bước:

Bước 1: Chia mảng thành 2 nửa — A[0..n/2-1] và B[n/2..n-1]
Bước 2: Duyệt tất cả tập con mỗi nửa → lưu tổng vào 2 danh sách
Bước 3: Sort 1 danh sách, dùng binary search ghép kết quả

Ví dụ: Đếm số tập con có tổng = K

// N ≤ 40 → chia đôi, mỗi nửa ≤ 20
int half = n / 2;
vector<long long> sumA, sumB;

// Bước 1-2: Sinh tổng tất cả tập con nửa trái
for (int mask = 0; mask < (1 << half); mask++) {
    long long s = 0;
    for (int i = 0; i < half; i++)
        if (mask & (1 << i)) s += a[i];
    sumA.push_back(s);
}

// Sinh tổng tất cả tập con nửa phải
int half2 = n - half;
for (int mask = 0; mask < (1 << half2); mask++) {
    long long s = 0;
    for (int i = 0; i < half2; i++)
        if (mask & (1 << i)) s += a[half + i];
    sumB.push_back(s);
}

// Bước 3: Sort nửa B, duyệt nửa A + binary search
sort(sumB.begin(), sumB.end());
long long ans = 0;
for (long long sa : sumA) {
    long long need = K - sa;
    // Đếm số phần tử trong sumB = need
    ans += upper_bound(sumB.begin(), sumB.end(), need)
         - lower_bound(sumB.begin(), sumB.end(), need);
}
Độ phức tạp: O(2^(N/2) × log) — với N=40 thì ≈ 2²⁰ × 20 ≈ 20 triệu phép tính → chạy tốt.

🔑 KINH NGHIỆM TÌM CÔNG THỨC DP (Bước 3)

Bước 3 là bước khó nhất. Dưới đây là các mẹo giúp bạn tìm công thức nhanh:

Mẹo 1 — "Bước cuối cùng":
Đứng ở phần tử thứ i, hỏi: "Ở bước cuối, ta có mấy lựa chọn?"
Mỗi lựa chọn sẽ tương ứng với 1 nhánh trong công thức. 
VD Balo: Với món đồ thứ i, ta có 2 lựa chọn:
  → Không chọn:  dp[i][w] = dp[i-1][w]
  → Chọn:        dp[i][w] = dp[i-1][w - weight[i]] + value[i]
  ⟹ dp[i][w] = max(không chọn, chọn)

VD Leo cầu thang (mỗi bước leo 1 hoặc 2 bậc):
  → Leo 1 bậc từ bậc i-1:  dp[i-1] cách
  → Leo 2 bậc từ bậc i-2:  dp[i-2] cách
  ⟹ dp[i] = dp[i-1] + dp[i-2]
Mẹo 2 — Thử bằng tay n nhỏ:
Tính tay n=1, n=2, n=3, n=4. Nhìn vào dãy kết quả → tìm quy luật. 
VD Lát gạch 2×n: n=1→1, n=2→2, n=3→3, n=4→5
  Nhận ra: 1, 2, 3, 5... → dp[i] = dp[i-1] + dp[i-2]
Mẹo 3 — 1D hay 2D?
  • Bài chỉ có 1 ràng buộc (vị trí i) → dp[i] 1 chiều
  • Bài có 2 ràng buộc (vd: vật thứ i + trọng lượng w) → dp[i][w] 2 chiều
  • 2D có thể tối ưu thành 1D bằng cách duyệt ngược (xem balo 0/1 bên dưới)
Mẹo 4 — Phân biệt "đếm" và "tối ưu":
  • "Đếm số cách" → cộng dồn: dp[i] += dp[j]
  • "Tìm max/min" → so sánh: dp[i] = max(dp[i], dp[j] + ...)

Ví dụ 1: Fibonacci — Bài DP đầu tiên

Tìm số Fibonacci thứ n. Dãy: 0, 1, 1, 2, 3, 5, 8, 13, 21...

Bước 1: dp[i] = số Fibonacci thứ i
Bước 2: dp[0] = 0, dp[1] = 1
Bước 3: dp[i] = dp[i-1] + dp[i-2]   ← "bước cuối": cộng 2 số trước
Bước 4: Đáp án = dp[n]

int fibo(int n) {
    vector<int> dp(n + 1);
    dp[0] = 0; dp[1] = 1;                   // Bước 2: Biên
    for (int i = 2; i <= n; i++)
        dp[i] = dp[i-1] + dp[i-2];          // Bước 3: Công thức
    return dp[n];                            // Bước 4: Đáp án
}

Ví dụ 2: Balo 0/1 — Mỗi vật dùng 1 lần

n món đồ, mỗi món có trọng lượng w[i] và giá trị v[i]. Balo chứa tối đa W. Chọn sao cho tổng giá trị lớn nhất.

Bước 1: dp[i][w] = giá trị lớn nhất khi xét i món đầu, balo chứa w
Bước 2: dp[0][w] = 0  (0 món → giá trị 0)
Bước 3: "Bước cuối" — món thứ i có 2 lựa chọn:
  → Không chọn: dp[i][w] = dp[i-1][w]
  → Chọn (nếu w[i] ≤ w): dp[i][w] = dp[i-1][w-w[i]] + v[i]
  ⟹ dp[i][w] = max(không chọn, chọn)
Bước 4: Đáp án = dp[n][W]

Code 2D (dễ hiểu):

// Bước 1: dp[i][w] = giá trị max xét i món, balo chứa w
vector<vector<int>> dp(n+1, vector<int>(W+1, 0));
// Bước 2: dp[0][w] = 0 (đã khởi tạo)
// Bước 3: Công thức chuyển trạng thái
for (int i = 1; i <= n; i++)
    for (int w = 0; w <= W; w++) {
        dp[i][w] = dp[i-1][w];              // Không chọn món i
        if (weights[i-1] <= w)
            dp[i][w] = max(dp[i][w],
                dp[i-1][w - weights[i-1]] + values[i-1]); // Chọn
    }
// Bước 4: Đáp án
cout << dp[n][W];

Code 1D tối ưu (duyệt NGƯỢC — mỗi vật chỉ dùng 1 lần):

long long dp[MAX_W + 1] = {0};

// Đáp án: dp[W]

Truy vết — In ra các món đã chọn (từ bảng 2D):

// Bước 4: Truy ngược từ dp[n][W]

Ví dụ 3: Đổi tiền — Mỗi loại dùng không giới hạn

Có các mệnh giá coins[] = {1, 2, 5}. Đếm số cách đổi amount = 5.

Bước 1: dp[w] = số cách đổi số tiền w
Bước 2: dp[0] = 1  (1 cách đổi 0 đồng: không chọn gì)
Bước 3: dp[w] += dp[w - coin]  ← "nếu dùng đồng coin, còn lại w-coin đã có dp[w-coin] cách"
Bước 4: Đáp án = dp[amount]

long long soCach[MAX_SUM + 1] = {0};
soCach[0] = 1;                               // Bước 2: Biên
for (int coin : coins)                        // Duyệt từng mệnh giá
    for (int j = coin; j <= S; j++)          // ← XUÔI! (dùng không giới hạn)
        soCach[j] += soCach[j - coin];        // Bước 3: Cộng dồn số cách
// Đáp án: soCach[S]

// ═══ DP ĐẾM SỐ CÁCH ĐẠT TỔNG ═══
// Dùng các giá trị 1..k đạt tổng n
soCach[0] = 1;
for (int val = 1; val <= k; val++)
    for (int t = val; t <= n; t++)
        soCach[t] += soCach[t - val];

Ví dụ 4: LIS — Dãy con tăng dài nhất

💡 Hình dung: Có dãy học sinh xếp hàng lộn xộn: 3 1 4 1 5 9. Bạn muốn chọn ra nhiều bạn nhất sao cho chiều cao TĂNG DẦN (không cần đứng cạnh nhau). Chọn: 1, 4, 5, 9 → được 4 bạn.

DẤU HIỆU: "dãy con tăng dài nhất", "chọn nhiều nhất sao cho tăng dần"
Bước 1: dp[i] = độ dài dãy con tăng dài nhất KẾT THÚC tại vị trí i
        trace[i] = vị trí phần tử đứng trước a[i] trong LIS (để truy vết)
Bước 2: dp[i] = 1 (mỗi phần tử tự nó là dãy dài 1), trace[i] = -1
Bước 3: "Bước cuối" — phần tử a[i] có thể nối sau a[j] nào?
  → Duyệt j < i: nếu a[j] < a[i] và dp[j]+1 > dp[i]
  ⟹ dp[i] = dp[j] + 1, trace[i] = j
Bước 4: Đáp án = max(dp[0..n-1]). Truy vết theo trace[] từ vị trí đáp án.

LIS — O(n²) dễ hiểu (dùng khi N ≤ 5000):

// Bước 1: dp[i] = độ dài LIS kết thúc tại i
int dp[5005], trace[5005];
int ans = 1, lastIdx = 0;
for (int i = 0; i < n; i++) {
    dp[i] = 1; trace[i] = -1;               // Bước 2: Biên
    for (int j = 0; j < i; j++)
        if (a[j] < a[i] && dp[j] + 1 > dp[i]) {
            dp[i] = dp[j] + 1;              // Bước 3: Công thức
            trace[i] = j;                   // Lưu vết
        }
    if (dp[i] > ans) { ans = dp[i]; lastIdx = i; }
}

// Bước 4: Truy vết
vector<int> lis;
for (int i = lastIdx; i != -1; i = trace[i])
    lis.push_back(a[i]);
reverse(lis.begin(), lis.end());

LIS — O(n log n) tối ưu dùng Binary Search:

// dp[i] = phần tử nhỏ nhất kết thúc dãy tăng dài i+1
vector<int> dp;
for (int i = 0; i < n; i++) {
    int pos = lower_bound(dp.begin(), dp.end(), a[i]) - dp.begin();
    if (pos == (int)dp.size())
        dp.push_back(a[i]);                  // Mở rộng dãy
    else
        dp[pos] = a[i];                      // Thay thế để giữ dãy "tốt" hơn
}
int LIS = dp.size();

B4. Divide and Conquer — Chia để trị

💡 Hình dung: Bạn có chồng 1000 bài thi cần xếp theo điểm. Chia đôi thành 2 chồng 500, mỗi chồng chia đôi tiếp... đến khi 1 tờ thì đã "có thứ tự". Rồi gộp 2 chồng đã sắp thành 1 chồng sắp → nhanh hơn xếp cả 1000 tờ cùng lúc.

DẤU HIỆU:
  • Bài toán có thể chia thành 2 nửa giống nhau, giải riêng, rồi gộp kết quả
  • "Đếm nghịch thế", "đếm cặp (i,j) thỏa điều kiện" (dùng merge sort biến đổi)
  • "Tìm cặp điểm gần nhất" (closest pair)

Mẫu chung 3 bước:

void solve(int l, int r) {
    if (l >= r) return;          // Bước 1: Cơ sở — 1 phần tử đã "xong"

    int mid = (l + r) / 2;
    solve(l, mid);               // Bước 2: Giải nửa trái
    solve(mid + 1, r);           //         Giải nửa phải

    merge(l, mid, r);            // Bước 3: GỘP kết quả 2 nửa
}

Ví dụ: Đếm nghịch thế (số cặp i < j mà a[i] > a[j]) — dùng Merge Sort biến đổi:

long long cnt = 0;
int temp[100005];

void mergeCount(int a[], int l, int mid, int r) {
    int i = l, j = mid + 1, k = l;
    while (i <= mid && j <= r) {
        if (a[i] <= a[j])
            temp[k++] = a[i++];
        else {
            temp[k++] = a[j++];
            cnt += mid - i + 1;  // Tất cả a[i..mid] > a[j] → đều là nghịch thế
        }
    }
    while (i <= mid) temp[k++] = a[i++];
    while (j <= r)   temp[k++] = a[j++];
    for (int t = l; t <= r; t++) a[t] = temp[t];
}

void solve(int a[], int l, int r) {
    if (l >= r) return;
    int mid = (l + r) / 2;
    solve(a, l, mid);
    solve(a, mid + 1, r);
    mergeCount(a, l, mid, r);
}
Độ phức tạp: O(N log N) — vì mỗi tầng chia đôi xử lý O(N), có log N tầng.

B5. Contribution Technique — Kỹ thuật đóng góp

💡 Hình dung: Thầy giáo hỏi "tổng điểm của TẤT CẢ các nhóm có thể chọn từ lớp". Thay vì liệt kê mọi nhóm (cực nhiều) → hỏi TỪNG học sinh: "bạn xuất hiện trong bao nhiêu nhóm?" rồi nhân điểm × số nhóm.
Đảo góc nhìn: không duyệt đoạn → duyệt phần tử, tính mỗi phần tử đóng góp vào bao nhiêu đoạn.

DẤU HIỆU:
  • "Tổng ... của tất cả đoạn con" (tổng min, tổng max, tổng XOR...)
  • "Tổng |a[i] - a[j]| với mọi cặp (i,j)"
  • Duyệt mọi tập con quá lâu → tính đóng góp từng phần tử

Ví dụ 1: Tổng các phần tử của tất cả đoạn con liên tiếp.

Mảng: [2, 5, 3]    (chỉ số 1..3)

Liệt kê mọi đoạn con:
  [2] [5] [3] [2,5] [5,3] [2,5,3]
Tổng = 2 + 5 + 3 + 7 + 8 + 10 = 35

Cách đóng góp — a[i] nằm trong bao nhiêu đoạn con?
  Đoạn chứa vị trí i: chọn trái từ 1..i, phải từ i..n
  → Số đoạn chứa i = i × (n - i + 1)

  a[1]=2: 1 × 3 = 3 đoạn → đóng góp 2 × 3 = 6
  a[2]=5: 2 × 2 = 4 đoạn → đóng góp 5 × 4 = 20
  a[3]=3: 3 × 1 = 3 đoạn → đóng góp 3 × 3 = 9
  Tổng = 6 + 20 + 9 = 35 ✓

Code mẫu — Tổng mọi đoạn con:

long long ans = 0;
for (int i = 1; i <= n; i++) {
    long long soDoạn = (long long)i * (n - i + 1);
    ans += a[i] * soDoạn;
}

Ví dụ 2: Tổng |a[i] - a[j]| với mọi cặp i < j → sort trước, rồi mỗi a[i] đóng góp:

sort(a + 1, a + n + 1);
long long ans = 0;
for (int i = 1; i <= n; i++) {
    // a[i] lớn hơn (i-1) phần tử trước nó
    // → đóng góp +a[i] × (i-1) lần, -a[i] × (n-i) lần
    ans += (long long)a[i] * (2 * i - n - 1);
}
Công thức nhớ: Phần tử ở vị trí i (1-indexed) nằm trong i × (n − i + 1) đoạn con liên tiếp.

B6. Digit DP — Đếm số trong đoạn [L, R]

💡 Hình dung: Bạn đang đặt mật khẩu gồm 9 chữ số, mỗi ô chọn 0–9. Đề bài hỏi "có bao nhiêu mật khẩu từ 000000001 đến 999999999 mà tổng các chữ số chia hết cho 3?". Duyệt cả tỷ mật khẩu thì chết — nhưng bạn nhận ra: khi điền từng ô từ trái qua phải, mỗi lần chọn chữ số, bạn chỉ cần nhớ "tổng hiện tại mod 3 đang là bao nhiêu" — không cần nhớ toàn bộ mật khẩu. Đó chính là Digit DP: xây số từng chữ số, mỗi bước chỉ lưu trạng thái cần thiết.

DẤU HIỆU:
  • "Đếm số trong [L, R] thỏa điều kiện ..."
  • "Có bao nhiêu số không quá N mà ..."
  • Điều kiện liên quan đến các chữ số: tổng chữ số, chữ số lặp, chữ số tăng/giảm...
NGUYÊN TẮC CHÍNH:
  • Tách: count(L, R) = count(R) − count(L − 1)
  • Xây từng chữ số: Đệ quy từ chữ số đầu tiên → chữ số cuối
  • Biến tight: Nếu các chữ số trước = chữ số tương ứng của N → chữ số hiện tại ≤ chữ số tương ứng. Nếu đã nhỏ hơn → tự do chọn 0–9
  • Memo: Lưu dp[pos][state][tight]

Khuôn mẫu: Đếm số từ 1..N có tổng chữ số = S

string num;           // N dạng chuỗi
int memo[20][200][2]; // [vị trí][tổng hiện tại][tight]

// pos: chữ số đang xét, sum: tổng chữ số đã chọn
// tight: có bị giới hạn bởi N không
int solve(int pos, int sum, bool tight) {
    if (pos == (int)num.size())
        return (sum == S) ? 1 : 0;          // Đã xây xong
    if (memo[pos][sum][tight] != -1)
        return memo[pos][sum][tight];

    int lim = tight ? (num[pos] - '0') : 9; // Giới hạn chữ số
    int res = 0;
    for (int d = 0; d <= lim; d++)
        res += solve(pos + 1, sum + d, tight && (d == lim));

    return memo[pos][sum][tight] = res;
}

// Đếm số trong [1, N] có tổng chữ số = S
int countUpTo(long long N) {
    num = to_string(N);
    memset(memo, -1, sizeof(memo));
    return solve(0, 0, true);
}

B7. Mo's Algorithm — Truy vấn offline trên đoạn

💡 Hình dung: Giáo viên cho 100 câu hỏi: "Từ bạn thứ L đến bạn thứ R trong lớp, có bao nhiêu họ khác nhau?". Nếu trả lời lung tung (hỏi đoạn [3,50] rồi nhảy sang [80,90] rồi quay lại [5,45]) thì mỗi lần phải đếm lại từ đầu. Nhưng nếu bạn sắp xếp lại thứ tự câu hỏi cho gần nhau ([3,50] → [5,45] → [3,48]...) thì mỗi lần chỉ cần thêm/bớt 1–2 bạn ở rìa là xong — không đếm lại! Mo's Algorithm chính là cách sắp xếp câu hỏi thông minh để "trượt" đoạn [L,R] ít nhất có thể.

DẤU HIỆU:
  • Nhiều truy vấn dạng [L, R] — offline (có thể trả lời không theo thứ tự)
  • Có thể thêm/bớt 1 phần tử vào kết quả nhanh (O(1) hoặc O(log))
  • N, Q ≤ 10⁵ và không có cập nhật
Ý TƯỞNG:
  1. Chia mảng thành √N khối (block)
  2. Sắp câu hỏi: cùng khối L → tăng theo R
  3. Duy trì kết quả đoạn hiện tại, mỗi bước thêm/xóa 1 phần tử
Độ phức tạp: O((N + Q) × √N) 
// ═══ Mo's Algorithm — Đếm số phần tử khác nhau trong đoạn [L,R] ═══
int BLOCK;
struct Query {
    int l, r, idx;
};

int cnt[1000005];   // cnt[v] = số lần v xuất hiện trong đoạn hiện tại
int curAns = 0;     // Số phần tử khác nhau hiện tại

void add(int val) {
    if (cnt[val] == 0) curAns++;
    cnt[val]++;
}

void remove(int val) {
    cnt[val]--;
    if (cnt[val] == 0) curAns--;
}

// Trong main:
BLOCK = max(1, (int)sqrt(n));
sort(queries, queries + q, [](const Query &a, const Query &b) {
    if (a.l / BLOCK != b.l / BLOCK) return a.l / BLOCK < b.l / BLOCK;
    return (a.l / BLOCK & 1) ? a.r > b.r : a.r < b.r; // Tối ưu zigzag
});

int curL = 0, curR = -1;
for (int i = 0; i < q; i++) {
    while (curR < queries[i].r) add(a[++curR]);
    while (curL > queries[i].l) add(a[--curL]);
    while (curR > queries[i].r) remove(a[curR--]);
    while (curL < queries[i].l) remove(a[curL++]);
    ans[queries[i].idx] = curAns;
}

💡 Hình dung: Bạn cần tính Fibonacci thứ 10¹⁸. Tính từng bước dp[1], dp[2], ..., dp[10¹⁸] thì đợi cả trăm năm không xong. Bước 1: Viết hệ truy hồi thành dạng ma trận

Suy ra:

Bước 2: Code nhân ma trận + lũy thừa nhanh

typedef vector<vector<long long>> Matrix;
const long long MOD = 1e9 + 7;

Matrix multiply(const Matrix &A, const Matrix &B) {
    int n = A.size();
    Matrix C(n, vector<long long>(n, 0));
    for (int i = 0; i < n; i++)
        for (int k = 0; k < n; k++) if (A[i][k])
            for (int j = 0; j < n; j++)
                C[i][j] = (C[i][j] + A[i][k] * B[k][j]) % MOD;
    return C;
}

Matrix matpow(Matrix A, long long p) {
    int n = A.size();
    Matrix res(n, vector<long long>(n, 0));
    for (int i = 0; i < n; i++) res[i][i] = 1;  // Ma trận đơn vị
    while (p > 0) {
        if (p & 1) res = multiply(res, A);
        A = multiply(A, A);
        p >>= 1;
    }
    return res;
}

// Fibonacci thứ n (n ≤ 10¹⁸):
long long fib(long long n) {
    if (n <= 1) return n;
    Matrix A = {{1, 1}, {1, 0}};
    Matrix R = matpow(A, n - 1);
    return R[0][0];                   // f(n) = R[0][0]*f(1) + R[0][1]*f(0)
}

C3. DSU (Disjoint Set Union) — Gộp nhóm

💡 Hình dung: Lớp có 40 bạn, ban đầu ai cũng là 1 nhóm riêng (40 nhóm). Khi 2 bạn kết bạn, nhóm của 2 bạn GỘP LẠI thành 1. Sau nhiều lần gộp, muốn hỏi "A và B có cùng nhóm không?" → chỉ cần xem 2 người có chung "Nhóm trưởng" không. Gộp nhóm + hỏi nhóm đều cực nhanh.

DẤU HIỆU: "cùng nhóm không?", "bao nhiêu nhóm/thành phần liên thông?", "gộp 2 tập hợp", Kruskal
// ═══ DSU — Phong cách quen thuộc ═══

const int MAXN = 100005;
int cha[MAXN];                // cha[i] = cha của đỉnh i
int hang[MAXN];               // hang[i] = chiều cao cây gốc i

void khoiTao(int n) {
    for (int i = 1; i <= n; i++) {
        cha[i] = i;            // Ban đầu mỗi đỉnh là gốc của chính nó
        hang[i] = 0;
    }
}

int tim_goc(int v) {
    if (v == cha[v]) return v;
    return cha[v] = tim_goc(cha[v]);  // Nén đường đi — RẤT QUAN TRỌNG!
}

void hop_nhat(int a, int b) {
    a = tim_goc(a);
    b = tim_goc(b);
    if (a == b) return;               // Đã cùng nhóm rồi
    // Union by rank: gốc nhỏ treo vào gốc lớn
    if (hang[a] < hang[b]) swap(a, b);
    cha[b] = a;
    if (hang[a] == hang[b]) hang[a]++;
}

bool cungNhom(int u, int v) {
    return tim_goc(u) == tim_goc(v);
}

// ═══ Ứng dụng: Đếm số nhóm / Tổng diện tích nhóm lớn nhất ═══

// Đếm số nhóm (thành phần liên thông)
int soNhom = 0;
for (int i = 1; i <= n; i++)
    if (cha[i] == i) soNhom++;

// Tổng giá trị nhóm lớn nhất (ví dụ: bài LATGACH)
long long tongDienTich[MAXN];
// Khi hop_nhat: tongDienTich[a] += tongDienTich[b];
long long maxTong = 0;
for (int i = 1; i <= n; i++)
    if (cha[i] == i)
        maxTong = max(maxTong, tongDienTich[i]);

    long long s = 0;
    for (; i > 0; i -= i & (-i))
        s += bit[i];
    return s;
}

Segment Tree cơ bản — Truy vấn tổng + Cập nhật điểm:

int st[400005]; // Kích thước 4 × n
int a[100005], n;

void build(int id, int l, int r) {
    if (l == r) { st[id] = a[l]; return; }
    int mid = (l + r) / 2;
    build(2*id, l, mid);
    build(2*id+1, mid+1, r);
    st[id] = st[2*id] + st[2*id+1];         // Đổi thành min/max/GCD tùy bài
}

void update(int id, int l, int r, int pos, int val) {
    if (l == r) { st[id] = val; return; }
    int mid = (l + r) / 2;
    if (pos <= mid) update(2*id, l, mid, pos, val);
    else update(2*id+1, mid+1, r, pos, val);
    st[id] = st[2*id] + st[2*id+1];
}

int query(int id, int l, int r, int ql, int qr) {
    if (ql > r || qr < l) return 0;         // Ngoài đoạn → giá trị trung tính
    if (ql <= l && r <= qr) return st[id]; // Nằm trọn trong
    int mid = (l + r) / 2;
    return query(2*id, l, mid, ql, qr)
         + query(2*id+1, mid+1, r, ql, qr);
}

// Gọi: build(1, 1, n);  update(1, 1, n, pos, val);  query(1, 1, n, L, R);

Segment Tree + Lazy Propagation — Cập nhật cả đoạn:

long long st[400005], lazy[400005];

void push_down(int id, int l, int r) {
    if (lazy[id]) {
        int mid = (l + r) / 2;
        st[2*id] += lazy[id] * (mid - l + 1);
        st[2*id+1] += lazy[id] * (r - mid);
        lazy[2*id] += lazy[id];
        lazy[2*id+1] += lazy[id];
        lazy[id] = 0;
    }
}

void updateRange(int id, int l, int r, int ql, int qr, long long val) {
    if (ql > r || qr < l) return;
    if (ql <= l && r <= qr) {
        st[id] += val * (r - l + 1);
        lazy[id] += val;
        return;
    }
    push_down(id, l, r);
    int mid = (l + r) / 2;
    updateRange(2*id, l, mid, ql, qr, val);
    updateRange(2*id+1, mid+1, r, ql, qr, val);
    st[id] = st[2*id] + st[2*id+1];
}

long long queryRange(int id, int l, int r, int ql, int qr) {
    if (ql > r || qr < l) return 0;
    if (ql <= l && r <= qr) return st[id];
    push_down(id, l, r);
    int mid = (l + r) / 2;
    return queryRange(2*id, l, mid, ql, qr)
         + queryRange(2*id+1, mid+1, r, ql, qr);
}

C7. Sparse Table — Truy vấn tĩnh O(1)

💡 Hình dung: Bạn có bảng điểm cố định (không ai sửa), nhưng thầy hỏi rất nhiều lần "ai điểm thấp nhất từ bạn L đến bạn R?". // ═══ Sparse Table — RMQ (Range Minimum Query) O(1) ═══
int sp[100005][17]; // sp[i][j] = min đoạn dài 2^j bắt đầu từ i
int LOG[100005];

void build(int n) {
    LOG[1] = 0;
    for (int i = 2; i <= n; i++)
        LOG[i] = LOG[i/2] + 1;

    for (int i = 1; i <= n; i++)
        sp[i][0] = a[i];                    // Đoạn dài 1

    for (int j = 1; (1 << j) <= n; j++)
        for (int i = 1; i + (1 << j) - 1 <= n; i++)
            sp[i][j] = min(sp[i][j-1], sp[i + (1 << (j-1))][j-1]);
}

int query(int l, int r) {
    int k = LOG[r - l + 1];
    return min(sp[l][k], sp[r - (1 << k) + 1][k]);  // O(1)!
}
Lưu ý: Sparse Table chỉ hoạt động đúng với các phép toán overlap-friendly (min, max, GCD, OR, AND). Với phép tổng thì phải dùng BIT/Segment Tree vì 2 đoạn chồng nhau sẽ đếm lặp.

💡 Hình dung: Ném viên đá xuống hồ. Sóng lan ra vòng 1 (các đỉnh cách 1 bước), rồi vòng 2 (cách 2 bước), vòng 3... Vì sóng lan đều nên đỉnh nào sóng đến trước thì đó là đường ngắn nhất. Dùng "hàng đợi" (xếp hàng ai đến trước xử lý trước) để đảm bảo duyệt đúng thứ tự.

// ═══ BFS tìm đường đi từ S đến F ═══
// Phong cách: Free (đánh dấu) + Trace (truy vết)

vector<vector<int>> ke;
vector<bool> Free;
vector<int> Trace;
int n;

void BFS(int S) {
    Free.assign(n + 1, true);
    Trace.assign(n + 1, -1);
    
    queue<int> q;
    q.push(S);
    Free[S] = false;
    Trace[S] = S;
    
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        
        for (int v : ke[u]) {
            if (Free[v]) {
                Free[v] = false;
                Trace[v] = u;
                q.push(v);
            }
        }
    }
}

// ═══ Truy vết đường đi từ F về S ═══

    if (Trace[F] == -1) {

        duongDi.push_back(cur);
        cur = Trace[cur];
    }
    duongDi.push_back(S);
    reverse(duongDi.begin(), duongDi.end());
    
    for (int x : duongDi) out << x << " ";
}

int dx[] = {-1, 1, 0, 0};

        }

vector<vector<bool>> a;

D3. Dijkstra — Đường đi ngắn nhất (có trọng số)

💡 Hình dung: Bản đồ thành phố với các con đường dài ngắn khác nhau (có "trọng số" = chiều dài). Từ giao lộ A, muốn tìm đường ngắn nhất đến mọi giao lộ khác. Cách làm: luôn chọn giao lộ GẦN NHẤT chưa thăm → cập nhật khoảng cách các giao lộ liền kề → lặp lại cho đến hết. Dùng "thùng lấy quả nhỏ nhất" (Priority Queue) để chọn nhanh.

DẤU HIỆU: "đường đi ngắn nhất" trên đồ thị có trọng số dương
// ═══ Dijkstra — O(E log V) ═══

vector<vector<pair<int,int>>> ke;   // ke[u] = {{v, w}, ...}
vector<long long> dist;
vector<int> Trace;
vector<bool> Free;
int n;

void Dijkstra(int S) {
    dist.assign(n + 1, 1e18);
    Trace.assign(n + 1, -1);
    Free.assign(n + 1, true);
    
    // Min heap: {khoảng cách, đỉnh}
    priority_queue<pair<long long,int>, 
                   vector<pair<long long,int>>, 
                   greater<pair<long long,int>>> pq;
    
    dist[S] = 0;
    Trace[S] = S;
    pq.push({0, S});
    
    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        
        if (!Free[u]) continue;   // Đã xét rồi → bỏ qua
        Free[u] = false;
        
        for (auto [v, w] : ke[u]) {
            if (Free[v] && dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                Trace[v] = u;
                pq.push({dist[v], v});
            }
        }
    }
}

// Truy vết đường đi ngắn nhất từ S đến F
void TruyVet(int S, int F) {
    if (dist[F] >= 1e18) { out << -1; return; }
    vector<int> duongDi;
    int cur = F;
    while (cur != S) {
        duongDi.push_back(cur);
        cur = Trace[cur];
    }
    duongDi.push_back(S);
    reverse(duongDi.begin(), duongDi.end());
    
    out << "Khoang cach: " << dist[F] << '\n';
    for (int x : duongDi) out << x << " ";
}

D4. Floyd-Warshall — Đường đi ngắn nhất giữa MỌI cặp đỉnh

💡 Hình dung: Dijkstra tìm đường ngắn nhất từ 1 giao lộ. Floyd tìm đường ngắn nhất giữa TẤT CẢ các cặp giao lộ cùng lúc. Code cực ngắn — chỉ 3 vòng for lồng nhau. Nhược điểm: chỉ dùng khi số giao lộ nhỏ (≤ 400) vì chạy chậm hơn Dijkstra nhiều.

DẤU HIỆU: "đường đi ngắn nhất giữa mọi cặp đỉnh", N ≤ 400, "kiểm tra 2 đỉnh có đường đi không"
// ═══ Floyd-Warshall — O(N³) ═══
long long d[405][405];   // d[i][j] = khoảng cách ngắn nhất từ i đến j
int n;

void Floyd() {
    // Khởi tạo: d[i][j] = trọng số cạnh (i,j)
    // Nếu không có cạnh: d[i][j] = 1e18
    // d[i][i] = 0

        for (int i = 1; i <= n; i++)      // Đỉnh xuất phát
            for (int j = 1; j <= n; j++)  // Đỉnh đích
                if (d[i][k] + d[k][j] < d[i][j])
                    d[i][j] = d[i][k] + d[k][j];
}

D5. Bellman-Ford — Đường đi ngắn nhất (có trọng số ÂM)

💡 Hình dung: Giống Dijkstra — tìm đường ngắn nhất từ 1 điểm đến mọi điểm khác. Nhưng Dijkstra "phá sản" khi gặp đường có chi phí âm (đường giảm giá). Bellman-Ford chậm hơn nhưng xử lý được cả trọng số âm. Cách làm: lặp đi lặp lại (n-1) vòng, mỗi vòng duyệt TẤT CẢ các cạnh và cập nhật khoảng cách nếu tìm được đường ngắn hơn.

DẤU HIỆU: "đường đi ngắn nhất" + đồ thị có trọng số ÂM, hoặc cần phát hiện chu trình âm
SO SÁNH: Dijkstra: O(E log V), nhanh, nhưng CHỈ dùng khi trọng số ≥ 0
Bellman-Ford: O(V × E), chậm hơn, nhưng chấp nhận trọng số âm 
// ═══ Bellman-Ford — O(V × E) ═══
// Biểu diễn bằng MA TRẬN KỀ: a[u][v] = trọng số

vector<vector<int>> a;
vector<int> d;                       // d[i] = khoảng cách ngắn nhất S → i
vector<int> Trace;
int n, m, S, F;

void BellmanFord() {
    // Lặp n-1 vòng
    for (int lap = 1; lap <= n - 1; lap++) {
        bool coCapNhat = false;
        // Duyệt tất cả các cạnh (u, v)
        for (int u = 1; u <= n; u++) {
            for (int v = 1; v <= n; v++) {
                if (a[u][v] != INT_MAX && d[u] != INT_MAX) {
                    if (d[u] + a[u][v] < d[v]) {
                        d[v] = d[u] + a[u][v];
                        Trace[v] = u;
                        coCapNhat = true;
                    }
                }
            }
        }
        if (!coCapNhat) break;       // Dừng sớm nếu không cập nhật gì
    }
}

void Result() {
    if (d[F] == INT_MAX) {
        out << "Path from " << S << " to " << F << " not found";
    } else {
        out << "Distance: " << d[F] << '\n';
        // Truy vết đường đi
        vector<int> duongDi;
        int cur = F;
        while (cur != S) {
            duongDi.push_back(cur);
            cur = Trace[cur];
        }
        duongDi.push_back(S);
        reverse(duongDi.begin(), duongDi.end());
        for (int x : duongDi) out << x << " ";
    }
}

int main() {
    in >> n >> m >> S >> F;
    a.assign(n + 1, vector<int>(n + 1, INT_MAX));
    d.assign(n + 1, INT_MAX);
    Trace.assign(n + 1, S);
    for (int i = 1; i <= n; i++) a[i][i] = 0;
    while (m--) {
        int u, v, w;
        in >> u >> v >> w;
        a[u][v] = w;                 // Đồ thị CÓ HƯỚNG
    }
    d[S] = 0;
    BellmanFord();
    Result();
    return 0;
}

D6. Prim — Cây khung nhỏ nhất (Minimum Spanning Tree)

💡 Hình dung: Có n ngôi nhà và nhiều con đường nối giữa chúng, mỗi đường có chi phí xây dựng khác nhau. Cần nối TẤT CẢ các nhà lại bằng đường, sao cho tổng chi phí THẤP NHẤT và không có đường thừa (không tạo vòng). Cách làm: bắt đầu từ 1 nhà, mỗi bước chọn con đường RẺ NHẤT nối nhà đã kết nối → nhà chưa kết nối. Lặp lại cho đến khi tất cả nhà đều được nối.

DẤU HIỆU: "nối tất cả các điểm với chi phí nhỏ nhất", "cây khung nhỏ nhất", "mạng lưới tối ưu"
// ═══ Prim — O(V²) với ma trận kề ═══

vector<vector<int>> a;              // Ma trận kề: a[u][v] = trọng số
vector<bool> Free;
int n, m, total = 0;
vector<pair<int,int>> result;        // Lưu các cạnh trong cây khung

void Prim() {
    Free[1] = false;                 // Bắt đầu từ đỉnh 1
    int dem = 0;
    while (dem < n - 1) {
        int u = -1, v = -1, minC = INT_MAX;
        // Tìm cạnh nhỏ nhất nối tập đã chọn ↔ tập chưa chọn
        for (int i = 1; i <= n; i++) {
            if (!Free[i]) {          // Đỉnh đã thuộc cây khung
                for (int j = 1; j <= n; j++) {
                    if (Free[j] && a[i][j] < minC) {
                        u = i; v = j;
                        minC = a[i][j];
                    }
                }
            }
        }
        if (u == -1 || v == -1) return;  // Đồ thị không liên thông
        total += a[u][v];
        result.push_back({u, v});
        Free[v] = false;
        dem++;
    }
}

int main() {
    in >> n >> m;
    a.assign(n + 1, vector<int>(n + 1, INT_MAX));
    Free.assign(n + 1, true);
    while (m--) {
        int u, v, w;
        in >> u >> v >> w;
        a[u][v] = a[v][u] = w;      // Đồ thị VÔ HƯỚNG
    }
    Prim();
    // In kết quả
    if ((int)result.size() != n - 1) {
        out << "Do thi khong lien thong";
    } else {
        for (auto [u, v] : result)
            out << "(" << u << ", " << v << ") = " << a[u][v] << '\n';
        out << "Tong trong so: " << total;
    }
    return 0;
}

D7. Ford-Fulkerson — Luồng cực đại (Maximum Flow)

💡 Hình dung: Mạng ống nước — nước chảy từ bể chứa (nguồn S) đến nhà máy (đích F). Mỗi ống có sức chứa tối đa (capacity). Hỏi: tổng lượng nước lớn nhất có thể chảy từ S đến F? Cách làm: liên tục tìm đường còn chỗ trống từ S→F (bằng BFS), rồi đẩy nước qua đường đó. Lặp lại đến khi không còn đường nào nữa.

DẤU HIỆU: "luồng cực đại", "lưu lượng tối đa", "cắt nhỏ nhất" (max-flow min-cut)
// ═══ Ford-Fulkerson (dùng BFS — Edmonds-Karp) — O(V × E²) ═══

vector<vector<int>> Cap, Flow;      // Cap = sức chứa, Flow = luồng hiện tại
vector<int> Trace;
int n, m, S, F;

bool BFS_TimDuong() {
    Trace.assign(n + 1, -1);
    queue<int> q;
    q.push(S);
    Trace[S] = S;
    while (!q.empty()) {
        int u = q.front(); q.pop();
        for (int v = 1; v <= n; v++) {
            if (Trace[v] == -1 && Cap[u][v] > Flow[u][v]) {
                Trace[v] = u;
                if (v == F) return true;   // Tìm được đường tăng
                q.push(v);
            }
        }
    }
    return false;                          // Không còn đường tăng
}

void MaxFlow() {
    while (BFS_TimDuong()) {
        // Tìm lượng nước nhỏ nhất trên đường tăng
        int minCap = INT_MAX;
        for (int v = F; v != S; v = Trace[v]) {
            int u = Trace[v];
            minCap = min(minCap, Cap[u][v] - Flow[u][v]);
        }
        // Cập nhật luồng dọc đường tăng
        for (int v = F; v != S; v = Trace[v]) {
            int u = Trace[v];
            Flow[u][v] += minCap;
            Flow[v][u] -= minCap;          // Luồng ngược
        }
    }
}

int main() {
    in >> n >> m >> S >> F;
    Cap.assign(n + 1, vector<int>(n + 1, 0));
    Flow.assign(n + 1, vector<int>(n + 1, 0));
    while (m--) {
        int u, v, w;
        in >> u >> v >> w;
        Cap[u][v] += w;                    // Hỗ trợ cạnh song song
    }
    MaxFlow();
    // Tính tổng luồng ra từ nguồn S
    int res = 0;
    for (int i = 1; i <= n; i++) res += Flow[S][i];
    out << "Max flow: " << res;
    return 0;
}

D8. Kruskal — Cây khung nhỏ nhất (dùng DSU)

💡 Hình dung: Giống Prim (nối tất cả nhà với chi phí thấp nhất), nhưng cách làm khác: liệt kê TẤT CẢ các con đường, sắp theo chi phí tăng dần, rồi chọn từng con đường rẻ nhất mà không tạo vòng. Dùng DSU để kiểm tra "2 đầu đường đã nối chưa?".

DẤU HIỆU: Giống Prim. Khi nào chọn Kruskal? Khi đồ thị có nhiều cạnh, ít đỉnh hoặc đã có sẵn DSU.
// ═══ Kruskal — O(E log E) dùng DSU ═══

struct Edge {
    int u, v, w;
    bool operator<(const Edge &o) const { return w < o.w; }
};

int cha[100005], hang[100005];

int tim_goc(int v) {
    if (v == cha[v]) return v;
    return cha[v] = tim_goc(cha[v]);
}

void hop_nhat(int a, int b) {
    a = tim_goc(a); b = tim_goc(b);
    if (a == b) return;
    if (hang[a] < hang[b]) swap(a, b);
    cha[b] = a;
    if (hang[a] == hang[b]) hang[a]++;
}

int main() {
    ifstream in("TENBAI.INP");
    ofstream out("TENBAI.OUT");
    int n, m;
    in >> n >> m;
    
    vector<Edge> edges(m);
    for (int i = 0; i < m; i++)
        in >> edges[i].u >> edges[i].v >> edges[i].w;
    
    sort(edges.begin(), edges.end());  // Sắp cạnh tăng theo trọng số
    
    for (int i = 1; i <= n; i++) { cha[i] = i; hang[i] = 0; }
    
    long long total = 0;
    int dem = 0;
    for (auto &[u, v, w] : edges) {
        if (tim_goc(u) != tim_goc(v)) {   // Không cùng nhóm → không tạo vòng
            hop_nhat(u, v);
            total += w;
            dem++;
            if (dem == n - 1) break;       // Đủ n-1 cạnh = xong
        }
    }
    
    if (dem == n - 1)
        out << "Tong trong so: " << total;
    else
        out << "Do thi khong lien thong";
    return 0;
}

D9. Topo Sort — Sắp xếp topo (Thứ tự thực hiện)

💡 Hình dung: Trong môn học, muốn học "Giải tích 2" thì phải học "Giải tích 1" trước. Muốn học "CTDL" thì phải học "Lập trình cơ bản" trước. Sắp xếp topo = tìm thứ tự học môn sao cho không vi phạm tiên quyết. Dùng BFS với "đếm bậc vào" (Kahn's algorithm).

DẤU HIỆU: "thứ tự thực hiện công việc", "môn tiên quyết", "đồ thị có hướng không chu trình (DAG)"
// ═══ Topo Sort (Kahn's Algorithm — BFS) — O(V + E) ═══

int n, m;
vector<vector<int>> ke;        // Danh sách kề (đồ thị CÓ HƯỚNG)
vector<int> bacVao;            // bacVao[v] = số cạnh đi VÀO v

void TopoSort() {
    queue<int> q;
    // Bước 1: Đẩy các đỉnh có bậc vào = 0 vào queue
    for (int i = 1; i <= n; i++)
        if (bacVao[i] == 0) q.push(i);
    
    vector<int> ketQua;
    while (!q.empty()) {
        int u = q.front(); q.pop();
        ketQua.push_back(u);
        
        // Bước 2: Xóa đỉnh u → giảm bậc vào các đỉnh kề
        for (int v : ke[u]) {
            bacVao[v]--;
            if (bacVao[v] == 0) q.push(v);
        }
    }
    
    if ((int)ketQua.size() != n) {
        out << "Co chu trinh!";       // Không phải DAG
    } else {
        for (int x : ketQua) out << x << " ";
    }
}

int main() {
    ifstream in("TENBAI.INP");
    ofstream out("TENBAI.OUT");
    in >> n >> m;
    ke.assign(n + 1, vector<int>());
    bacVao.assign(n + 1, 0);
    while (m--) {
        int u, v;
        in >> u >> v;                  // Cạnh u → v (u phải xong trước v)
        ke[u].push_back(v);
        bacVao[v]++;
    }
    TopoSort();
    return 0;
}

E1. Kiểm tra số nguyên tố — O(√n)

💡 Số nguyên tố là số chỉ chia hết cho 1 và chính nó (ví dụ: 2, 3, 5, 7, 11...). Để kiểm tra số n có phải SNT không, không cần thử chia từ 2 đến n. Chỉ cần thử đến √n là đủ — ví dụ: kiểm tra 100 chỉ cần thử đến 10.

bool laSNT(long long n) {
    if (n < 2) return false;
    if (n == 2) return true;
    if (n % 2 == 0) return false;
    for (long long i = 3; i * i <= n; i += 2)
        if (n % i == 0) return false;
    return true;
}

E3. Lũy thừa nhanh — a^n mod m trong O(log n)

💡 Hình dung: Tính 2¹⁰ bình thường cần nhân 10 lần. Nhưng: 2¹⁰ = (2⁵)², và 2⁵ = 2×(2²)² → chỉ cần vài lần nhân! Với số mũ hàng tỷ, thay vì nhân hàng tỷ lần, ta chỉ cần ~30 lần nhân. Kết hợp lấy phần dư (mod) sau mỗi bước để số không bị quá lớn.

long long power(long long a, long long n, long long mod) {
    long long res = 1;
    a %= mod;
    while (n > 0) {
        if (n & 1) res = res * a % mod;
        a = a * a % mod;
        n >>= 1;
    }
    return res;
}

E4. GCD / LCM / Phân tích thừa số

💡 GCD (Ước Chung Lớn Nhất): số lớn nhất chia hết cho cả a và b. Ví dụ: GCD(12, 18) = 6. LCM (Bội Chung Nhỏ Nhất): số nhỏ nhất chia hết cho cả a và b. Ví dụ: LCM(4, 6) = 12. Mẹo: LCM = a ÷ GCD × b. Phân tích thừa số: tách số ra thành tích các số nguyên tố, ví dụ: 60 = 2² × 3 × 5.

vector<pair<int,int>> phanTich(int n) {
    vector<pair<int,int>> res;
    for (int i = 2; i * i <= n; i++) {
        if (n % i == 0) {
            int cnt = 0;
            while (n % i == 0) { n /= i; cnt++; }
            res.push_back({i, cnt});
        }
    }
    if (n > 1) res.push_back({n, 1});
    return res;
}
