A1. Prefix Sum — Tổng tiền tố

💡 Hình dung: Em có 1 cuốn sổ ghi "tổng tiền từ ngày 1 đến hôm nay" mỗi ngày. Khi sếp hỏi "tổng tiền từ ngày 3 đến ngày 7", thay vì cộng lại 5 ngày, em chỉ cần lấy: tổng_đến_ngày_7 − tổng_đến_ngày_2 = ra ngay! Tính trước 1 lần → trả lời bao nhiêu câu hỏi cũng chỉ cần 1 phép trừ.

DẤU HIỆU: "tổng đoạn [L,R]", "trung bình đoạn", "đoạn con có tổng = k" → truy vấn nhiều lần
// ═══ PREFIX SUM 1D ═══
// Build: O(n) | Query: O(1)
long long prefix[100005];
prefix[0] = 0;
for (int i = 1; i <= n; i++)
    prefix[i] = prefix[i-1] + a[i];

// Tổng đoạn [l, r]
long long tong = prefix[r] - prefix[l-1];
📝 VÍ DỤ CHẠY TAY: 
Mảng:    a[] = { 3,  1,  4,  1,  5 }   (đánh số từ 1)
              a[1] a[2] a[3] a[4] a[5]

Bước 1 — Tính prefix:
  prefix[0] = 0
  prefix[1] = 0 + 3 = 3
  prefix[2] = 3 + 1 = 4
  prefix[3] = 4 + 4 = 8
  prefix[4] = 8 + 1 = 9
  prefix[5] = 9 + 5 = 14

  prefix[] = { 0, 3, 4, 8, 9, 14 }

Bước 2 — Truy vấn: Tổng đoạn [2, 4] = ?
  = prefix[4] - prefix[1] = 9 - 3 = 6
  Kiểm tra: a[2] + a[3] + a[4] = 1 + 4 + 1 = 6 ✓

// ═══ PREFIX SUM 2D ═══
// Build: O(m×n) | Query: O(1)
// sum[i][j] = tổng HCN từ (1,1) đến (i,j)
for (int i = 1; i <= m; i++)
    for (int j = 1; j <= n; j++)

// Tổng HCN từ (r1,c1) đến (r2,c2)
ll tong = sum[r2][c2] - sum[r1-1][c2] - sum[r2][c1-1] + sum[r1-1][c1-1];

A2. Two Pointers — Hai con trỏ

💡 Hình dung: Hai bạn L và R đứng trên một hàng ghế. R đi sang phải để mở rộng đoạn, nếu đoạn vi phạm điều kiện thì L đi sang phải để thu hẹp. Cả hai chỉ đi về bên phải, không lùi → thay vì thử tất cả các cặp vị trí (rất lâu), ta chỉ cần mỗi người đi qua hàng ghế 1 lần → nhanh gấp nhiều lần.

DẤU HIỆU: "đoạn con dài nhất có tổng ≤ K", "cặp (i,j) thỏa a[i]+a[j] = k" (mảng đã sort), "đoạn ngắn nhất chứa đủ k giá trị"
// ═══ Tìm đoạn con dài nhất có tổng ≤ K ═══
// O(n) — L chỉ tiến, không lùi
int L = 0, maxLen = 0;
long long sum = 0;

for (int R = 0; R < n; R++) {
    sum += a[R];                    // Mở rộng phải
    while (sum > K && L <= R) {    // Thu hẹp trái nếu vi phạm
        sum -= a[L];
        L++;
    }
    maxLen = max(maxLen, R - L + 1);
}

// ═══ Đếm cặp tổng = K (mảng đã sort) ═══
sort(a, a + n);
int L = 0, R = n - 1, cnt = 0;
while (L < R) {
    int s = a[L] + a[R];
    if (s == K)      { cnt++; L++; R--; }
    else if (s < K)  L++;
    else             R--;
}

A3. Sliding Window — Cửa sổ trượt

💡 Hình dung: Em nhìn qua một cái cửa sổ chỉ thấy được k ngôi nhà liên tiếp trên phố. Em đi bộ — cửa sổ trượt sang phải: mất 1 nhà bên trái, thêm 1 nhà bên phải. Em muốn biết "nhà cao nhất đang thấy" nhưng không muốn đếm lại k nhà mỗi bước. Mẹo: ghi nhớ thông minh — chỉ giữ lại "ứng viên" có thể là nhà cao nhất, bỏ những nhà chắc chắn thấp hơn.

DẤU HIỆU: "max/min trong cửa sổ kích thước k", "đoạn có max - min ≤ K"
Khác Two Pointers: Sliding Window thường duy trì thêm trạng thái (deque, map, tần suất) bên trong cửa sổ.
// ═══ Max trong mỗi cửa sổ kích thước k ═══
// Dùng deque đơn điệu giảm — O(n)
deque<int> dq;  // Lưu CHỈ SỐ

for (int i = 0; i < n; i++) {
    while (!dq.empty() && dq.front() <= i - k)   // Ngoài cửa sổ
        dq.pop_front();
    while (!dq.empty() && a[dq.back()] <= a[i])   // Nhỏ hơn → vô dụng
        dq.pop_back();
    dq.push_back(i);
    if (i >= k - 1)
        out << a[dq.front()] << " ";  // Max cửa sổ hiện tại
}

A4. Binary Search — Tìm kiếm nhị phân

💡 Hình dung trò chơi "Đoán số": bạn nghĩ 1 số từ 1 đến 1.000.000, em đoán và được trả lời "lớn hơn" hoặc "nhỏ hơn". Thay vì đoán từ 1, 2, 3... em luôn đoán SỐ GIỮA → mỗi lần loại bỏ NỬA số còn lại. 1 triệu số → chỉ cần ~20 lần đoán!
"Chặt nhị phân đáp án" cùng ý tưởng: thử đáp án ở giữa, kiểm tra có được không, rồi thu hẹp phạm vi.

DẤU HIỆU:
  • "tìm x trong dãy đã sort" → lower_bound, upper_bound
  • "đếm phần tử trong [L, R]" → upper_bound(R) - lower_bound(L)
  • "tìm giá trị nhỏ nhất/lớn nhất thỏa..." → Chặt nhị phân đáp án (RẤT PHỔ BIẾN!)
// ═══ Hàm có sẵn (mảng PHẢI sort trước) ═══
sort(a, a + n);

int pos1 = lower_bound(a, a+n, x) - a;  // Vị trí đầu tiên >= x
int pos2 = upper_bound(a, a+n, x) - a;  // Vị trí đầu tiên > x
int cntX = pos2 - pos1;                  // Số phần tử = x
int cntLR = upper_bound(a,a+n,R) - lower_bound(a,a+n,L);  // Trong [L,R]

// ═══ CHẶT NHỊ PHÂN ĐÁP ÁN ═══
// "Tìm giá trị nhỏ nhất sao cho check(mid) = true"
int lo = MIN_ANS, hi = MAX_ANS, ans = -1;
while (lo <= hi) {
    int mid = lo + (hi - lo) / 2;
    if (check(mid)) {   // Hàm kiểm tra mid có thỏa không
        ans = mid;
        hi = mid - 1;   // Tìm nhỏ hơn
    } else {
        lo = mid + 1;
    }
}

A5. Sorting — Sắp xếp

💡 Hình dung: Bài tú lơ khơ — khi xòe tay không xếp, tìm 1 lá bài rất lâu. Nhưng nếu xếp theo số (1, 2, 3...), tìm hay ghép cặp đều nhanh. Trong lập trình, sắp xếp mảng trước thường giúp bài toán dễ giải hơn rất nhiều. C++ có sẵn hàm sort() chạy rất nhanh.

DẤU HIỆU: "ghép cặp tối ưu", "|a[i]-a[j]| ≤ k", "sắp xếp theo nhiều tiêu chí"
MẸO: Sort trước → nhiều bài giải được bằng Greedy hoặc Two Pointers

// Sort struct theo nhiều tiêu chí
bool cmp(SinhVien a, SinhVien b) {

// Sort pair (tự động theo first, rồi second)

    else

}

C3. DSU (Disjoint Set Union) — Gộp nhóm

💡 Hình dung: Lớp có 40 bạn, ban đầu ai cũng là 1 nhóm riêng (40 nhóm). Khi 2 bạn kết bạn, nhóm của 2 bạn GỘP LẠI thành 1. Sau nhiều lần gộp, muốn hỏi "A và B có cùng nhóm không?" → chỉ cần xem 2 người có chung "Nhóm trưởng" không. Gộp nhóm + hỏi nhóm đều cực nhanh.

DẤU HIỆU: "cùng nhóm không?", "bao nhiêu nhóm/thành phần liên thông?", "gộp 2 tập hợp", Kruskal
// ═══ DSU — Phong cách quen thuộc ═══

const int MAXN = 100005;
int cha[MAXN];                // cha[i] = cha của đỉnh i
int hang[MAXN];               // hang[i] = chiều cao cây gốc i

void khoiTao(int n) {
    for (int i = 1; i <= n; i++) {
        cha[i] = i;            // Ban đầu mỗi đỉnh là gốc của chính nó
        hang[i] = 0;
    }
}

int tim_goc(int v) {
    if (v == cha[v]) return v;
    return cha[v] = tim_goc(cha[v]);  // Nén đường đi — RẤT QUAN TRỌNG!
}

void hop_nhat(int a, int b) {
    a = tim_goc(a);
    b = tim_goc(b);
    if (a == b) return;               // Đã cùng nhóm rồi
    // Union by rank: gốc nhỏ treo vào gốc lớn
    if (hang[a] < hang[b]) swap(a, b);
    cha[b] = a;
    if (hang[a] == hang[b]) hang[a]++;
}

bool cungNhom(int u, int v) {
    return tim_goc(u) == tim_goc(v);
}

// ═══ Ứng dụng: Đếm số nhóm / Tổng diện tích nhóm lớn nhất ═══

// Đếm số nhóm (thành phần liên thông)
int soNhom = 0;
for (int i = 1; i <= n; i++)
    if (cha[i] == i) soNhom++;

// Tổng giá trị nhóm lớn nhất (ví dụ: bài LATGACH)

    if (cha[i] == i)

long long bit[100005];
int n;

void capNhat(int i, long long delta) {
    for (; i <= n; i += i & (-i))
        bit[i] += delta;
}

long long tong(int i) {
    long long s = 0;
    for (; i > 0; i -= i & (-i))
        s += bit[i];
    return s;
}

long long tongDoan(int l, int r) {
    return tong(r) - tong(l - 1);
}

D1. BFS — Duyệt theo chiều rộng

💡 Hình dung: Ném viên đá xuống hồ. Sóng lan ra vòng 1 (các đỉnh cách 1 bước), rồi vòng 2 (cách 2 bước), vòng 3... Vì sóng lan đều nên đỉnh nào sóng đến trước thì đó là đường ngắn nhất. Dùng "hàng đợi" (xếp hàng ai đến trước xử lý trước) để đảm bảo duyệt đúng thứ tự.

DẤU HIỆU: "đường đi ngắn nhất" (đồ thị không trọng số), "duyệt theo tầng", "lưới/mê cung", "truy vết đường đi"
// ═══ BFS tìm đường đi từ S đến F ═══
// Phong cách: Free (đánh dấu) + Trace (truy vết)

vector<vector<int>> ke;
vector<bool> Free;
vector<int> Trace;
int n;

void BFS(int S) {
    Free.assign(n + 1, true);
    Trace.assign(n + 1, -1);
    
    queue<int> q;
    q.push(S);
    Free[S] = false;
    Trace[S] = S;
    
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        
        for (int v : ke[u]) {
            if (Free[v]) {
                Free[v] = false;
                Trace[v] = u;
                q.push(v);
            }
        }
    }
}

// ═══ Truy vết đường đi từ F về S ═══
void TruyVet(int S, int F) {
    if (Trace[F] == -1) {
        out << -1;  // Không có đường đi
        return;
    }
    vector<int> duongDi;
    int cur = F;
    while (cur != S) {
        duongDi.push_back(cur);
        cur = Trace[cur];
    }
    duongDi.push_back(S);
    reverse(duongDi.begin(), duongDi.end());
    
    for (int x : duongDi) out << x << " ";
}

D1b. BFS trên lưới 2D (Mê cung)

💡 Hình dung: Bản đồ ô vuông, có tường ('#') và đường đi ('.'). Em đứng ở ô xuất phát, mỗi bước đi sang ô kề bên (lên/xuống/trái/phải). Hỏi: cần ít nhất bao nhiêu bước để đến ô đích? → Dùng BFS, sóng lan từ ô xuất phát ra 4 hướng.

DẤU HIỆU: "lưới ô vuông", "mê cung", "số bước ít nhất từ ô (r1,c1) đến (r2,c2)"
// ═══ BFS trên lưới m×n ═══
int m, n;                           // số hàng, số cột
char a[105][105];                   // '.' = đi được, '#' = tường
int dist[105][105];                 // dist[i][j] = số bước đến ô (i,j)

// 4 hướng: lên, xuống, trái, phải
int dx[] = {-1, 1, 0, 0};
int dy[] = {0, 0, -1, 1};

void BFS_Luoi(int sr, int sc) {     // Xuất phát từ ô (sr, sc)
    memset(dist, -1, sizeof(dist)); // -1 = chưa thăm
    dist[sr][sc] = 0;
    queue<pair<int,int>> q;
    q.push({sr, sc});
    
    while (!q.empty()) {
        auto [x, y] = q.front();
        q.pop();
        
        for (int k = 0; k < 4; k++) {    // Thử 4 hướng
            int nx = x + dx[k];
            int ny = y + dy[k];
            // Kiểm tra: trong lưới + đi được + chưa thăm
            if (nx >= 0 && nx < m && ny >= 0 && ny < n
                && a[nx][ny] != '#' && dist[nx][ny] == -1) {
                dist[nx][ny] = dist[x][y] + 1;
                q.push({nx, ny});
            }
        }
    }
}

D2. DFS — Duyệt theo chiều sâu

💡 Hình dung: Khám phá mê cung bằng cách luôn rẽ vào ngõ đầu tiên gặp được, đi sâu cho đến khi đường cùng thì quay lại, thử ngõ tiếp theo. Ví dụ dùng: đếm xem bản đồ có bao nhiêu "đảo" riêng biệt (nhóm đỉnh nối với nhau). Mỗi lần bắt đầu DFS từ 1 đỉnh chưa thăm = tìm được 1 đảo mới.

DẤU HIỆU: "thành phần liên thông", "kiểm tra chu trình", "tô màu đồ thị"
LƯU Ý: Dùng stack thay đệ quy để tránh tràn bộ nhớ khi N lớn (N > 10⁴)
// ═══ DFS dùng STACK (an toàn, không tràn) ═══

vector<vector<int>> ke;
vector<bool> Free;
vector<int> Trace;
int n;

void DFS(int S) {
    Free.assign(n + 1, true);
    Trace.assign(n + 1, -1);
    
    stack<int> stk;
    stk.push(S);
    Free[S] = false;
    
    while (!stk.empty()) {
        int u = stk.top();
        stk.pop();
        
        for (int v : ke[u]) {
            if (Free[v]) {
                Free[v] = false;
                Trace[v] = u;
                stk.push(v);
            }
        }
    }
}

D2b. Kiểm tra đồ thị liên thông

💡 Hình dung: Mạng internet — tất cả các máy tính đều nối được với nhau không? Nếu có nhóm máy bị cô lập → mạng "không liên thông". Cách kiểm tra: chạy BFS/DFS từ đỉnh 1. Nếu thăm được TẤT CẢ đỉnh → liên thông (YES). Còn đỉnh nào chưa thăm → không liên thông (NO).

DẤU HIỆU: "kiểm tra liên thông", "tất cả các điểm có nối được với nhau không"
// ═══ Kiểm tra liên thông bằng DFS ═══

vector<vector<bool>> a;
vector<bool> Free;
int n, m;

void DFS(int start) {
    stack<int> st;
    st.push(start);
    Free[start] = false;
    while (!st.empty()) {
        int u = st.top(); st.pop();
        for (int v = 1; v <= n; v++)
            if (Free[v] && a[u][v]) {
                Free[v] = false;
                st.push(v);
            }
    }
}

void CheckLT() {
    DFS(1);   // Duyệt từ đỉnh 1
    bool lienThong = true;
    for (int i = 1; i <= n; i++)
        if (Free[i]) { lienThong = false; break; }
    out << (lienThong ? "YES" : "NO");
}

int main() {
    in >> n >> m;
    a.assign(n + 1, vector<bool>(n + 1, false));
    Free.assign(n + 1, true);
    while (m--) {
        int u, v;
        in >> u >> v;
        a[u][v] = a[v][u] = true;
    }
    CheckLT();
    return 0;
}

D3. Dijkstra — Đường đi ngắn nhất (có trọng số)

💡 Hình dung: Bản đồ thành phố với các con đường dài ngắn khác nhau (có "trọng số" = chiều dài). Từ giao lộ A, muốn tìm đường ngắn nhất đến mọi giao lộ khác. Cách làm: luôn chọn giao lộ GẦN NHẤT chưa thăm → cập nhật khoảng cách các giao lộ liền kề → lặp lại cho đến hết. Dùng "thùng lấy quả nhỏ nhất" (Priority Queue) để chọn nhanh.

DẤU HIỆU: "đường đi ngắn nhất" trên đồ thị có trọng số dương
// ═══ Dijkstra — O(E log V) ═══

vector<vector<pair<int,int>>> ke;   // ke[u] = {{v, w}, ...}
vector<long long> dist;
vector<int> Trace;
vector<bool> Free;
int n;

void Dijkstra(int S) {
    dist.assign(n + 1, 1e18);
    Trace.assign(n + 1, -1);
    Free.assign(n + 1, true);
    
    // Min heap: {khoảng cách, đỉnh}
    priority_queue<pair<long long,int>, 
                   vector<pair<long long,int>>, 
                   greater<pair<long long,int>>> pq;
    
    dist[S] = 0;
    Trace[S] = S;
    pq.push({0, S});
    
    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        
        if (!Free[u]) continue;   // Đã xét rồi → bỏ qua
        Free[u] = false;
        
        for (auto [v, w] : ke[u]) {
            if (Free[v] && dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                Trace[v] = u;
                pq.push({dist[v], v});
            }
        }
    }
}

// Truy vết đường đi ngắn nhất từ S đến F
void TruyVet(int S, int F) {
    if (dist[F] >= 1e18) { out << -1; return; }
    vector<int> duongDi;
    int cur = F;
    while (cur != S) {
        duongDi.push_back(cur);
        cur = Trace[cur];
    }
    duongDi.push_back(S);
    reverse(duongDi.begin(), duongDi.end());
    
    out << "Khoang cach: " << dist[F] << '\n';
    for (int x : duongDi) out << x << " ";
}

Đồ thị có hướng, 5 đỉnh:
1→2 (w=2), 1→3 (w=5), 2→3 (w=1), 2→4 (w=7), 3→4 (w=3), 4→5 (w=1)

  Free[1] = false
  → Cập nhật: dist[2] = 0+2 = 2, dist[3] = 0+5 = 5
  dist = [_, 0, 2, 5, ∞, ∞]

Bước 2: Lấy đỉnh gần nhất = đỉnh 2 (dist=2)
  Free[2] = false
  → Cập nhật: dist[3] = min(5, 2+1) = 3, dist[4] = min(∞, 2+7) = 9
  dist = [_, 0, 2, 3, 9, ∞]

Bước 3: Lấy đỉnh gần nhất = đỉnh 3 (dist=3)
  Free[3] = false

D4. Floyd-Warshall — Đường đi ngắn nhất giữa MỌI cặp đỉnh

💡 Hình dung: Dijkstra tìm đường ngắn nhất từ 1 giao lộ. Floyd tìm đường ngắn nhất giữa TẤT CẢ các cặp giao lộ cùng lúc. Code cực ngắn — chỉ 3 vòng for lồng nhau. Nhược điểm: chỉ dùng khi số giao lộ nhỏ (≤ 400) vì chạy chậm hơn Dijkstra nhiều.

DẤU HIỆU: "đường đi ngắn nhất giữa mọi cặp đỉnh", N ≤ 400, "kiểm tra 2 đỉnh có đường đi không"
// ═══ Floyd-Warshall — O(N³) ═══
long long d[405][405];   // d[i][j] = khoảng cách ngắn nhất từ i đến j
int n;

void Floyd() {
    // Khởi tạo: d[i][j] = trọng số cạnh (i,j)
    // Nếu không có cạnh: d[i][j] = 1e18
    // d[i][i] = 0
    
    for (int k = 1; k <= n; k++)          // Đỉnh trung gian
        for (int i = 1; i <= n; i++)      // Đỉnh xuất phát
            for (int j = 1; j <= n; j++)  // Đỉnh đích
                if (d[i][k] + d[k][j] < d[i][j])
                    d[i][j] = d[i][k] + d[k][j];
}

D5. Bellman-Ford — Đường đi ngắn nhất (có trọng số ÂM)

💡 Hình dung: Giống Dijkstra — tìm đường ngắn nhất từ 1 điểm đến mọi điểm khác. Nhưng Dijkstra "phá sản" khi gặp đường có chi phí âm (đường giảm giá). Bellman-Ford chậm hơn nhưng xử lý được cả trọng số âm. Cách làm: lặp đi lặp lại (n-1) vòng, mỗi vòng duyệt TẤT CẢ các cạnh và cập nhật khoảng cách nếu tìm được đường ngắn hơn.

DẤU HIỆU: "đường đi ngắn nhất" + đồ thị có trọng số ÂM, hoặc cần phát hiện chu trình âm
SO SÁNH: Dijkstra: O(E log V), nhanh, nhưng CHỈ dùng khi trọng số ≥ 0
Bellman-Ford: O(V × E), chậm hơn, nhưng chấp nhận trọng số âm 
// ═══ Bellman-Ford — O(V × E) ═══
// Biểu diễn bằng MA TRẬN KỀ: a[u][v] = trọng số

vector<vector<int>> a;
vector<int> d;                       // d[i] = khoảng cách ngắn nhất S → i
vector<int> Trace;
int n, m, S, F;

void BellmanFord() {
    // Lặp n-1 vòng
    for (int lap = 1; lap <= n - 1; lap++) {
        bool coCapNhat = false;
        // Duyệt tất cả các cạnh (u, v)
        for (int u = 1; u <= n; u++) {
            for (int v = 1; v <= n; v++) {
                if (a[u][v] != INT_MAX && d[u] != INT_MAX) {
                    if (d[u] + a[u][v] < d[v]) {
                        d[v] = d[u] + a[u][v];
                        Trace[v] = u;
                        coCapNhat = true;
                    }
                }
            }
        }
        if (!coCapNhat) break;       // Dừng sớm nếu không cập nhật gì
    }
}

void Result() {
    if (d[F] == INT_MAX) {
        out << "Path from " << S << " to " << F << " not found";
    } else {
        out << "Distance: " << d[F] << '\n';
        // Truy vết đường đi
        vector<int> duongDi;
        int cur = F;
        while (cur != S) {
            duongDi.push_back(cur);
            cur = Trace[cur];
        }
        duongDi.push_back(S);
        reverse(duongDi.begin(), duongDi.end());
        for (int x : duongDi) out << x << " ";
    }
}

int main() {
    in >> n >> m >> S >> F;
    a.assign(n + 1, vector<int>(n + 1, INT_MAX));
    d.assign(n + 1, INT_MAX);
    Trace.assign(n + 1, S);
    for (int i = 1; i <= n; i++) a[i][i] = 0;
    while (m--) {
        int u, v, w;
        in >> u >> v >> w;
        a[u][v] = w;                 // Đồ thị CÓ HƯỚNG
    }
    d[S] = 0;
    BellmanFord();
    Result();
    return 0;
}
📝 VÍ DỤ CHẠY TAY BELLMAN-FORD: Tìm đường ngắn nhất từ đỉnh 1 đến đỉnh 4 
Đồ thị có hướng, 4 đỉnh, 5 cạnh:
1→2 (w=2), 1→3 (w=4), 2→3 (w=-3), 2→4 (w=3), 3→4 (w=2)

Khởi tạo: d = [_, 0, ∞, ∞, ∞]   (S=1)

Vòng 1: Duyệt tất cả cạnh:
  Cạnh 1→2: d[2] = min(∞, 0+2) = 2     Trace[2]=1
  Cạnh 1→3: d[3] = min(∞, 0+4) = 4     Trace[3]=1
  Cạnh 2→3: d[3] = min(4, 2+(-3)) = -1  Trace[3]=2  ← trọng số ÂM!
  Cạnh 2→4: d[4] = min(∞, 2+3) = 5     Trace[4]=2
  Cạnh 3→4: d[4] = min(5, -1+2) = 1    Trace[4]=3
  → d = [_, 0, 2, -1, 1]

Vòng 2: Không có cập nhật → Dừng sớm!

Kết quả: d[4] = 1
Truy vết: 4←Trace[4]=3←Trace[3]=2←Trace[2]=1
Đường đi: 1 → 2 → 3 → 4 (chi phí = 2 + (-3) + 2 = 1)

D6. Prim — Cây khung nhỏ nhất (Minimum Spanning Tree)

💡 Hình dung: Có n ngôi nhà và nhiều con đường nối giữa chúng, mỗi đường có chi phí xây dựng khác nhau. Cần nối TẤT CẢ các nhà lại bằng đường, sao cho tổng chi phí THẤP NHẤT và không có đường thừa (không tạo vòng). Cách làm: bắt đầu từ 1 nhà, mỗi bước chọn con đường RẺ NHẤT nối nhà đã kết nối → nhà chưa kết nối. Lặp lại cho đến khi tất cả nhà đều được nối.

DẤU HIỆU: "nối tất cả các điểm với chi phí nhỏ nhất", "cây khung nhỏ nhất", "mạng lưới tối ưu"
// ═══ Prim — O(V²) với ma trận kề ═══

vector<vector<int>> a;              // Ma trận kề: a[u][v] = trọng số
vector<bool> Free;
int n, m, total = 0;
vector<pair<int,int>> result;        // Lưu các cạnh trong cây khung

void Prim() {
    Free[1] = false;                 // Bắt đầu từ đỉnh 1
    int dem = 0;
    while (dem < n - 1) {
        int u = -1, v = -1, minC = INT_MAX;
        // Tìm cạnh nhỏ nhất nối tập đã chọn ↔ tập chưa chọn
        for (int i = 1; i <= n; i++) {
            if (!Free[i]) {          // Đỉnh đã thuộc cây khung
                for (int j = 1; j <= n; j++) {
                    if (Free[j] && a[i][j] < minC) {
                        u = i; v = j;
                        minC = a[i][j];
                    }
                }
            }
        }
        if (u == -1 || v == -1) return;  // Đồ thị không liên thông
        total += a[u][v];
        result.push_back({u, v});
        Free[v] = false;
        dem++;
    }
}

int main() {
    in >> n >> m;
    a.assign(n + 1, vector<int>(n + 1, INT_MAX));
    Free.assign(n + 1, true);
    while (m--) {
        int u, v, w;
        in >> u >> v >> w;
        a[u][v] = a[v][u] = w;      // Đồ thị VÔ HƯỚNG
    }
    Prim();
    // In kết quả
    if ((int)result.size() != n - 1) {
        out << "Do thi khong lien thong";
    } else {
        for (auto [u, v] : result)
            out << "(" << u << ", " << v << ") = " << a[u][v] << '\n';

D7. Ford-Fulkerson — Luồng cực đại (Maximum Flow)

💡 Hình dung: Mạng ống nước — nước chảy từ bể chứa (nguồn S) đến nhà máy (đích F). Mỗi ống có sức chứa tối đa (capacity). Hỏi: tổng lượng nước lớn nhất có thể chảy từ S đến F? Cách làm: liên tục tìm đường còn chỗ trống từ S→F (bằng BFS), rồi đẩy nước qua đường đó. Lặp lại đến khi không còn đường nào nữa.

DẤU HIỆU: "luồng cực đại", "lưu lượng tối đa", "cắt nhỏ nhất" (max-flow min-cut)
// ═══ Ford-Fulkerson (dùng BFS — Edmonds-Karp) — O(V × E²) ═══

vector<vector<int>> Cap, Flow;      // Cap = sức chứa, Flow = luồng hiện tại
vector<int> Trace;
int n, m, S, F;

bool BFS_TimDuong() {
    Trace.assign(n + 1, -1);
    queue<int> q;
    q.push(S);
    Trace[S] = S;
    while (!q.empty()) {
        int u = q.front(); q.pop();
        for (int v = 1; v <= n; v++) {
            if (Trace[v] == -1 && Cap[u][v] > Flow[u][v]) {
                Trace[v] = u;
                if (v == F) return true;   // Tìm được đường tăng
                q.push(v);
            }
        }
    }
    return false;                          // Không còn đường tăng
}

void MaxFlow() {
    while (BFS_TimDuong()) {
        // Tìm lượng nước nhỏ nhất trên đường tăng
        int minCap = INT_MAX;
        for (int v = F; v != S; v = Trace[v]) {
            int u = Trace[v];
            minCap = min(minCap, Cap[u][v] - Flow[u][v]);
        }
        // Cập nhật luồng dọc đường tăng
        for (int v = F; v != S; v = Trace[v]) {
            int u = Trace[v];
            Flow[u][v] += minCap;
            Flow[v][u] -= minCap;          // Luồng ngược
        }
    }
}

int main() {
    in >> n >> m >> S >> F;
    Cap.assign(n + 1, vector<int>(n + 1, 0));
    Flow.assign(n + 1, vector<int>(n + 1, 0));
    while (m--) {
        int u, v, w;
        in >> u >> v >> w;
        Cap[u][v] += w;                    // Hỗ trợ cạnh song song
    }
    MaxFlow();
    // Tính tổng luồng ra từ nguồn S
    int res = 0;
    for (int i = 1; i <= n; i++) res += Flow[S][i];
    out << "Max flow: " << res;
    return 0;
}

D8. Kruskal — Cây khung nhỏ nhất (dùng DSU)

💡 Hình dung: Giống Prim (nối tất cả nhà với chi phí thấp nhất), nhưng cách làm khác: liệt kê TẤT CẢ các con đường, sắp theo chi phí tăng dần, rồi chọn từng con đường rẻ nhất mà không tạo vòng. Dùng DSU để kiểm tra "2 đầu đường đã nối chưa?".

DẤU HIỆU: Giống Prim. Khi nào chọn Kruskal? Khi đồ thị có nhiều cạnh, ít đỉnh hoặc đã có sẵn DSU.
// ═══ Kruskal — O(E log E) dùng DSU ═══

struct Edge {
    int u, v, w;
    bool operator<(const Edge &o) const { return w < o.w; }
};

int cha[100005], hang[100005];

int tim_goc(int v) {
    if (v == cha[v]) return v;
    return cha[v] = tim_goc(cha[v]);
}

void hop_nhat(int a, int b) {
    a = tim_goc(a); b = tim_goc(b);
    if (a == b) return;
    if (hang[a] < hang[b]) swap(a, b);
    cha[b] = a;
    if (hang[a] == hang[b]) hang[a]++;
}

int main() {
    ifstream in("TENBAI.INP");
    ofstream out("TENBAI.OUT");
    int n, m;
    in >> n >> m;
    
    vector<Edge> edges(m);
    for (int i = 0; i < m; i++)
        in >> edges[i].u >> edges[i].v >> edges[i].w;
    
    sort(edges.begin(), edges.end());  // Sắp cạnh tăng theo trọng số
    
    for (int i = 1; i <= n; i++) { cha[i] = i; hang[i] = 0; }
    
    long long total = 0;
    int dem = 0;
    for (auto &[u, v, w] : edges) {
        if (tim_goc(u) != tim_goc(v)) {   // Không cùng nhóm → không tạo vòng
            hop_nhat(u, v);
            total += w;
            dem++;
            if (dem == n - 1) break;       // Đủ n-1 cạnh = xong
        }
    }
    
    if (dem == n - 1)
        out << "Tong trong so: " << total;
    else
        out << "Do thi khong lien thong";
    return 0;
}

D9. Topo Sort — Sắp xếp topo (Thứ tự thực hiện)

💡 Hình dung: Trong môn học, muốn học "Giải tích 2" thì phải học "Giải tích 1" trước. Muốn học "CTDL" thì phải học "Lập trình cơ bản" trước. Sắp xếp topo = tìm thứ tự học môn sao cho không vi phạm tiên quyết. Dùng BFS với "đếm bậc vào" (Kahn's algorithm).

DẤU HIỆU: "thứ tự thực hiện công việc", "môn tiên quyết", "đồ thị có hướng không chu trình (DAG)"
// ═══ Topo Sort (Kahn's Algorithm — BFS) — O(V + E) ═══

int n, m;
vector<vector<int>> ke;        // Danh sách kề (đồ thị CÓ HƯỚNG)
vector<int> bacVao;            // bacVao[v] = số cạnh đi VÀO v

void TopoSort() {
    queue<int> q;
    // Bước 1: Đẩy các đỉnh có bậc vào = 0 vào queue
    for (int i = 1; i <= n; i++)
        if (bacVao[i] == 0) q.push(i);
    
    vector<int> ketQua;
    while (!q.empty()) {
        int u = q.front(); q.pop();
        ketQua.push_back(u);
        
        // Bước 2: Xóa đỉnh u → giảm bậc vào các đỉnh kề
        for (int v : ke[u]) {
            bacVao[v]--;
            if (bacVao[v] == 0) q.push(v);
        }
    }
    
    if ((int)ketQua.size() != n) {
        out << "Co chu trinh!";       // Không phải DAG
    } else {
        for (int x : ketQua) out << x << " ";
    }
}

int main() {
    ifstream in("TENBAI.INP");
    ofstream out("TENBAI.OUT");
    in >> n >> m;
    ke.assign(n + 1, vector<int>());
    bacVao.assign(n + 1, 0);
    while (m--) {
        int u, v;
        in >> u >> v;                  // Cạnh u → v (u phải xong trước v)
        ke[u].push_back(v);
        bacVao[v]++;
    }
    TopoSort();
    return 0;
}

bool laSNT(long long n) {

long long power(long long a, long long n, long long mod) {
    long long res = 1;
    a %= mod;
    while (n > 0) {
        if (n & 1) res = res * a % mod;
        a = a * a % mod;
        n >>= 1;
    }
    return res;
}
