Bài 1: BẬC ĐỈNH
BACDINH.INP / BACDINH.OUT

Cho đồ thị vô hướng gồm N đỉnh (đánh số 1 đến N) và M cạnh. Hãy tính bậc (số cạnh nối) của mỗi đỉnh.

Dữ liệu vào (BACDINH.INP):
  • Dòng 1: Hai số nguyên N, M (1 ≤ N ≤ 105; 0 ≤ M ≤ 105)
  • M dòng tiếp theo: Mỗi dòng hai số u, v — cạnh nối đỉnh u và v
Kết quả ra (BACDINH.OUT):
  • Một dòng gồm N số nguyên: bậc của đỉnh 1, 2, ..., N (cách nhau bởi dấu cách)

BACDINH.INP

5 4
1 2
1 3
2 3
3 4

BACDINH.OUT

2 2 3 1 0
📝 Code C++ — D0. Biểu diễn đồ thị 
// Bài 1: BACDINH — Biểu diễn đồ thị, tính bậc đỉnh
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("BACDINH.INP");
    ofstream out("BACDINH.OUT");

    int n, m;
    in >> n >> m;

    vector<int> deg(n + 1, 0);
    for(int i = 0; i < m; i++) {
        int u, v;
        in >> u >> v;
        deg[u]++;
        deg[v]++;
    }

    for(int i = 1; i <= n; i++) {
        if(i > 1) out << " ";
        out << deg[i];
    }

    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: Đọc từng cạnh (u, v), tăng bậc cả hai đỉnh. Độ phức tạp: O(N + M).
Bài 2: TÌM ĐƯỜNG
TIMDUONG.INP / TIMDUONG.OUT

Cho đồ thị vô hướng không trọng số gồm N đỉnh và M cạnh. Tìm khoảng cách ngắn nhất (số cạnh ít nhất) từ đỉnh S đến đỉnh T. Nếu không tồn tại đường đi, in -1.

Dữ liệu vào (TIMDUONG.INP):
  • Dòng 1: Bốn số nguyên N, M, S, T (1 ≤ N ≤ 105; 0 ≤ M ≤ 2×105)
  • M dòng tiếp theo: Mỗi dòng hai số u, v — cạnh nối u và v
Kết quả ra (TIMDUONG.OUT):
  • Một số nguyên duy nhất là khoảng cách ngắn nhất, hoặc -1

📝 Code C++ — D1. BFS 
// Bài 2: TIMDUONG — BFS tìm đường ngắn nhất
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("TIMDUONG.INP");
    ofstream out("TIMDUONG.OUT");

    int n, m, s, t;
    in >> n >> m >> s >> t;

    vector<vector<int>> adj(n + 1);
    for(int i = 0; i < m; i++) {
        int u, v;
        in >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    vector<int> dist(n + 1, -1);
    queue<int> q;
    dist[s] = 0;
    q.push(s);

    while(!q.empty()) {
        int u = q.front(); q.pop();
        for(int i = 0; i < (int)adj[u].size(); i++) {
            int v = adj[u][i];
            if(dist[v] == -1) {
                dist[v] = dist[u] + 1;
                q.push(v);
            }
        }
    }

    out << dist[t];
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: BFS từ đỉnh S. Mỗi đỉnh được thăm đúng 1 lần, dist[v] = dist[u] + 1. Đảm bảo đường ngắn nhất trên đồ thị không trọng số. Độ phức tạp: O(N + M).
Bài 3: CHU TRÌNH
CHUTRINH.INP / CHUTRINH.OUT

Cho đồ thị có hướng gồm N đỉnh và M cung. Hãy kiểm tra đồ thị có chứa chu trình hay không.

Dữ liệu vào (CHUTRINH.INP):
  • Dòng 1: Hai số nguyên N, M (1 ≤ N ≤ 105; 0 ≤ M ≤ 2×105)
  • M dòng tiếp theo: Mỗi dòng hai số u, v — cung từ u đến v
Kết quả ra (CHUTRINH.OUT):
  • In "YES" nếu có chu trình, ngược lại in "NO"

1 2
2 3
3 4
4 2

CHUTRINH.OUT

📝 Code C++ — D2. DFS phát hiện chu trình 
// Bài 3: CHUTRINH — DFS tô màu phát hiện chu trình
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 100005;
vector<int> adj[MAXN];
int color[MAXN]; // 0=trắng, 1=xám, 2=đen
bool hasCycle;

void dfs(int u) {
    color[u] = 1;
    for(int i = 0; i < (int)adj[u].size(); i++) {
        int v = adj[u][i];
        if(color[v] == 1) { hasCycle = true; return; }
        if(color[v] == 0) { dfs(v); if(hasCycle) return; }
    }
    color[u] = 2;
}

int main() {
    ifstream in("CHUTRINH.INP");
    ofstream out("CHUTRINH.OUT");

    int n, m;
    in >> n >> m;
    for(int i = 0; i < m; i++) {
        int u, v;
        in >> u >> v;
        adj[u].push_back(v);
    }

    hasCycle = false;
    memset(color, 0, sizeof(color));
    for(int i = 1; i <= n; i++) {
        if(color[i] == 0) {
            dfs(i);
            if(hasCycle) break;
        }
    }

    out << (hasCycle ? "YES" : "NO");
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: DFS tô 3 màu: trắng (chưa thăm), xám (đang thăm), đen (đã xong). Nếu gặp đỉnh xám → có chu trình. Độ phức tạp: O(N + M).

📝 Code C++ — D3. Liên thông (BFS) 
// Bài 4: LIENTHONG — Đếm thành phần liên thông
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("LIENTHONG.INP");
    ofstream out("LIENTHONG.OUT");

    int n, m;
    in >> n >> m;

    vector<vector<int>> adj(n + 1);
    for(int i = 0; i < m; i++) {
        int u, v;
        in >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    vector<bool> visited(n + 1, false);
    int cnt = 0;

    for(int i = 1; i <= n; i++) {
        if(!visited[i]) {
            cnt++;
            queue<int> q;
            visited[i] = true;
            q.push(i);
            while(!q.empty()) {
                int u = q.front(); q.pop();
                for(int j = 0; j < (int)adj[u].size(); j++) {
                    int v = adj[u][j];
                    if(!visited[v]) {
                        visited[v] = true;
                        q.push(v);
                    }
                }
            }
        }
    }

    out << cnt;
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: Với mỗi đỉnh chưa thăm, chạy BFS/DFS để đánh dấu toàn bộ thành phần liên thông. Đếm số lần khởi tạo BFS mới. Độ phức tạp: O(N + M).
Bài 5: MÊ CUNG
MECUNG.INP / MECUNG.OUT

Cho mê cung hình chữ nhật kích thước M×N ô. Mỗi ô là đường đi ('.') hoặc tường ('#'). Một nhân vật đứng ở ô (1,1) và muốn đi đến ô (M,N). Mỗi bước, nhân vật di chuyển sang một ô kề cạnh (lên, xuống, trái, phải).

Hãy tìm số bước đi ít nhất. Nếu không thể đến được, in -1.

Dữ liệu vào (MECUNG.INP):
  • Dòng 1: Hai số nguyên M, N (1 ≤ M, N ≤ 1000)
  • M dòng tiếp theo: Mỗi dòng một xâu N ký tự ('.' hoặc '#')
Kết quả ra (MECUNG.OUT):
  • Một số nguyên duy nhất là số bước ít nhất, hoặc -1

📝 Code C++ — D4. BFS trên lưới 2D 
// Bài 5: MECUNG — BFS trên lưới 2D
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("MECUNG.INP");
    ofstream out("MECUNG.OUT");

    int m, n;
    in >> m >> n;
    vector<string> grid(m);
    for(int i = 0; i < m; i++) in >> grid[i];

    vector<vector<int>> dist(m, vector<int>(n, -1));
    queue<pair<int,int>> q;
    dist[0][0] = 0;
    q.push(make_pair(0, 0));

    int dx[] = {0, 0, 1, -1};
    int dy[] = {1, -1, 0, 0};

    while(!q.empty()) {
        pair<int,int> cur = q.front(); q.pop();
        for(int d = 0; d < 4; d++) {
            int nx = cur.first + dx[d];
            int ny = cur.second + dy[d];
            if(nx >= 0 && nx < m && ny >= 0 && ny < n
               && grid[nx][ny] == '.' && dist[nx][ny] == -1) {
                dist[nx][ny] = dist[cur.first][cur.second] + 1;
                q.push(make_pair(nx, ny));
            }
        }
    }

    out << dist[m - 1][n - 1];
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: Coi mỗi ô '.' là một đỉnh, hai ô kề cạnh là một cạnh. Chạy BFS trên lưới từ (0,0) → tìm dist[m-1][n-1]. Độ phức tạp: O(M × N).

📝 Code C++ — D5. Dijkstra 
// Bài 6: THANHPHO — Dijkstra đường đi ngắn nhất
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("THANHPHO.INP");
    ofstream out("THANHPHO.OUT");

    int n, m;
    in >> n >> m;

    vector<vector<pair<int,int>>> adj(n + 1);
    for(int i = 0; i < m; i++) {
        int u, v, w;
        in >> u >> v >> w;
        adj[u].push_back(make_pair(v, w));
        adj[v].push_back(make_pair(u, w));
    }

    const long long INF = 1e18;
    vector<long long> dist(n + 1, INF);
    priority_queue<pair<long long,int>,
        vector<pair<long long,int>>,
        greater<pair<long long,int>>> pq;

    dist[1] = 0;
    pq.push(make_pair(0LL, 1));

    while(!pq.empty()) {
        long long d = pq.top().first;
        int u = pq.top().second;
        pq.pop();
        if(d > dist[u]) continue;
        for(int i = 0; i < (int)adj[u].size(); i++) {
            int v = adj[u][i].first;
            int w = adj[u][i].second;
            if(dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                pq.push(make_pair(dist[v], v));
            }
        }
    }

    out << (dist[n] >= INF ? -1LL : dist[n]);
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: Dijkstra với hàng đợi ưu tiên (min-heap). Mỗi bước lấy đỉnh có dist nhỏ nhất để cập nhật các đỉnh kề. Chỉ áp dụng cho trọng số dương. Độ phức tạp: O((N + M) log N).

📝 Code C++ — D6. Bellman-Ford 
// Bài 7: CHIPHI — Bellman-Ford (trọng số âm)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("CHIPHI.INP");
    ofstream out("CHIPHI.OUT");

    int n, m;
    in >> n >> m;

    vector<int> eu(m), ev(m), ew(m);
    for(int i = 0; i < m; i++)
        in >> eu[i] >> ev[i] >> ew[i];

    const long long INF = 1e18;
    vector<long long> dist(n + 1, INF);
    dist[1] = 0;

    for(int iter = 0; iter < n - 1; iter++) {
        for(int i = 0; i < m; i++) {
            if(dist[eu[i]] < INF &&
               dist[eu[i]] + ew[i] < dist[ev[i]]) {
                dist[ev[i]] = dist[eu[i]] + ew[i];
            }
        }
    }

    if(dist[n] >= INF)
        out << "NO";
    else
        out << dist[n];

    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: Bellman-Ford: lặp N-1 vòng, mỗi vòng duyệt tất cả cạnh và cập nhật dist. Hỗ trợ trọng số âm (không có chu trình âm). Độ phức tạp: O(N × M).
Bài 8: KHOẢNG CÁCH

📝 Code C++ — D7. Floyd-Warshall 
// Bài 8: KHOANGCACH — Floyd all-pairs shortest path
#include <bits/stdc++.h>
using namespace std;

const long long INF = 1e18;
long long dist[405][405];

int main() {
    ifstream in("KHOANGCACH.INP");
    ofstream out("KHOANGCACH.OUT");

    int n, m, q;
    in >> n >> m >> q;

    for(int i = 1; i <= n; i++)
        for(int j = 1; j <= n; j++)
            dist[i][j] = (i == j) ? 0 : INF;

    for(int i = 0; i < m; i++) {
        int u, v;
        long long w;
        in >> u >> v >> w;
        dist[u][v] = min(dist[u][v], w);
        dist[v][u] = min(dist[v][u], w);
    }

    // Floyd-Warshall
    for(int k = 1; k <= n; k++)
        for(int i = 1; i <= n; i++)
            for(int j = 1; j <= n; j++)
                if(dist[i][k] + dist[k][j] < dist[i][j])
                    dist[i][j] = dist[i][k] + dist[k][j];

    for(int i = 0; i < q; i++) {
        int u, v;
        in >> u >> v;
        out << (dist[u][v] >= INF ? -1LL : dist[u][v]) << "\n";
    }

    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: Floyd-Warshall: 3 vòng lặp k, i, j. Cập nhật dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j]). Tính đường ngắn nhất giữa mọi cặp đỉnh. Độ phức tạp: O(N³).

// Bài 9: MANGLUOI — Prim MST
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("MANGLUOI.INP");
    ofstream out("MANGLUOI.OUT");

    int n, m;
    in >> n >> m;

    vector<vector<pair<int,int>>> adj(n + 1);
    for(int i = 0; i < m; i++) {
        int u, v, w;
        in >> u >> v >> w;
        adj[u].push_back(make_pair(v, w));
        adj[v].push_back(make_pair(u, w));
    }

    vector<bool> inMST(n + 1, false);
    priority_queue<pair<int,int>,
        vector<pair<int,int>>,
        greater<pair<int,int>>> pq;
    pq.push(make_pair(0, 1));

    long long total = 0;
    int cnt = 0;

    while(!pq.empty() && cnt < n) {
        int w = pq.top().first;
        int u = pq.top().second;
        pq.pop();
        if(inMST[u]) continue;
        inMST[u] = true;
        total += w;
        cnt++;
        for(int i = 0; i < (int)adj[u].size(); i++) {
            int v = adj[u][i].first;
            int ww = adj[u][i].second;
            if(!inMST[v]) pq.push(make_pair(ww, v));
        }
    }

    out << (cnt < n ? -1LL : total);
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: Prim dùng min-heap. Bắt đầu từ đỉnh 1, mỗi bước thêm cạnh trọng số nhỏ nhất nối cây hiện tại với đỉnh mới. Độ phức tạp: O((N + M) log N).

📝 Code C++ — D9. Kruskal + DSU 
// Bài 10: NOICAU — Kruskal + Union-Find (DSU)
#include <bits/stdc++.h>
using namespace std;

int par[100005];

int find(int x) {
    if(par[x] != x) par[x] = find(par[x]);
    return par[x];
}

bool unite(int x, int y) {
    x = find(x); y = find(y);
    if(x == y) return false;
    par[x] = y;
    return true;
}

struct Edge { int u, v, w; };

bool cmpEdge(const Edge &a, const Edge &b) {
    return a.w < b.w;
}

int main() {
    ifstream in("NOICAU.INP");
    ofstream out("NOICAU.OUT");

    int n, m;
    in >> n >> m;

    vector<Edge> edges(m);
    for(int i = 0; i < m; i++)
        in >> edges[i].u >> edges[i].v >> edges[i].w;

    sort(edges.begin(), edges.end(), cmpEdge);
    for(int i = 1; i <= n; i++) par[i] = i;

    long long total = 0;
    int cnt = 0;
    for(int i = 0; i < m && cnt < n - 1; i++) {
        if(unite(edges[i].u, edges[i].v)) {
            total += edges[i].w;
            cnt++;
        }
    }

    out << (cnt < n - 1 ? -1LL : total);
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: Kruskal: sắp xếp cạnh theo trọng số tăng dần, dùng Disjoint Set Union (DSU) để kiểm tra/hợp nhất thành phần liên thông. Chọn N-1 cạnh nhẹ nhất không tạo chu trình. Độ phức tạp: O(M log M).

// Bài 11: MONHOC — Topo Sort (Kahn's BFS, ưu tiên nhỏ)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("MONHOC.INP");
    ofstream out("MONHOC.OUT");

    int n, m;
    in >> n >> m;

    vector<vector<int>> adj(n + 1);
    vector<int> indeg(n + 1, 0);
    for(int i = 0; i < m; i++) {
        int u, v;
        in >> u >> v;
        adj[u].push_back(v);
        indeg[v]++;
    }

    // Min-heap để ưu tiên số nhỏ
    priority_queue<int, vector<int>, greater<int>> pq;
    for(int i = 1; i <= n; i++)
        if(indeg[i] == 0) pq.push(i);

    vector<int> order;
    while(!pq.empty()) {
        int u = pq.top(); pq.pop();
        order.push_back(u);
        for(int i = 0; i < (int)adj[u].size(); i++) {
            int v = adj[u][i];
            indeg[v]--;
            if(indeg[v] == 0) pq.push(v);
        }
    }

    if((int)order.size() < n) {
        out << -1;
    } else {
        for(int i = 0; i < n; i++) {
            if(i > 0) out << " ";
            out << order[i];
        }
    }

    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: Kahn's Algorithm (BFS-based Topo Sort): dùng min-heap chứa đỉnh có bậc vào = 0. Lấy đỉnh nhỏ nhất, giảm bậc vào hàng xóm. Nếu output < N đỉnh → có chu trình. Độ phức tạp: O((N + M) log N).

📝 Code C++ — D11. Edmonds-Karp (Ford-Fulkerson BFS) 
// Bài 12: LUONGMAX — Edmonds-Karp (Max Flow)
#include <bits/stdc++.h>
using namespace std;

int cap[105][105];
int n, m, S, T;

int bfs(vector<int> &parent) {
    fill(parent.begin(), parent.end(), -1);
    parent[S] = S;
    queue<pair<int,int>> q;
    q.push(make_pair(S, 1000000000));
    while(!q.empty()) {
        int u = q.front().first;
        int flow = q.front().second;
        q.pop();
        for(int v = 1; v <= n; v++) {
            if(parent[v] == -1 && cap[u][v] > 0) {
                parent[v] = u;
                int new_flow = min(flow, cap[u][v]);
                if(v == T) return new_flow;
                q.push(make_pair(v, new_flow));
            }
        }
    }
    return 0;
}

int main() {
    ifstream in("LUONGMAX.INP");
    ofstream out("LUONGMAX.OUT");

    in >> n >> m >> S >> T;
    memset(cap, 0, sizeof(cap));
    for(int i = 0; i < m; i++) {
        int u, v, c;
        in >> u >> v >> c;
        cap[u][v] += c;
    }

    long long maxflow = 0;
    vector<int> parent(n + 1);
    int aug;
    while((aug = bfs(parent)) > 0) {
        maxflow += aug;
        int v = T;
        while(v != S) {
            int u = parent[v];
            cap[u][v] -= aug;
            cap[v][u] += aug;
            v = u;
        }
    }

    out << maxflow;
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: Edmonds-Karp (Ford-Fulkerson + BFS): lặp tìm đường tăng luồng bằng BFS trên đồ thị thặng dư. Mỗi lần tăng luồng bằng bottleneck. Cập nhật cung thuận/nghịch. Độ phức tạp: O(V × E²).

// Bài 13: HAIPHAN — BFS tô màu kiểm tra bipartite
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("HAIPHAN.INP");
    ofstream out("HAIPHAN.OUT");

    int n, m;
    in >> n >> m;

    vector<vector<int>> adj(n + 1);
    for(int i = 0; i < m; i++) {
        int u, v;
        in >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    vector<int> color(n + 1, -1);
    bool bipartite = true;

    for(int i = 1; i <= n && bipartite; i++) {
        if(color[i] != -1) continue;
        queue<int> q;
        color[i] = 0;
        q.push(i);
        while(!q.empty() && bipartite) {
            int u = q.front(); q.pop();
            for(int j = 0; j < (int)adj[u].size(); j++) {
                int v = adj[u][j];
                if(color[v] == -1) {
                    color[v] = 1 - color[u];
                    q.push(v);
                } else if(color[v] == color[u]) {
                    bipartite = false;
                }
            }
        }
    }

    out << (bipartite ? "YES" : "NO");
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: BFS tô 2 màu: tô đỉnh kề bằng màu đối. Nếu gặp đỉnh kề cùng màu → không phải hai phía. Đồ thị hai phía ⇔ không chứa chu trình lẻ. Độ phức tạp: O(N + M).

// Bài 14: LANDICH — Multi-source BFS
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("LANDICH.INP");
    ofstream out("LANDICH.OUT");

    int m, n, k;
    in >> m >> n >> k;

    vector<vector<int>> dist(m, vector<int>(n, -1));
    queue<pair<int,int>> q;

    for(int i = 0; i < k; i++) {
        int r, c;
        in >> r >> c;
        r--; c--;
        dist[r][c] = 0;
        q.push(make_pair(r, c));
    }

    int dx[] = {0, 0, 1, -1};
    int dy[] = {1, -1, 0, 0};
    int ans = 0;

    while(!q.empty()) {
        pair<int,int> cur = q.front(); q.pop();
        for(int d = 0; d < 4; d++) {
            int nx = cur.first + dx[d];
            int ny = cur.second + dy[d];
            if(nx >= 0 && nx < m && ny >= 0 && ny < n
               && dist[nx][ny] == -1) {
                dist[nx][ny] = dist[cur.first][cur.second] + 1;
                if(dist[nx][ny] > ans) ans = dist[nx][ny];
                q.push(make_pair(nx, ny));
            }
        }
    }

    out << ans;
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: Multi-source BFS: đưa tất cả K ổ dịch vào hàng đợi ban đầu (dist = 0), sau đó BFS bình thường. Kết quả = max dist trong toàn lưới. Độ phức tạp: O(M × N).

// Bài 15: DEMDUONG — Topo Sort + DP đếm đường trên DAG
#include <bits/stdc++.h>
using namespace std;

const int MOD = 1e9 + 7;

int main() {
    ifstream in("DEMDUONG.INP");
    ofstream out("DEMDUONG.OUT");

    int n, m;
    in >> n >> m;

    vector<vector<int>> adj(n + 1);
    vector<int> indeg(n + 1, 0);
    for(int i = 0; i < m; i++) {
        int u, v;
        in >> u >> v;
        adj[u].push_back(v);
        indeg[v]++;
    }

    // Topo sort (Kahn)
    queue<int> q;
    for(int i = 1; i <= n; i++)
        if(indeg[i] == 0) q.push(i);

    vector<int> order;
    while(!q.empty()) {
        int u = q.front(); q.pop();
        order.push_back(u);
        for(int i = 0; i < (int)adj[u].size(); i++) {
            int v = adj[u][i];
            indeg[v]--;
            if(indeg[v] == 0) q.push(v);
        }
    }

    // DP đếm đường
    vector<long long> dp(n + 1, 0);
    dp[1] = 1;
    for(int i = 0; i < (int)order.size(); i++) {
        int u = order[i];
        for(int j = 0; j < (int)adj[u].size(); j++) {
            int v = adj[u][j];
            dp[v] = (dp[v] + dp[u]) % MOD;
        }
    }

    out << dp[n];
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: Sắp xếp topo, rồi DP theo thứ tự topo: dp[v] += dp[u] với mỗi cung u→v. dp[1] = 1. Kết hợp Topological Sort và DP trên DAG. Độ phức tạp: O(N + M).
