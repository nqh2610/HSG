Bài 1: Mã hóa (MAHOA)
📝 Code C++ — Xử lý xâu nén O(|S|) 
// Bai 1: MAHOA — String RLE decode, find char at position K
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("MAHOA.INP");
    ofstream out("MAHOA.OUT");

    string s;
    long long k;
    in >> s >> k;

    int i = 0;
    long long pos = 0;
    while (i < (int)s.size()) {
        char ch = s[i];
        i++;
        long long num = 0;
        while (i < (int)s.size() && s[i] >= '0' && s[i] <= '9') {
            num = num * 10 + (s[i] - '0');
            i++;
        }
        if (num == 0) num = 1;
        if (pos + num >= k) {
            out << ch;
            in.close();
            out.close();
            return 0;
        }
        pos += num;
    }

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (xâu giải nén ≤ 10⁶): Giải nén thật ra string rồi lấy ký tự tại vị trí K. Đơn giản nhưng tốn bộ nhớ.
Subtask 2 (K ≤ 10¹⁸): KHÔNG được giải nén — xâu có thể dài hàng tỷ tỷ ký tự!

Ý tưởng: Duyệt qua xâu nén, đọc từng cặp (ký tự, số lần). Duy trì biến pos = số ký tự đã "đi qua".
  • Đọc ký tự ch và số lần num
  • Nếu pos + num ≥ K → ký tự thứ K chính là ch → xuất và dừng
  • Ngược lại: pos += num, tiếp tục

Kỹ thuật parse: Về xâu "x10y5": ký tự 'x' theo sau bởi "10" (2 chữ số). Phải đọc hết chữ số: num = num × 10 + (s[i] − '0').

Tại sao dùng long long? K có thể đạt 10¹⁸, vượt xa int (≈ 2 × 10⁹). Biến pos và num phải là long long.

Độ phức tạp: O(|S|) — duyệt xâu nén đúng 1 lần, không giải nén.
Bài 2: Cứu hộ (CUUHO)
📝 Code C++ — Re-rooting DP on Tree O(N) 
// Bai 2: CUUHO — Re-rooting DP on Tree O(N)
#include <bits/stdc++.h>
using namespace std;

int n;
vector<pair<int, long long> > adj[100001];
long long sz[100001], dist_arr[100001], dp[100001];

void dfs1(int u, int p, long long d) {
    sz[u] = 1;
    dist_arr[u] = d;
    for (int i = 0; i < (int)adj[u].size(); i++) {
        int v = adj[u][i].first;
        long long w = adj[u][i].second;
        if (v != p) {
            dfs1(v, u, d + w);
            sz[u] += sz[v];
        }
    }
}

void dfs2(int u, int p) {
    for (int i = 0; i < (int)adj[u].size(); i++) {
        int v = adj[u][i].first;
        long long w = adj[u][i].second;
        if (v != p) {
            dp[v] = dp[u] + (n - 2 * sz[v]) * w;
            dfs2(v, u);
        }
    }
}

int main() {
    ifstream in("CUUHO.INP");
    ofstream out("CUUHO.OUT");

    in >> n;
    for (int i = 0; i < n - 1; i++) {
        int u, v;
        long long w;
        in >> u >> v >> w;
        adj[u].push_back(make_pair(v, w));
        adj[v].push_back(make_pair(u, w));
    }

    dfs1(1, 0, 0);
    dp[1] = 0;
    for (int i = 1; i <= n; i++) dp[1] += dist_arr[i];
    dfs2(1, 0);

    long long ans = dp[1];
    for (int i = 2; i <= n; i++) ans = min(ans, dp[i]);
    out << ans;

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 1000): BFS/DFS từ mỗi đỉnh, tính tổng khoảng cách. O(N²) ≈ 10⁶ — đủ nhanh.
Subtask 2 (Full): Kỹ thuật Re-rooting (đổi gốc) trên cây O(N):

Bước 1 — DFS từ gốc (dfs1):
  • Chọn gốc = 1, tính dist_arr[v] = khoảng cách từ 1 đến v, và sz[v] = số đỉnh trong cây con gốc v
  • dp[1] = tổng dist_arr[i] ∀ i = tổng khoảng cách từ gốc 1

Bước 2 — Re-rooting (dfs2):
Khi "chuyển gốc" từ u sang con v (cạnh u–v trọng số w):
  • sz[v] đỉnh trong cây con v → gần hơn w khi gốc chuyển sang v
  • (n − sz[v]) đỉnh còn lại → xa hơn w
→ dp[v] = dp[u] − sz[v] × w + (n − sz[v]) × w = dp[u] + (n − 2 × sz[v]) × w

Ví dụ: Cây 5 đỉnh, gốc 1: dp[1] = 9. Chuyển sang gốc 2 (w=1, sz[2]=3): dp[2] = 9 + (5 − 6) × 1 = 8. Đúng!

Tại sao đúng? Khi gốc dời từ u sang v, mỗi đỉnh trong subtree(v) giảm w, mỗi đỉnh ngoài subtree(v) tăng w. Tổng thay đổi = (n − 2 × sz[v]) × w.

Đáp án: min(dp[i]) ∀ i.
Độ phức tạp: O(N) — 2 lần DFS, mỗi lần O(N).
Bài 3: Mở đường (MODUONG)
📝 Code C++ — 0-1 BFS O(N×M) 
// Bai 3: MODUONG — 0-1 BFS O(N*M)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("MODUONG.INP");
    ofstream out("MODUONG.OUT");

    int n, m;
    in >> n >> m;
    vector<string> grid(n);
    for (int i = 0; i < n; i++) in >> grid[i];

    vector<vector<int> > d(n, vector<int>(m, 1000000));
    deque<pair<int, int> > dq;
    d[0][0] = 0;
    dq.push_front(make_pair(0, 0));

    int dx[] = {0, 0, 1, -1};
    int dy[] = {1, -1, 0, 0};

    while (!dq.empty()) {
        int x = dq.front().first, y = dq.front().second;
        dq.pop_front();
        for (int i = 0; i < 4; i++) {
            int nx = x + dx[i], ny = y + dy[i];
            if (nx < 0 || nx >= n || ny < 0 || ny >= m) continue;
            int w = (grid[nx][ny] == '#') ? 1 : 0;
            if (d[x][y] + w < d[nx][ny]) {
                d[nx][ny] = d[x][y] + w;
                if (w == 0) dq.push_front(make_pair(nx, ny));
                else dq.push_back(make_pair(nx, ny));
            }
        }
    }

    out << d[n - 1][m - 1];

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N, M ≤ 50): Dijkstra trên lưới. Mỗi ô là đỉnh, cạnh trọng 0 (đường) hoặc 1 (tường). O(NM × log(NM)) ≈ 10⁴ × 13 ≈ 10⁵.
Subtask 2 (Full): 0-1 BFS — BFS biến thể với deque cho đồ thị có trọng số 0 hoặc 1:

Ý tưởng: Thay vì priority queue (Dijkstra), dùng deque:
  • Cạnh trọng 0 (ô trống) → đẩy vào đầu deque (push_front)
  • Cạnh trọng 1 (phá tường) → đẩy vào cuối deque (push_back)

Tại sao đúng? Deque luôn giữ thứ tự tăng dần: các đỉnh chi phí nhỏ ở đầu, chi phí lớn ở cuối. Lấy từ đầu → đảm bảo xử lý đỉnh theo thứ tự chi phí tăng dần (giống Dijkstra).

So sánh hiệu năng:
  • Dijkstra: O(NM × log(NM)) — priority queue overhead
  • 0-1 BFS: O(NM) — deque, mỗi ô vào/ra tối đa 1 lần
→ Nhanh hơn Dijkstra vì trọng số chỉ có 0 và 1.

Ví dụ: Lưới 3×3a ".#. / ### / .#.": phải phá ít nhất 2 bức tường. Deque BFS tìm ra d[2][2] = 2.

Lưu ý: Ô (1,1) luôn là '.' → d[0][0] = 0. Không cần kiểm tra ô ban đầu.
Độ phức tạp: O(N × M) ≈ 10⁶ — rất nhanh.
