Bài 1: Cặp may mắn (DEMCAP)
📝 Code C++ — Mảng Counting (đếm số dư) O(N + K) 
// Bai 1: DEMCAP — Counting Array (Remainder frequency) O(N + K)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("DEMCAP.INP");
    ofstream out("DEMCAP.OUT");

    int n, k;
    in >> n >> k;

    vector<long long> cnt(k, 0);
    for (int i = 0; i < n; i++) {
        long long a;
        in >> a;
        cnt[a % k]++;
    }

    long long ans = cnt[0] * (cnt[0] - 1) / 2;
    for (int r = 1; r * 2 < k; r++) {
        ans += cnt[r] * cnt[k - r];
    }
    if (k % 2 == 0) {
        ans += cnt[k / 2] * (cnt[k / 2] - 1) / 2;
    }

    out << ans;

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 1000): Duyệt tất cả cặp (i, j), kiểm tra (a[i]+a[j]) % K == 0: O(N²) ≈ 10⁶.
Subtask 2 (Full): Kỹ thuật Mảng Counting (đếm tần suất số dư):

Nhận xét: (a[i] + a[j]) chia hết cho K ⟺ (a[i]%K + a[j]%K) % K = 0. Tức là số dư của a[i] và a[j] khi chia cho K phải "bù nhau": r + (K−r) = K, hoặc cả hai có số dư 0.

Bước 1: Tạo mảng cnt[0..K−1], trong đó cnt[r] = số phần tử có a[i] % K = r.
Bước 2: Đếm cặp:
  • Cặp (r=0, r=0): C(cnt[0], 2) = cnt[0]×(cnt[0]−1)/2
  • Cặp (r, K−r) với 1 ≤ r < K−r: cnt[r] × cnt[K−r]
  • Nếu K chẵn, cặp (K/2, K/2): C(cnt[K/2], 2)

Tổng: O(N + K) — rất nhanh.
Bài 2: Nhận dự án (XEPLICH)
📝 Code C++ — Tham lam (Sắp xếp + Set) O(N log N) 
// Bai 2: XEPLICH — Greedy (Sort by value + Set) O(N log N)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("XEPLICH.INP");
    ofstream out("XEPLICH.OUT");

    int n;
    in >> n;

    vector<pair<long long, int> > jobs(n);
    int maxD = 0;
    for (int i = 0; i < n; i++) {
        in >> jobs[i].second >> jobs[i].first; // deadline, value
        maxD = max(maxD, jobs[i].second);
    }

    sort(jobs.begin(), jobs.end(), greater<pair<long long, int> >());

    set<int> avail;
    for (int d = 1; d <= maxD; d++) avail.insert(d);

    long long total = 0;
    for (int i = 0; i < n; i++) {
        int d = jobs[i].second;
        long long v = jobs[i].first;
        set<int>::iterator it = avail.upper_bound(d);
        if (it != avail.begin()) {
            --it;
            total += v;
            avail.erase(it);
        }
    }

    out << total;

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 1000): Thử tất cả tập con, hoặc DP trạng thái: O(N² × max_d).
Subtask 2 (Full): Kỹ thuật Tham lam + Set:

Ý tưởng: Ưu tiên dự án giá trị cao nhất. Với mỗi dự án, xếp vào ngày trống muộn nhất trước hạn (để "dành" ngày sớm cho dự án khác).

Bước 1: Sắp xếp dự án theo giá trị giảm dần.
Bước 2: Dùng set chứa tất cả ngày trống {1, 2, …, max_d}. Với mỗi dự án (d, v):
  • Tìm ngày trống lớn nhất ≤ d bằng upper_bound(d) rồi lùi 1 bước.
  • Nếu tìm được → xếp, cộng v, xóa ngày đó khỏi set.
  • Nếu không → bỏ qua (không còn ngày trống nào phù hợp).

Tại sao đúng? Dự án giá trị cao được xét trước → luôn được ưu tiên. Chọn ngày muộn nhất → tối đa hóa số ngày sớm còn trống cho dự án có hạn sớm hơn.
Tổng: O(N log N) — sort O(N log N) + N lần set operation O(log N).
Bài 3: Xưởng sửa xe (THUEVIEC)
📝 Code C++ — Sắp xếp + Chặt nhị phân + QHĐ O(N log N) 
// Bai 3: THUEVIEC — Sort + Binary Search + DP O(N log N)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("THUEVIEC.INP");
    ofstream out("THUEVIEC.OUT");

    int n;
    in >> n;

    vector<int> s(n), e(n);
    vector<long long> p(n);
    for (int i = 0; i < n; i++) in >> s[i] >> e[i] >> p[i];

    // Sort by end time
    vector<int> idx(n);
    for (int i = 0; i < n; i++) idx[i] = i;
    sort(idx.begin(), idx.end(), [&](int a, int b) {
        return e[a] < e[b];
    });

    vector<int> ends(n);
    for (int i = 0; i < n; i++) ends[i] = e[idx[i]];

    vector<long long> dp(n);
    dp[0] = p[idx[0]];

    for (int i = 1; i < n; i++) {
        int si = s[idx[i]];
        long long pi = p[idx[i]];

        // Find latest j with ends[j] < si
        int lo = 0, hi = i - 1, j = -1;
        while (lo <= hi) {
            int mid = lo + (hi - lo) / 2;
            if (ends[mid] < si) {
                j = mid;
                lo = mid + 1;
            } else {
                hi = mid - 1;
            }
        }

        long long take = pi + (j >= 0 ? dp[j] : 0);
        dp[i] = max(dp[i - 1], take);
    }

    out << dp[n - 1];

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 1000): Sắp xếp theo thời gian kết thúc, QHĐ O(N²): dp[i] = max(dp[i−1], p[i] + max dp[j] với e[j] < s[i]).
Subtask 2 (Full): Bài này kết hợp 3 kỹ thuật:

1. Sắp xếp: Sắp xếp tất cả đơn hàng theo thời gian kết thúc e[i] tăng dần.

2. Quy hoạch động: Gọi dp[i] = lợi nhuận tối đa xét i đơn đầu tiên (sau khi sắp xếp).
  • Không nhận đơn i: dp[i] = dp[i−1]
  • Nhận đơn i: dp[i] = p[i] + dp[j], trong đó j là đơn hàng gần nhất kết thúc trước khi đơn i bắt đầu (e[j] < s[i]).
  • dp[i] = max(dp[i−1], p[i] + dp[j])

3. Chặt nhị phân: Tìm j nhanh bằng chặt nhị phân trên mảng ends[] đã sắp xếp — tìm chỉ số lớn nhất sao cho ends[j] < s[i]. Thay vì duyệt O(N), tìm trong O(log N).

Nếu không có chặt nhị phân: Vòng lặp tìm j là O(N) → tổng O(N²) — chỉ đủ subtask 1.
Với chặt nhị phân: Tìm j trong O(log N) → tổng O(N log N) — đủ full.
