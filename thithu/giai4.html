Bài 1: Xâu ngoặc hợp lệ (NGOAC)
📝 Code C++ — Stack O(N) 
// Bài 1: NGOAC — Stack O(N)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("NGOAC.INP");
    ofstream out("NGOAC.OUT");

    string s;
    in >> s;
    int n = s.size();

    stack<int> st;
    st.push(-1); // sentinel — đánh dấu biên trái
    int ans = 0;

    for (int i = 0; i < n; i++) {
        if (s[i] == '(') {
            st.push(i);
        } else {
            st.pop();
            if (st.empty()) {
                st.push(i); // ')' thừa → đặt biên mới
            } else {
                ans = max(ans, i - st.top());
            }
        }
    }

    out << ans;

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (|S| ≤ 1000): Duyệt mọi cặp (l, r), kiểm tra xâu S[l..r] có hợp lệ không (dùng biến đếm): nếu mở '(' cộng 1, đóng ')' trừ 1, không bao giờ âm, cuối = 0 → hợp lệ. Cập nhật max(r−l+1). Độ phức tạp O(N²).
Subtask 2 (Full): Kỹ thuật Stack.
Ý tưởng: Dùng stack lưu chỉ số (index) các ký tự chưa ghép được. Ban đầu đẩy −1 vào stack làm "biên trái" (sentinel).
— Gặp '(': đẩy chỉ số vào stack (chờ ghép).
— Gặp ')': pop stack. Nếu stack rỗng → ')' thừa, đẩy chỉ số hiện tại làm biên mới. Nếu stack không rỗng → đoạn hợp lệ từ (top+1) đến i, độ dài = i − st.top().
Mỗi ký tự vào/ra stack đúng 1 lần → O(N). Bộ nhớ O(N).
Bài 2: Xâu đối xứng (DOIXUNG)
📝 Code C++ — DP Palindrome O(N²) 
// Bài 2: DOIXUNG — DP Palindrome O(N²)
#include <bits/stdc++.h>
using namespace std;

int dp[5001][5001]; // dp[i][j] = LPS of s[i..j]

int main() {
    ifstream in("DOIXUNG.INP");
    ofstream out("DOIXUNG.OUT");

    string s;
    in >> s;
    int n = s.size();

    // Base: dp[i][i] = 1 (1 ký tự là palindrome)
    for (int i = 0; i < n; i++) dp[i][i] = 1;

    // Điền bảng theo độ dài tăng dần
    for (int len = 2; len <= n; len++) {
        for (int i = 0; i <= n - len; i++) {
            int j = i + len - 1;
            if (s[i] == s[j]) {
                dp[i][j] = dp[i + 1][j - 1] + 2;
            } else {
                dp[i][j] = max(dp[i + 1][j], dp[i][j - 1]);
            }
        }
    }

    // Đáp án = |S| − LPS(S)
    out << n - dp[0][n - 1];

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (|S| ≤ 20): Brute force — thử mọi cách chèn hoặc dùng DP O(N²) cũng chạy thoải mái.
Subtask 2 (Full): QHĐ trên đoạn (Interval DP) — Dãy con đối xứng dài nhất (LPS).
Nhận xét quan trọng: Số ký tự cần chèn = |S| − LPS(S), với LPS(S) = Longest Palindromic Subsequence. Vì dãy con đối xứng dài nhất chính là phần "khung" không cần thêm gì, chỉ cần chèn để bổ sung các ký tự còn lại.

Gọi dp[i][j] = độ dài LPS của xâu con S[i..j]:
— Nếu S[i] = S[j]: dp[i][j] = dp[i+1][j−1] + 2 (hai đầu ghép cặp).
— Nếu S[i] ≠ S[j]: dp[i][j] = max(dp[i+1][j], dp[i][j−1]) (bỏ đầu hoặc bỏ cuối).
Base: dp[i][i] = 1.
Đáp án: N − dp[0][N−1]. Độ phức tạp O(N²) thời gian và bộ nhớ.
Bài 3: Công việc (CONGVIEC)
📝 Code C++ — Sắp xếp + QHĐ O(N log N) 
// Bài 3: CONGVIEC — Sorting + DP (Weighted Job Scheduling)
#include <bits/stdc++.h>
using namespace std;

struct Job {
    int s, e;
    long long w;
};

bool cmp(const Job &a, const Job &b) {
    return a.e < b.e;
}

int main() {
    ifstream in("CONGVIEC.INP");
    ofstream out("CONGVIEC.OUT");

    int n;
    in >> n;

    vector<Job> job(n + 1);
    job[0].s = 0; job[0].e = 0; job[0].w = 0;
    for (int i = 1; i <= n; i++)
        in >> job[i].s >> job[i].e >> job[i].w;

    sort(job.begin() + 1, job.end(), cmp);

    // dp[i] = max profit considering jobs 1..i (sorted by end)
    vector<long long> dp(n + 1, 0);
    for (int i = 1; i <= n; i++) {
        // Binary search: latest j with job[j].e <= job[i].s
        int lo = 0, hi = i - 1, best = 0;
        while (lo <= hi) {
            int mid = (lo + hi) / 2;
            if (job[mid].e <= job[i].s) {
                best = mid;
                lo = mid + 1;
            } else {
                hi = mid - 1;
            }
        }
        dp[i] = max(dp[i - 1], job[i].w + dp[best]);
    }

    out << dp[n];

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 1000): Sắp xếp theo thời gian kết thúc. DP: dp[i] = max(dp[i−1], w[i] + dp[j]) với j là công việc gần nhất không chồng chéo. Tìm j bằng duyệt tuyến tính → O(N²).
Subtask 2 (Full): Bài này cần kết hợp 2 kỹ thuật:

1. Sắp xếp (Sorting): Sắp xếp N công việc theo thời gian kết thúc e[i] tăng dần. Nhờ đó, mảng e[] đã có thứ tự → dùng được tìm kiếm nhị phân.

2. Quy hoạch động (DP): Gọi dp[i] = lợi nhuận lớn nhất khi xét i công việc đầu (đã sắp xếp). Với mỗi công việc i, có 2 lựa chọn:
  • Không chọn i: dp[i] = dp[i−1]
  • Chọn i: dp[i] = w[i] + dp[j], trong đó j là công việc cuối cùng kết thúc trước khi i bắt đầu (e[j] ≤ s[i]). Tìm j bằng binary search trên mảng e[] đã sắp xếp → O(log N).

Tại sao Greedy không đúng? Bài Activity Selection (không trọng số) dùng greedy chọn việc kết thúc sớm nhất. Nhưng ở đây mỗi việc có lợi nhuận khác nhau — việc kết thúc sớm có thể lợi nhuận thấp hơn việc dài nhưng lời nhiều. Cần DP để cân nhắc tất cả phương án.
Tổng: O(N log N) — sắp xếp O(N log N) + DP N bước × binary search O(log N) = O(N log N).
