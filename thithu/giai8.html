Bài 1: Tăng điểm (TANGDIEM)
📝 Code C++ — Difference Array O(N + Q) 
// Bai 1: TANGDIEM — Difference Array O(N + Q)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("TANGDIEM.INP");
    ofstream out("TANGDIEM.OUT");

    int n, q;
    in >> n >> q;

    vector<long long> d(n + 2, 0);
    for (int i = 0; i < q; i++) {
        int l, r;
        long long v;
        in >> l >> r >> v;
        d[l] += v;
        d[r + 1] -= v;
    }

    long long mx = 0, cur = 0;
    for (int i = 1; i <= n; i++) {
        cur += d[i];
        mx = max(mx, cur);
    }

    out << mx;

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N, Q ≤ 1000): Với mỗi đợt, duyệt từ l đến r cộng v: O(N×Q) ≈ 10⁶.
Subtask 2 (Full): Kỹ thuật Difference Array (Mảng hiệu):

Ý tưởng: Thay vì cộng v vào từng phần tử trong [l, r] (tốn O(r−l+1) mỗi lần), chỉ cần đánh dấu 2 điểm:
  • d[l] += v (bắt đầu cộng)
  • d[r+1] −= v (kết thúc cộng)

Sau tất cả Q đợt: Duyệt prefix sum trên d[] để khôi phục mảng gốc. Giá trị tại vị trí i = tổng d[1] + d[2] + … + d[i].

Tại sao đúng? Khi cộng v vào d[l] và trừ v khỏi d[r+1], prefix sum tại mọi vị trí từ l đến r sẽ tăng thêm v, còn từ r+1 trở đi bị triệt tiêu.
Tổng: O(N + Q) — mỗi đợt O(1), khôi phục O(N).
Bài 2: Xếp hàng (GIAITHUA)
📝 Code C++ — Xử lí số lớn (Big Number × int) O(N × D) 
// Bai 2: GIAITHUA — Big Number Multiply O(N * D)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("GIAITHUA.INP");
    ofstream out("GIAITHUA.OUT");

    int n;
    in >> n;

    vector<int> result;
    result.push_back(1);

    for (int i = 2; i <= n; i++) {
        long long carry = 0;
        for (int j = 0; j < (int)result.size(); j++) {
            long long prod = (long long)result[j] * i + carry;
            result[j] = prod % 10;
            carry = prod / 10;
        }
        while (carry > 0) {
            result.push_back(carry % 10);
            carry /= 10;
        }
    }

    for (int i = (int)result.size() - 1; i >= 0; i--)
        out << result[i];

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 20): Kết quả nằm trong long long (20! ≈ 2.4×10¹⁸). Tính trực tiếp bằng vòng lặp nhân dần.
Subtask 2 (Full — N ≤ 1000): 1000! có ~2568 chữ số → cần xử lí số lớn:

Biểu diễn: Dùng vector<int> lưu từng chữ số (hàng đơn vị ở vị trí 0, hàng cao nhất ở cuối).

Phép nhân số lớn × số nhỏ:
  1. Duyệt từ chữ số thấp → cao (vị trí 0 → cuối)
  2. Tại mỗi vị trí j: prod = result[j] × i + carry
  3. Chữ số mới = prod % 10, carry = prod / 10
  4. Sau khi hết digits, nếu carry > 0 thì thêm digits mới

Ý tưởng chính: Bắt đầu result = 1, lần lượt nhân với 2, 3, …, N. Đây chính là phép nhân dây chuyền giống cách nhân trên giấy.
Chỉ cần 1 phép toán: nhân số lớn × số nhỏ. Không cần chia, không cần cộng hai số lớn.
Tổng: O(N × D) trong đó D là số chữ số kết quả (~2568). Tức ~2.5×10⁶ phép tính — rất nhanh.
Bài 3: Hệ thống ống dẫn (ONGDOI)
📝 Code C++ — Luồng cực đại Dinic O(V² × E) 
// Bai 3: ONGDOI — Max Flow (Dinic) O(V^2 * E)
#include <bits/stdc++.h>
using namespace std;

const long long INF = 1e18;
const int MAXV = 510;

struct Edge {
    int to, rev;
    long long cap;
};

vector<Edge> graph[MAXV];
int level[MAXV], iter_[MAXV];

void add_edge(int from, int to, long long cap) {
    graph[from].push_back({to, (int)graph[to].size(), cap});
    graph[to].push_back({from, (int)graph[from].size() - 1, 0});
}

bool bfs(int s, int t) {
    memset(level, -1, sizeof(level));
    queue<int> q;
    level[s] = 0;
    q.push(s);
    while (!q.empty()) {
        int v = q.front(); q.pop();
        for (int i = 0; i < (int)graph[v].size(); i++) {
            Edge& e = graph[v][i];
            if (e.cap > 0 && level[e.to] < 0) {
                level[e.to] = level[v] + 1;
                q.push(e.to);
            }
        }
    }
    return level[t] >= 0;
}

long long dfs(int v, int t, long long f) {
    if (v == t) return f;
    for (int& i = iter_[v]; i < (int)graph[v].size(); i++) {
        Edge& e = graph[v][i];
        if (e.cap > 0 && level[v] < level[e.to]) {
            long long d = dfs(e.to, t, min(f, e.cap));
            if (d > 0) {
                e.cap -= d;
                graph[e.to][e.rev].cap += d;
                return d;
            }
        }
    }
    return 0;
}

long long max_flow(int s, int t) {
    long long flow = 0;
    while (bfs(s, t)) {
        memset(iter_, 0, sizeof(iter_));
        long long d;
        while ((d = dfs(s, t, INF)) > 0)
            flow += d;
    }
    return flow;
}

int main() {
    ifstream in("ONGDOI.INP");
    ofstream out("ONGDOI.OUT");

    int n, m;
    in >> n >> m;

    for (int i = 0; i < m; i++) {
        int u, v;
        long long c;
        in >> u >> v >> c;
        add_edge(u, v, c);
    }

    out << max_flow(1, n);

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 10): DFS tìm tất cả đường đi từ 1→N, mỗi lần tìm được thì đẩy luồng dọc đường đi (Ford-Fulkerson đơn giản).
Subtask 2 (Full): Đây là bài toán Luồng cực đại (Max Flow) kinh điển. Dùng thuật toán Dinic:

Ý tưởng Dinic:
  1. BFS phân tầng: Từ nguồn (trạm 1), BFS gán level cho mỗi đỉnh. Chỉ đi các cạnh còn dung lượng dư.
  2. DFS tìm đường tăng luồng: Chỉ đi theo chiều level tăng (tránh vòng lặp), đẩy luồng tối đa dọc đường. Dùng kỹ thuật iter_ để tránh duyệt lại cạnh đã cạn.
  3. Lặp: Quay lại bước 1 cho đến khi BFS không tìm được đường nào từ nguồn đến đích.

Đồ thị dư (Residual graph): Mỗi cạnh (u, v, cap) có cạnh ngược (v, u, 0). Khi đẩy d đơn vị luồng qua cạnh gốc: cap gốc giảm d, cap ngược tăng d. Cạnh ngược cho phép "hủy" luồng đã đẩy sai.

Tại sao đúng? Theo định lý Max-Flow = Min-Cut, luồng cực đại = tổng dung lượng lát cắt nhỏ nhất. Khi BFS không tìm được đường → ta đã đạt min-cut → đã đạt max flow.
Tổng: O(V² × E). Với N ≤ 500, M ≤ 5000: chạy thoải mái trong 1 giây.
