Bài 1: Phân công nhóm (CHIAVIEC)
📝 Code C++ — Binary Search on Answer O(N log S) 
// Bai 1: CHIAVIEC — Binary Search on Answer O(N log S)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("CHIAVIEC.INP");
    ofstream out("CHIAVIEC.OUT");

    int n, k;
    in >> n >> k;
    vector<long long> a(n);
    long long lo = 0, hi = 0;
    for (int i = 0; i < n; i++) {
        in >> a[i];
        lo = max(lo, a[i]);
        hi += a[i];
    }

    while (lo < hi) {
        long long mid = (lo + hi) / 2;
        int cnt = 1;
        long long sum = 0;
        for (int i = 0; i < n; i++) {
            if (sum + a[i] > mid) {
                cnt++;
                sum = a[i];
            } else {
                sum += a[i];
            }
        }
        if (cnt <= k) hi = mid;
        else lo = mid + 1;
    }

    out << lo;

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 1000): Quy hoạch động O(N²K): dp[i][j] = tổng max nhỏ nhất khi chia i phần tử đầu thành j nhóm. Chậm nhưng đúng.
Subtask 2 (Full): Chặt nhị phân đáp án (Binary Search on Answer):

Nhận xét quan trọng: Đáp án T (tổng lớn nhất nhỏ nhất) thoả mãn: nếu chia được với giới hạn T thì cũng chia được với T+1, T+2,… → tính chất đơn điệu → dùng chặt nhị phân!

Khoảng chặt nhị phân:
  • lo = max(ai) — mỗi nhóm phải chứa ít nhất phần tử lớn nhất
  • hi = sum(ai) — trường hợp K = 1, chỉ có 1 nhóm

Hàm kiểm tra check(mid): Với giới hạn mid, duyệt tham lam từ trái sang phải: cộng dồn a[i] vào nhóm hiện tại, nếu tổng vượt mid thì mở nhóm mới. Đếm số nhóm cần dùng. Nếu ≤ K → mid khả thi.

Tại sao tham lam đúng? Nếu giới hạn mid khả thi, việc "nhồi tối đa" vào nhóm hiện tại là tốt nhất vì nhường dư lượng cho các nhóm sau.

Độ phức tạp: O(N × log(S)) với S = sum(ai) ≤ 1014 → log(S) ≈ 47. Tổng ≈ 5 × 10⁶ — rất nhanh.
Bài 2: Xâu đối xứng (XAUDX)
📝 Code C++ — LCS(S, rev(S)) + Rolling Array O(N²) 
// Bai 2: XAUDX — Min insertions = |S| - LCS(S, rev(S)), O(N^2) time O(N) space
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("XAUDX.INP");
    ofstream out("XAUDX.OUT");

    string s;
    in >> s;
    int n = s.size();
    string t = s;
    reverse(t.begin(), t.end());

    vector<int> prev(n + 1, 0), curr(n + 1, 0);
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= n; j++) {
            if (s[i - 1] == t[j - 1]) curr[j] = prev[j - 1] + 1;
            else curr[j] = max(prev[j], curr[j - 1]);
        }
        swap(prev, curr);
        fill(curr.begin(), curr.end(), 0);
    }

    out << n - prev[n];

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (|S| ≤ 1000): DP bảng 2 chiều dp[N+1][N+1] chiếm ~4MB, chạy nhanh.
Subtask 2 (Full): Cùng thuật toán nhưng tối ưu bộ nhớ bằng rolling array.

Mấu chốt: Số kí tự cần chèn ít nhất = |S| − LPS(S), với LPS(S) là dãy con đối xứng dài nhất (Longest Palindromic Subsequence).

Tại sao? Dãy con đối xứng dài nhất đã "đối xứng sẵn" — phần còn lại (|S| − LPS) cần được "nhân đôi" bằng cách chèn kí tự đối xứng tương ứng.

Quy về LCS: LPS(S) = LCS(S, reverse(S)).
Chứng minh: Dãy con đối xứng của S cũng là dãy con chung của S và reverse(S), và ngược lại. Vậy LPS(S) = LCS(S, T) với T = reverse(S).

Công thức LCS:
  • dp[i][j] = LCS(S[1..i], T[1..j])
  • Nếu S[i] == T[j]: dp[i][j] = dp[i−1][j−1] + 1
  • Ngược lại: dp[i][j] = max(dp[i−1][j], dp[i][j−1])

Đáp án: N − dp[N][N].

Rolling array: dp[i][j] chỉ phụ thuộc hàng i−1 → giữ 2 vector prev, curr. Bộ nhớ O(N) thay vì O(N²).
Tổng: O(N²) ≈ 2.5 × 10⁷ cho N = 5000 — chạy tốt trong 1 giây.
Bài 3: Hệ thống cấp nước (CAPNUOC)
📝 Code C++ — Pre-merge Sources + Kruskal MST O(M log M) 
// Bai 3: CAPNUOC — Pre-merge sources + Kruskal MST O(M log M)
#include <bits/stdc++.h>
using namespace std;

struct Edge { int u, v; long long w; };

int par[100005], rnk[100005];

int find(int x) { return par[x] == x ? x : par[x] = find(par[x]); }

bool unite(int a, int b) {
    a = find(a); b = find(b);
    if (a == b) return false;
    if (rnk[a] < rnk[b]) swap(a, b);
    par[b] = a;
    if (rnk[a] == rnk[b]) rnk[a]++;
    return true;
}

int main() {
    ifstream in("CAPNUOC.INP");
    ofstream out("CAPNUOC.OUT");

    int n, m, k;
    in >> n >> m >> k;

    for (int i = 1; i <= n; i++) { par[i] = i; rnk[i] = 0; }

    vector<int> src(k);
    for (int i = 0; i < k; i++) in >> src[i];

    // Pre-merge all source cities
    for (int i = 1; i < k; i++) unite(src[0], src[i]);

    vector<Edge> edges(m);
    for (int i = 0; i < m; i++) in >> edges[i].u >> edges[i].v >> edges[i].w;

    sort(edges.begin(), edges.end(),
        [](const Edge& a, const Edge& b) { return a.w < b.w; });

    long long totalCost = 0;
    for (int i = 0; i < m; i++) {
        if (unite(edges[i].u, edges[i].v)) {
            totalCost += edges[i].w;
        }
    }

    // Check if all connected
    int root = find(1);
    bool connected = true;
    for (int i = 2; i <= n; i++) {
        if (find(i) != root) { connected = false; break; }
    }

    if (connected) out << totalCost;
    else out << -1;

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N, M ≤ 1000): Prim hoặc Kruskal cơ bản — O(M log M) hoặc O(N²).
Subtask 2 (N, M ≤ 10000): Kruskal + DSU, vẫn nhanh.
Subtask 3 (Full): Pre-merge + Kruskal:

Biến đổi so với MST truyền thống:
Bài MST thông thường nối tất cả N đỉnh thành 1 thành phần liên thông. Ở đây, các nguồn nước đã "tương đương" nhau — chỉ cần mỗi thành phố nối được đến ít nhất một nguồn.

Ý tưởng chính: Gộp trước (pre-merge) tất cả K nguồn nước vào cùng một thành phần liên thông trong DSU (chi phí 0). Sau đó chạy Kruskal như bình thường — các cạnh nối giữa hai nguồn sẽ tự động bị bỏ qua (vì đã cùng tập hợp).

Tại sao đúng? Việc gộp K nguồn nước tương đương với thêm một "siêu đỉnh ảo" nối đến tất cả K nguồn bằng cạnh chi phí 0, rồi tìm MST trên đồ thị mở rộng. Kết quả chính là MST bỏ các cạnh chi phí 0 → đúng tổng chi phí thực tế.

Thuật toán:
  1. Khởi tạo DSU cho N đỉnh
  2. Union tất cả K nguồn nước (chi phí 0)
  3. Sort M cạnh theo chi phí tăng dần
  4. Kruskal: duyệt cạnh, nếu nối 2 thành phần khác nhau → thêm vào MST
  5. Kiểm tra liên thông: nếu tất cả cùng find → in tổng, ngược lại → -1

Độ phức tạp: O(M log M) cho sort + O(M × α(N)) cho DSU ≈ O(M log M). Với M = 2 × 10⁵ → rất nhanh.
