Bài 1: Biểu đồ cột (HINCHU)
📝 Code C++ — Monotone Stack O(N) 
// Bai 1: HINCHU — Largest Rectangle in Histogram, Mono Stack O(N)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("HINCHU.INP");
    ofstream out("HINCHU.OUT");

    int n;
    in >> n;
    vector<int> h(n);
    for (int i = 0; i < n; i++) in >> h[i];

    stack<int> st;
    long long ans = 0;

    for (int i = 0; i <= n; i++) {
        int cur = (i == n) ? 0 : h[i];
        while (!st.empty() && h[st.top()] > cur) {
            long long height = h[st.top()]; st.pop();
            long long width = st.empty() ? i : i - st.top() - 1;
            ans = max(ans, height * width);
        }
        st.push(i);
    }

    out << ans;

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 5000): Duyệt mọi cặp (l, r): tìm min(h[l..r]) bằng cách cập nhật khi mở rộng r, tính diện tích = min × (r−l+1). Tổng O(N²) ≈ 2.5 × 10⁷.
Subtask 2 (Full): Monotone Stack (Ngăn xếp đơn điệu):

Ý tưởng: Với mỗi cột i, tìm "phạm vi mở rộng" — đoạn liên tiếp mà h[i] là giá trị nhỏ nhất. Diện tích = h[i] × chiều rộng đoạn đó.

Thuật toán: Duy trì stack chứa chỉ số với chiều cao tăng dần từ đáy lên đỉnh:
  1. Duyệt i = 0 đến N (thêm "cột ảo" h = 0 ở cuối để xả stack)
  2. Nếu h[i] < h[đỉnh stack]: lấy đỉnh ra, tính hình chữ nhật:
    • height = chiều cao cột vừa lấy ra
    • width = khoảng cách từ phần tử tiếp theo trong stack đến i
    • Cập nhật ans = max(ans, height × width)
  3. Đẩy i vào stack

Tại sao đúng? Khi cột j bị lấy ra tại vị trí i, nó biết: bên phải bị chặn bởi cột i (thấp hơn), bên trái bị chặn bởi cột ở đỉnh stack mới (thấp hơn hoặc bằng). Vậy đoạn [left+1, i−1] chính là phạm vi mà h[j] có thể mở rộng.
Mỗi phần tử được push/pop đúng 1 lần → tổng O(N).
Bài 2: Biến đổi xâu (DOIXAU)
📝 Code C++ — Edit Distance DP O(N×M) 
// Bai 2: DOIXAU — Edit Distance DP O(N*M), space O(M)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("DOIXAU.INP");
    ofstream out("DOIXAU.OUT");

    string a, b;
    in >> a >> b;

    int n = a.size(), m = b.size();
    vector<int> prev(m + 1), curr(m + 1);

    for (int j = 0; j <= m; j++) prev[j] = j;

    for (int i = 1; i <= n; i++) {
        curr[0] = i;
        for (int j = 1; j <= m; j++) {
            if (a[i - 1] == b[j - 1]) curr[j] = prev[j - 1];
            else curr[j] = 1 + min({prev[j - 1], prev[j], curr[j - 1]});
        }
        swap(prev, curr);
    }

    out << prev[m];

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (|A|, |B| ≤ 1000): DP 2D bảng dp[N+1][M+1] — chiếm ~4MB, chạy nhanh.
Subtask 2 (Full): Cùng thuật toán nhưng tối ưu bộ nhớ bằng rolling array (chỉ giữ 2 hàng prev, curr).

Công thức QHĐ (Edit Distance):
Gọi dp[i][j] = số phép biến đổi ít nhất để biến A[1..i] thành B[1..j].

Cơ sở:
  • dp[0][j] = j (chèn j kí tự)
  • dp[i][0] = i (xóa i kí tự)

Chuyển trạng thái: Với mỗi (i, j):
  • Nếu A[i] == B[j]: dp[i][j] = dp[i−1][j−1] (không cần thao tác)
  • Nếu A[i] ≠ B[j]: dp[i][j] = 1 + min(
    dp[i−1][j−1], ← thay A[i] → B[j]
    dp[i−1][j], ← xóa A[i]
    dp[i][j−1] ← chèn B[j]
  )

Tối ưu bộ nhớ: dp[i][j] chỉ phụ thuộc hàng i và i−1 → chỉ cần 2 mảng 1 chiều. Bộ nhớ giảm từ O(N×M) xuống O(M).
Tổng: O(N×M) thời gian ≈ 2.5 × 10⁷ cho N, M = 5000 — chạy tốt trong 1 giây.
Bài 3: Mạng liên lạc mật (MANG)
📝 Code C++ — Offline Sort + DSU O((M+Q) log(M+Q)) 
// Bai 3: MANG — Offline Sort + DSU O((M+Q) log(M+Q))
#include <bits/stdc++.h>
using namespace std;

struct Edge { int u, v, w; };
struct Query { int u, v, t, idx; };

int par[100005], rnk[100005];

int find(int x) { return par[x] == x ? x : par[x] = find(par[x]); }

void unite(int a, int b) {
    a = find(a); b = find(b);
    if (a == b) return;
    if (rnk[a] < rnk[b]) swap(a, b);
    par[b] = a;
    if (rnk[a] == rnk[b]) rnk[a]++;
}

int main() {
    ifstream in("MANG.INP");
    ofstream out("MANG.OUT");

    int n, m, q;
    in >> n >> m >> q;

    vector<Edge> edges(m);
    for (int i = 0; i < m; i++) in >> edges[i].u >> edges[i].v >> edges[i].w;

    vector<Query> queries(q);
    for (int i = 0; i < q; i++) {
        in >> queries[i].u >> queries[i].v >> queries[i].t;
        queries[i].idx = i;
    }

    // Sort edges descending by weight
    sort(edges.begin(), edges.end(),
        [](const Edge& a, const Edge& b) { return a.w > b.w; });
    // Sort queries descending by threshold
    sort(queries.begin(), queries.end(),
        [](const Query& a, const Query& b) { return a.t > b.t; });

    for (int i = 1; i <= n; i++) { par[i] = i; rnk[i] = 0; }

    vector<string> ans(q);
    int ei = 0;
    for (int i = 0; i < q; i++) {
        while (ei < m && edges[ei].w >= queries[i].t) {
            unite(edges[ei].u, edges[ei].v);
            ei++;
        }
        ans[queries[i].idx] =
            (find(queries[i].u) == find(queries[i].v)) ? "YES" : "NO";
    }

    for (int i = 0; i < q; i++) out << ans[i] << "\n";

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N, M, Q ≤ 1000): Với mỗi truy vấn (u, v, T): BFS/DFS chỉ đi qua các cạnh có w ≥ T, kiểm tra u và v liên thông. Tổng O(Q × (N+M)) ≈ 3 × 10⁶.
Subtask 2 (N, M, Q ≤ 5000): Cùng BFS/DFS nhưng chậm hơn. Tổng O(Q × (N+M)) ≈ 5 × 10⁷ — vẫn chấp nhận được.
Subtask 3 (Full): Offline Sort + Disjoint Set Union (DSU):

Biến đổi so với bài duyệt đồ thị truyền thống:
Bài BFS/DFS thông thường kiểm tra liên thông trên đồ thị cố định. Ở đây, mỗi truy vấn có ngưỡng T khác nhau, nên đồ thị "thay đổi" theo truy vấn — chỉ giữ cạnh w ≥ T. BFS/DFS mỗi truy vấn → quá chậm O(Q×M).

Ý tưởng Offline: Nếu sắp xếp truy vấn theo T giảm dần, thì khi T giảm, chỉ có thêm cạnh (không mất cạnh). Xử lý giống "thêm cạnh dần vào đồ thị" — DSU hoàn hảo!

Thuật toán:
  1. Sort cạnh theo w giảm dần
  2. Sort truy vấn theo T giảm dần (lưu index gốc)
  3. Xử lý truy vấn lần lượt: trước mỗi truy vấn T, thêm tất cả cạnh có w ≥ T vào DSU (con trỏ ei chỉ tiến)
  4. Kiểm tra find(u) == find(v) → YES/NO
  5. Ghi đáp án vào vị trí gốc, in theo thứ tự ban đầu

Tại sao nhanh? Mỗi cạnh được thêm đúng 1 lần. Sort: O((M+Q) log(M+Q)). DSU: O((M+Q) × α(N)) ≈ O(M+Q). Tổng: O((M+Q) log(M+Q)).
