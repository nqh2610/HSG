Bài 1: Cặp số (CAPSO)
📝 Code C++ — Sort + Two Pointers O(N log N) 
// Bài 1: CAPSO — Sort + Two Pointers O(N log N)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("CAPSO.INP");
    ofstream out("CAPSO.OUT");

    int n;
    long long s;
    in >> n >> s;

    vector<long long> a(n);
    for (int i = 0; i < n; i++) in >> a[i];

    sort(a.begin(), a.end());

    long long cnt = 0;
    int l = 0, r = n - 1;
    while (l < r) {
        if (a[l] + a[r] <= s) {
            cnt += r - l;
            l++;
        } else {
            r--;
        }
    }

    out << cnt;

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 1000): Duyệt tất cả cặp (i, j) rồi đếm: O(N²) ≈ 10⁶.
Subtask 2 (Full): Kỹ thuật Sort + Two Pointers:

Bước 1: Sắp xếp mảng tăng dần O(N log N).
Bước 2: Hai con trỏ l = 0, r = N−1. Nếu a[l] + a[r] ≤ S thì tất cả (l, l+1), (l, l+2), …, (l, r) đều hợp lệ → cộng (r − l) cặp, rồi tăng l. Ngược lại giảm r.

Tại sao đúng? Mảng đã sắp xếp → a[l] nhỏ nhất bên trái, a[r] lớn nhất bên phải. Nếu a[l]+a[r] ≤ S thì a[l] ghép với bất kỳ ai từ l+1 đến r đều ≤ S. Ngược lại, a[r] quá lớn cho a[l] nên cũng quá lớn cho mọi phần tử → loại a[r].
Tổng: O(N log N) — sort O(N log N) + two pointers O(N).
Bài 2: Hành lý thám hiểm (BALO)
📝 Code C++ — QHĐ Ba lô 2 ràng buộc O(N×W×V) 
// Bài 2: BALO — DP Knapsack 2D (weight + volume) O(N*W*V)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("BALO.INP");
    ofstream out("BALO.OUT");

    int n, W, V;
    in >> n >> W >> V;

    vector<int> w(n), s(n), p(n);
    for (int i = 0; i < n; i++) in >> w[i] >> s[i] >> p[i];

    // dp[j][k] = max value with weight <= j, volume <= k
    vector<vector<long long> > dp(W + 1, vector<long long>(V + 1, 0));
    for (int i = 0; i < n; i++) {
        for (int j = W; j >= w[i]; j--) {
            for (int k = V; k >= s[i]; k--) {
                dp[j][k] = max(dp[j][k], dp[j - w[i]][k - s[i]] + (long long)p[i]);
            }
        }
    }

    out << dp[W][V];

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 20): Duyệt tất cả 2N tập con (≈ 10⁶), kiểm tra cân nặng ≤ W và thể tích ≤ V, lấy max giá trị.
Subtask 2 (Full): Đây là biến thể Ba lô 0/1 với 2 ràng buộc:

So với Ba lô thường: Ba lô chuẩn có 1 ràng buộc → dp[j] 1 chiều. Bài này có 2 ràng buộc (cân nặng + thể tích) → dp[j][k] 2 chiều.

Công thức: dp[j][k] = max value khi tổng cân nặng ≤ j, tổng thể tích ≤ k.
Với mỗi vật i, duyệt j từ W→w[i], k từ V→s[i] (ngược để đảm bảo 0/1):
  dp[j][k] = max(dp[j][k], dp[j−w[i]][k−s[i]] + p[i])

Bộ nhớ: dp[501][501] ≈ 2 MB — vừa đủ.
Tổng: O(N × W × V) = 200 × 500 × 500 = 5×10⁷ — chạy vừa trong 1 giây.
Bài 3: Vận chuyển hàng (CAUNOI)
📝 Code C++ — Chặt nhị phân đáp án + BFS O(M log C) 
// Bài 3: CAUNOI — Binary Search Answer + BFS
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("CAUNOI.INP");
    ofstream out("CAUNOI.OUT");

    int n, m;
    in >> n >> m;

    vector<vector<pair<int,int> > > adj(n + 1);
    for (int i = 0; i < m; i++) {
        int u, v, c;
        in >> u >> v >> c;
        adj[u].push_back(make_pair(v, c));
        adj[v].push_back(make_pair(u, c));
    }

    // Binary search on answer
    int lo = 1, hi = 1000000000, ans = 0;
    while (lo <= hi) {
        int mid = lo + (hi - lo) / 2;

        // BFS: only use edges with capacity >= mid
        vector<bool> visited(n + 1, false);
        queue<int> q;
        q.push(1);
        visited[1] = true;
        while (!q.empty()) {
            int u = q.front(); q.pop();
            for (int j = 0; j < (int)adj[u].size(); j++) {
                int v = adj[u][j].first;
                int c = adj[u][j].second;
                if (!visited[v] && c >= mid) {
                    visited[v] = true;
                    q.push(v);
                }
            }
        }

        if (visited[n]) {
            ans = mid;
            lo = mid + 1;
        } else {
            hi = mid - 1;
        }
    }

    out << ans;

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 1000): Thử-sai: với mỗi tải trọng T, BFS/DFS chỉ đi cạnh ≥ T, kiểm tra đến được kho N không. Duyệt T từ max(c) xuống 1 → O(Cmax × (N+M)).
Subtask 2 (Full): Bài này cần kết hợp 2 kỹ thuật:

1. Chặt nhị phân đáp án (Binary Search the Answer): Đáp án T nằm trong [1, 10⁹]. Nhận xét: nếu xe tải T tấn đi được thì xe T−1 tấn cũng đi được (tính đơn điệu). Do đó chặt nhị phân tìm T lớn nhất mà vẫn đi được.

2. BFS kiểm tra: Với mỗi giá trị mid, chỉ dùng các cạnh có tải trọng ≥ mid. BFS từ kho 1 → kiểm tra đến được kho N không. Nếu được → tăng lo, không → giảm hi.

Tại sao không dùng Dijkstra? Dijkstra tối ưu tổng chi phí. Ở đây ta cần tối ưu min trên đường đi (bottleneck path) — chặt nhị phân + BFS đơn giản và hiệu quả hơn.
Tổng: O((N+M) × log C) ≈ (3×10⁵) × 30 ≈ 10⁷ — nhanh.
