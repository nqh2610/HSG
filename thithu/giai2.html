LỜI GIẢI — ĐỀ THI THỬ HSG TIN HỌC — ĐỀ 2

Bài 1: Trung bình (TRUNGBINH)
📝 Code C++ — Prefix sum + Map O(N log N) 
// Bài 1: TRUNGBINH — Prefix sum + Map
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("TRUNGBINH.INP");
    ofstream out("TRUNGBINH.OUT");

    int n;
    long long K;
    in >> n >> K;

    long long prefix = 0;
    map<long long, long long> cnt;
    cnt[0] = 1;
    long long ans = 0;

    for (int i = 0; i < n; i++) {
        long long a;
        in >> a;
        prefix += a - K;
        ans += cnt[prefix];
        cnt[prefix]++;
    }

    out << ans;

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 1000): Duyệt mọi cặp (l, r), tính tổng đoạn, kiểm tra tổng / (r−l+1) = K. Tương đương kiểm tra tổng = K × (r−l+1). Độ phức tạp O(N²).
Subtask 2 (Full): Ý tưởng cốt lõi: Trung bình đoạn [l, r] = K tương đương tổng (A[i] − K) từ l đến r bằng 0.
Đặt b[i] = A[i] − K. Cần đếm số đoạn [l, r] có tổng b[l..r] = 0.
Dùng prefix sum: P[0] = 0, P[i] = b[1] + ... + b[i]. Tổng b[l..r] = P[r] − P[l−1] = 0 ↔ P[r] = P[l−1].
Vậy đếm số cặp (i, j) mà i < j và P[i] = P[j]. Dùng map đếm số lần xuất hiện mỗi giá trị prefix sum, chỉ cần 1 vòng lặp.
Tổng: O(N log N) với map, hoặc O(N) với unordered_map.
Bài 2: Đếm xâu con (DEMXAU)
📝 Code C++ — QHĐ hai chiều O(|S| × |T|) 
// Bài 2: DEMXAU — Đếm xâu con (Distinct Subsequences)
#include <bits/stdc++.h>
using namespace std;

const int MOD = 1e9 + 7;
int dp[5001][5001];

int main() {
    ifstream in("DEMXAU.INP");
    ofstream out("DEMXAU.OUT");

    string S, T;
    in >> S >> T;
    int n = S.size(), m = T.size();

    for (int i = 0; i <= n; i++) dp[i][0] = 1;

    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            dp[i][j] = dp[i - 1][j];
            if (S[i - 1] == T[j - 1])
                dp[i][j] = (dp[i][j] + dp[i - 1][j - 1]) % MOD;
        }
    }

    out << dp[n][m];

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (|S|, |T| ≤ 20): Sinh tất cả 2|S| xâu con của S (≤ 2²⁰ ≈ 10⁶), đếm bao nhiêu xâu bằng T.
Subtask 2 (Full): Bài toán Distinct Subsequences. Gọi dp[i][j] = số cách chọn xâu con từ S[1..i] để tạo thành T[1..j].
Cơ sở: dp[i][0] = 1 (xâu T rỗng — luôn có đúng 1 cách: không chọn ký tự nào).
Chuyển trạng thái:
— Luôn có dp[i][j] = dp[i−1][j] (bỏ qua S[i], không dùng nó).
— Nếu S[i] = T[j]: dp[i][j] += dp[i−1][j−1] (dùng S[i] ghép với T[j]).
So sánh với LCS: Cùng bảng DP 2 chiều trên 2 xâu, nhưng thay vì lấy max thì ta cộng dồn số cách. Đây là biến thể tự nhiên của LCS — chuyển từ "tối ưu" sang "đếm".
Đáp án: dp[|S|][|T|] mod (10⁹ + 7). Tổng: O(|S| × |T|). Bộ nhớ: mảng dp[5001][5001] ≈ 100 MB.
Bài 3: Mê cung (MECUNG)
📝 Code C++ — BFS mở rộng trạng thái O(N × M × K) 
// Bài 3: MECUNG — BFS mở rộng trạng thái
#include <bits/stdc++.h>
using namespace std;

int m, n, K;
char a[1001][1001];
int dist[1001][1001][6];
int dx[] = {-1, 1, 0, 0};
int dy[] = {0, 0, -1, 1};

struct State {
    int x, y, k;
};

void BFS_Luoi() {
    memset(dist, -1, sizeof(dist));
    dist[0][0][0] = 0;
    queue<State> q;
    q.push({0, 0, 0});

    while (!q.empty()) {
        int x = q.front().x, y = q.front().y, k = q.front().k;
        q.pop();
        for (int d = 0; d < 4; d++) {
            int nx = x + dx[d], ny = y + dy[d];
            if (nx < 0 || nx >= m || ny < 0 || ny >= n) continue;

            if (a[nx][ny] != '#') {
                if (dist[nx][ny][k] == -1) {
                    dist[nx][ny][k] = dist[x][y][k] + 1;
                    q.push({nx, ny, k});
                }
            } else if (k + 1 <= K && dist[nx][ny][k + 1] == -1) {
                dist[nx][ny][k + 1] = dist[x][y][k] + 1;
                q.push({nx, ny, k + 1});
            }
        }
    }
}

int main() {
    ifstream in("MECUNG.INP");
    ofstream out("MECUNG.OUT");
    in >> m >> n >> K;
    for (int i = 0; i < m; i++) {
        string row;
        in >> row;
        for (int j = 0; j < n; j++)
            a[i][j] = row[j];
    }

    BFS_Luoi();

    int ans = -1;
    for (int k = 0; k <= K; k++) {
        if (dist[m-1][n-1][k] != -1) {
            if (ans == -1 || dist[m-1][n-1][k] < ans)
                ans = dist[m-1][n-1][k];
        }
    }

    out << ans;

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (K = 0): BFS thông thường trên lưới 2D. Trạng thái chỉ cần (hàng, cột). Tìm đường ngắn nhất từ (1,1) đến (N,M), không được đi vào ô tường. Độ phức tạp O(N × M).
Subtask 2 (Full): BFS mở rộng trạng thái — mỗi trạng thái là bộ ba (hàng, cột, số_chìa_đã_dùng).
Khi đi vào ô trống '.': chuyển sang (nr, nc, k) — không tốn chìa khóa.
Khi đi vào ô tường '#': chuyển sang (nr, nc, k+1) — tốn 1 chìa khóa (chỉ khi k+1 ≤ K).
BFS đảm bảo tìm đường ngắn nhất vì mỗi bước tốn đúng 1 đơn vị.
Đáp án: min(dist[N][M][k]) với k = 0, 1, …, K.
Không gian trạng thái: N × M × (K+1) ≤ 1000 × 1000 × 6 = 6 × 10⁶. Bộ nhớ dist ≈ 24 MB.
Tổng: O(N × M × K).
