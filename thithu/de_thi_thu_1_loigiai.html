LỜI GIẢI — ĐỀ THI THỬ HSG TIN HỌC — ĐỀ 1

Bài 1: Đoạn êm (DOANEM)
📝 Code C++ — O(N) Two Pointers + Deque 
// Bài 1: DOANEM — Two Pointers + 2 Deque đơn điệu
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("DOANEM.INP");
    ofstream out("DOANEM.OUT");

    int n;
    long long K;
    in >> n >> K;
    vector<long long> a(n);
    for(int i = 0; i < n; i++) in >> a[i];

    // dqMax: deque giữ chỉ số → a[] giảm dần (front = max)
    // dqMin: deque giữ chỉ số → a[] tăng dần (front = min)
    deque<int> dqMax, dqMin;
    long long ans = 0;
    int l = 0; // Con trỏ trái

    for(int r = 0; r < n; r++) {
        // Thêm a[r] vào 2 deque
        while(!dqMax.empty() && a[dqMax.back()] <= a[r]) dqMax.pop_back();
        dqMax.push_back(r);
        while(!dqMin.empty() && a[dqMin.back()] >= a[r]) dqMin.pop_back();
        dqMin.push_back(r);

        // Thu hẹp [l, r] cho đến khi max - min ≤ K
        while(a[dqMax.front()] - a[dqMin.front()] > K) {
            l++;
            if(dqMax.front() < l) dqMax.pop_front();
            if(dqMin.front() < l) dqMin.pop_front();
        }

        // Mọi đoạn [l, r], [l+1, r], ..., [r, r] đều êm
        ans += r - l + 1;
    }

    out << ans;
    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 100): Duyệt mọi cặp (l, r), tìm max/min bằng vòng lặp — O(N³).
Subtask 2 (N ≤ 5000): Duyệt (l, r), cập nhật max/min khi mở rộng r — O(N²).
Subtask 3 (N ≤ 10⁶): Nhận xét quan trọng: khi r cố định, nếu [l, r] êm thì [l+1, r], ..., [r, r] cũng êm (thu hẹp đoạn → max giảm hoặc giữ, min tăng hoặc giữ). Dùng hai con trỏ: con trỏ l chỉ tiến, con trỏ r quét từ trái sang phải. Duy trì max/min bằng 2 deque đơn điệu. Mỗi phần tử vào/ra deque đúng 1 lần → O(N).
Bài 2: Chia giai đoạn (CHIAGD)
📝 Code C++ — O(N²) DP 
// Bài 2: CHIAGD — QHĐ chia giai đoạn O(N²)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("CHIAGD.INP");
    ofstream out("CHIAGD.OUT");

    int n;
    long long S;
    in >> n >> S;

    // prefix[i] = t[1] + t[2] + ... + t[i]
    vector<long long> prefix(n + 1, 0);
    for(int i = 1; i <= n; i++) {
        long long t;
        in >> t;
        prefix[i] = prefix[i - 1] + t;
    }

    // dp[i] = chi phí nhỏ nhất xử lý i công việc đầu
    // Đáy: dp[0] = -S (vì giai đoạn đầu tiên không nghỉ trước)
    // dp[i] = min(dp[j] + (prefix[i]-prefix[j])² + S) với j = 0..i-1
    //   → chia giai đoạn [j+1..i], mệt = (tổng)², cộng S cho lần nghỉ
    //   → trick: dp[0] = -S để giai đoạn đầu không tính S thừa
    vector<long long> dp(n + 1, 1e18);
    dp[0] = -S; // Giai đoạn đầu: triệt tiêu +S trong công thức

    for(int i = 1; i <= n; i++) {
        for(int j = 0; j < i; j++) {
            long long T = prefix[i] - prefix[j]; // Tổng giai đoạn [j+1..i]
            long long cost = dp[j] + T * T + S;   // Mệt + nghỉ
            dp[i] = min(dp[i], cost);
        }
    }

    out << dp[n];
    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 100): Brute force — thử mọi cách chia bằng đệ quy/quay lui. Hoặc DP O(N²) cũng chạy thoải mái.
Subtask 2 (N ≤ 5000): DP O(N²). Gọi P[i] = prefix sum. dp[i] = min(dp[j] + (P[i]−P[j])² + S) với j = 0..i−1 — thử mọi điểm cắt j. Trick: dp[0] = −S để giai đoạn đầu không bị cộng thừa S (vì trước giai đoạn đầu không có nghỉ). Với N = 5000, vòng lặp O(N²) ≈ 2.5×10⁷ thao tác, chạy thoải mái.
Bài 3: Đường đi ưu tiên (DUONGUU)
📝 Code C++ — 2 lần Dijkstra + duyệt cạnh O(M log N) 
// Bài 3: DUONGUU — 2 lần Dijkstra + duyệt cạnh
#include <bits/stdc++.h>
using namespace std;

int n, m;
vector<vector<pair<int,int>>> ke, ke_nguoc;
vector<long long> dist;
vector<bool> Free;

void Dijkstra(int S) {
    dist.assign(n + 1, LLONG_MAX);
    Free.assign(n + 1, true);
    priority_queue<pair<long long,int>,
                   vector<pair<long long,int>>,
                   greater<pair<long long,int>>> pq;
    dist[S] = 0; pq.push(make_pair(0LL, S));

    while (!pq.empty()) {
        long long d = pq.top().first; int u = pq.top().second; pq.pop();
        if (!Free[u]) continue;
        Free[u] = false;
        for (int i = 0; i < (int)ke[u].size(); i++) {
            int v = ke[u][i].first, w = ke[u][i].second;
            if (Free[v] && dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                pq.push(make_pair(dist[v], v));
            }
        }
    }
}

int main() {
    ifstream in("DUONGUU.INP");
    ofstream out("DUONGUU.OUT");
    in >> n >> m;

    ke.assign(n + 1, vector<pair<int,int>>());
    ke_nguoc.assign(n + 1, vector<pair<int,int>>());
    vector<int> eu(m), ev(m);

    for (int i = 0; i < m; i++) {
        int w;
        in >> eu[i] >> ev[i] >> w;
        ke[eu[i]].push_back(make_pair(ev[i], w));
        ke_nguoc[ev[i]].push_back(make_pair(eu[i], w));
    }

    // Dijkstra lần 1: từ đỉnh 1 trên đồ thị gốc
    Dijkstra(1);
    vector<long long> dist1 = dist;

    // Dijkstra lần 2: từ đỉnh N trên đồ thị ngược
    swap(ke, ke_nguoc);
    Dijkstra(n);
    // dist chính là distN

    // Thử dùng VIP trên từng cạnh u→v: bỏ trọng số w
    // Chi phí = dist1[u] + 0 + distN[v]
    long long ans = dist1[n]; // Không dùng VIP
    for (int i = 0; i < m; i++) {
        if (dist1[eu[i]] != LLONG_MAX && dist[ev[i]] != LLONG_MAX)
            ans = min(ans, dist1[eu[i]] + dist[ev[i]]);
    }

    if (ans == LLONG_MAX) out << -1;
    else out << ans;

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 1000): Brute force — thử từng cạnh làm VIP (đặt trọng số = 0), mỗi lần chạy Dijkstra từ 1 → N. Độ phức tạp O(M × M log N).
Subtask 2 (Full): Ý tưởng cốt lõi: Nếu dùng VIP trên cạnh u→v, đường đi tối ưu sẽ là: 1 →...→ u → v →...→ N, với đoạn 1→u ngắn nhất và đoạn v→N ngắn nhất.
Chạy Dijkstra lần 1 từ đỉnh 1 trên đồ thị gốc → dist1[u] = khoảng cách ngắn nhất từ 1 đến u.
Chạy Dijkstra lần 2 từ đỉnh N trên đồ thị ngược → distN[v] = khoảng cách ngắn nhất từ v đến N.
Duyệt mọi cạnh (u, v): ans = min(dist1[u] + distN[v]).
Tổng: O(M log N) — chỉ 2 lần Dijkstra + 1 vòng duyệt cạnh.
