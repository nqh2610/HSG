Bài 1: Đoạn ngắn nhất (DONHAT)
📝 Code C++ — Two Pointers O(N) 
// Bài 1: DONHAT — Two Pointers
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("DONHAT.INP");
    ofstream out("DONHAT.OUT");

    int n;
    long long S;
    in >> n >> S;
    vector<long long> a(n);
    for (int i = 0; i < n; i++) in >> a[i];

    int ans = n + 1;
    long long sum = 0;
    int l = 0;

    for (int r = 0; r < n; r++) {
        sum += a[r];
        while (sum >= S) {
            ans = min(ans, r - l + 1);
            sum -= a[l];
            l++;
        }
    }

    if (ans == n + 1) out << -1;
    else out << ans;

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 1000): Duyệt mọi cặp (l, r), tính tổng đoạn, cập nhật min độ dài. Độ phức tạp O(N²).
Subtask 2 (Full): Hai con trỏ. Nhận xét: tất cả phần tử dương → khi mở rộng r, tổng tăng; khi thu hẹp l, tổng giảm.
Con trỏ r quét từ trái sang phải. Khi sum ≥ S, thu hẹp l (giảm sum) đồng thời cập nhật min(r−l+1). Con trỏ l chỉ tiến, không lùi → mỗi phần tử vào/ra cửa sổ đúng 1 lần → O(N).
Bài 2: Dãy Bitonic (DAYBIT)
📝 Code C++ — LIS hai chiều O(N²) 
// Bài 2: DAYBIT — LIS hai chiều
#include <bits/stdc++.h>
using namespace std;

int a[5001];
int lis[5001], lds[5001];

int main() {
    ifstream in("DAYBIT.INP");
    ofstream out("DAYBIT.OUT");

    int n;
    in >> n;
    for (int i = 0; i < n; i++) in >> a[i];

    // LIS kết thúc tại i (quét từ trái sang)
    for (int i = 0; i < n; i++) {
        lis[i] = 1;
        for (int j = 0; j < i; j++)
            if (a[j] < a[i])
                lis[i] = max(lis[i], lis[j] + 1);
    }

    // LDS bắt đầu từ i (quét từ phải sang)
    for (int i = n - 1; i >= 0; i--) {
        lds[i] = 1;
        for (int j = i + 1; j < n; j++)
            if (a[j] < a[i])
                lds[i] = max(lds[i], lds[j] + 1);
    }

    int ans = 1;
    for (int i = 0; i < n; i++)
        ans = max(ans, lis[i] + lds[i] - 1);

    out << ans;

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 100): Brute force — thử mọi tập con, kiểm tra bitonic. Hoặc dùng DP O(N²) cũng chạy thoải mái.
Subtask 2 (Full): Biến tấu của bài LIS (Dãy Con Tăng Dài Nhất).
Tính 2 mảng:
— lis[i] = độ dài dãy con tăng nghiêm ngặt dài nhất kết thúc tại vị trí i (quét trái → phải, giống LIS).
— lds[i] = độ dài dãy con giảm nghiêm ngặt dài nhất bắt đầu từ vị trí i (quét phải → trái, tức LIS trên dãy đảo).
Với mỗi vị trí i làm "đỉnh" bitonic: dãy tăng đến i rồi giảm từ i → độ dài = lis[i] + lds[i] − 1 (trừ 1 vì i đếm 2 lần).
Đáp án: max(lis[i] + lds[i] − 1). Tổng: O(N²). Bộ nhớ: 2 mảng O(N).
Bài 3: Chấm bài (CHAMBAI)
📝 Code C++ — Chặt nhị phân + Tham lam O(N log S) 
// Bài 3: CHAMBAI — Chặt nhị phân + Tham lam
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("CHAMBAI.INP");
    ofstream out("CHAMBAI.OUT");

    int n, K;
    in >> n >> K;
    vector<long long> a(n);
    long long lo = 0, hi = 0;
    for (int i = 0; i < n; i++) {
        in >> a[i];
        lo = max(lo, a[i]);
        hi += a[i];
    }

    while (lo < hi) {
        long long mid = (lo + hi) / 2;
        // Tham lam: chia thành ≤ K đoạn, mỗi đoạn sum ≤ mid?
        int segments = 1;
        long long sum = 0;
        for (int i = 0; i < n; i++) {
            if (sum + a[i] > mid) {
                segments++;
                sum = a[i];
            } else {
                sum += a[i];
            }
        }
        if (segments <= K)
            hi = mid;
        else
            lo = mid + 1;
    }

    out << lo;

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 100): DP O(N² × K). Gọi dp[i][j] = min max tổng đoạn khi chia i phần tử đầu thành j đoạn. Duyệt mọi điểm cắt.
Subtask 2 (Full): Bài này cần kết hợp 2 kỹ thuật:
1. Chặt nhị phân trên đáp án: Đáp án X nằm trong [max(ti), Σti]. Dùng binary search tìm X nhỏ nhất mà ta có thể chia thành ≤ K đoạn, mỗi đoạn tổng ≤ X.
2. Tham lam kiểm tra: Với X cố định, quét trái → phải. Cộng dồn ti vào đoạn hiện tại. Khi thêm ti vượt X, bắt đầu đoạn mới. Đếm số đoạn → nếu ≤ K thì X khả thi.
Khoảng tìm kiếm: [max(ti), Σti] ≤ 1015 → binary search chạy ≤ 50 vòng. Mỗi vòng kiểm tra O(N). Tổng: O(N log S).
