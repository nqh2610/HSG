Bài 1: Danh sách nộp bài (DANHSACH)
📝 Code C++ — Nén tọa độ + BIT O(N log N) 
// Bai 1: DANHSACH — Coordinate Compression + BIT O(N log N)
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 100005;
int bit[MAXN], sz;

void update(int i) {
    for (; i <= sz; i += i & (-i)) bit[i]++;
}

int query(int i) {
    int s = 0;
    for (; i > 0; i -= i & (-i)) s += bit[i];
    return s;
}

int main() {
    ifstream in("DANHSACH.INP");
    ofstream out("DANHSACH.OUT");

    int n;
    in >> n;
    vector<int> a(n);
    for (int i = 0; i < n; i++) in >> a[i];

    // Coordinate compression
    vector<int> v = a;
    sort(v.begin(), v.end());
    v.erase(unique(v.begin(), v.end()), v.end());
    sz = v.size();

    for (int i = 0; i < n; i++) {
        int r = lower_bound(v.begin(), v.end(), a[i]) - v.begin() + 1;
        if (i) out << " ";
        out << query(r - 1);
        update(r);
    }

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 5000): Duyệt trâu: với mỗi i, duyệt j từ 0 đến i−1, đếm a[j] < a[i]. Tổng O(N²) ≈ 2.5 × 10⁷.
Subtask 2 (Full): Nén tọa độ + Cây chỉ số nhị phân (BIT/Fenwick Tree):

Vấn đề: a[i] lên đến 10⁹ → không thể tạo mảng BIT cỡ 10⁹.
Giải pháp — Nén tọa độ: Thay mỗi giá trị bằng "thứ hạng" trong tập giá trị phân biệt (đã sắp xếp).
  1. Sao chép mảng, sort, loại trùng bằng unique
  2. Dùng lower_bound tìm rank của mỗi phần tử (1-indexed)
  → Giá trị nén nằm trong [1, K] với K ≤ N → BIT cỡ N là đủ

BIT (Binary Indexed Tree): Duy trì mảng tần suất. Khi xử lý phần tử i:
  • query(rank − 1): đếm tổng tần suất ở các rank 1..rank−1 = số phần tử đã thêm có giá trị < a[i]
  • update(rank): tăng tần suất tại rank lên 1

Mỗi thao tác BIT: O(log N). Tổng: O(N log N).
Bài 2: Đếm số hạnh phúc (DEMHP)
📝 Code C++ — Digit DP O(19 × K) 
// Bai 2: DEMHP — Digit DP O(19 * K)
#include <bits/stdc++.h>
using namespace std;

int K, len, d[19];
long long memo[19][201][2];

long long solve(int pos, int sum, int tight) {
    if (pos == len) return sum == 0 ? 1 : 0;
    if (memo[pos][sum][tight] != -1) return memo[pos][sum][tight];
    int lim = tight ? d[pos] : 9;
    long long res = 0;
    for (int x = 0; x <= lim; x++)
        res += solve(pos + 1, (sum + x) % K, tight && x == lim);
    return memo[pos][sum][tight] = res;
}

long long count(long long x) {
    if (x < 0) return 0;
    string s = to_string(x);
    len = s.size();
    for (int i = 0; i < len; i++) d[i] = s[i] - '0';
    memset(memo, -1, sizeof(memo));
    return solve(0, 0, 1);
}

int main() {
    ifstream in("DEMHP.INP");
    ofstream out("DEMHP.OUT");

    long long L, R;
    in >> L >> R >> K;

    out << count(R) - count(L - 1);

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (R ≤ 10⁶): Duyệt từng số x từ L đến R, tính tổng chữ số, kiểm tra chia hết K. Tổng O(R × log₁₀R) ≈ 7 × 10⁶.
Subtask 2 (Full): Digit DP (QHĐ trên từng chữ số):

Bước 1 — Tách bài toán: Đếm trong [L, R] = f(R) − f(L−1), với f(X) = đếm số trong [0, X] thỏa mãn.

Bước 2 — Digit DP: Xây dựng số từ trái sang phải (chữ số có trọng số lớn → nhỏ).
Trạng thái: solve(pos, sum, tight)
  • pos: vị trí chữ số đang xét (0 = chữ số đầu tiên)
  • sum: tổng chữ số hiện tại mod K
  • tight: có bị ràng buộc bởi giới hạn trên không?

Chuyển trạng thái: Tại vị trí pos, thử đặt chữ số x từ 0 đến lim:
  • Nếu tight = true: lim = chữ số tương ứng của X (không được vượt)
  • Nếu tight = false: lim = 9 (tự do)
  • Chuyển sang solve(pos+1, (sum+x) % K, tight && x==lim)

Cơ sở: Khi pos = len (hết chữ số), trả 1 nếu sum == 0 (chia hết K), ngược lại 0.
Tổng trạng thái: 19 × 200 × 2 = 7 600. Rất nhanh cho R ≤ 10¹⁸.
Bài 3: Lịch họp ngoại khóa (LICHHOP)
📝 Code C++ — Sort + DP + Binary Search O(N log N) 
// Bai 3: LICHHOP — Sort + DP + Binary Search O(N log N)
#include <bits/stdc++.h>
using namespace std;

struct Meeting {
    int s, e;
    long long w;
};

int main() {
    ifstream in("LICHHOP.INP");
    ofstream out("LICHHOP.OUT");

    int n;
    in >> n;
    vector<Meeting> m(n);
    for (int i = 0; i < n; i++) in >> m[i].s >> m[i].e >> m[i].w;

    // Sort by end time
    sort(m.begin(), m.end(), [](const Meeting& a, const Meeting& b) {
        return a.e < b.e;
    });

    // dp[i] = max weight considering first i meetings (1-indexed)
    vector<long long> dp(n + 1, 0);
    for (int i = 0; i < n; i++) {
        // Binary search: largest j with m[j].e <= m[i].s
        int lo = 0, hi = i - 1, p = -1;
        while (lo <= hi) {
            int mid = (lo + hi) / 2;
            if (m[mid].e <= m[i].s) { p = mid; lo = mid + 1; }
            else hi = mid - 1;
        }
        dp[i + 1] = max(dp[i], dp[p + 1] + m[i].w);
    }

    out << dp[n];

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 20): Duyệt mọi tập con 2N ≈ 10⁶, kiểm tra không chồng nhau và tính tổng. O(2N × N).
Subtask 2 (N ≤ 5000): Sort theo thời gian kết thúc. DP: với mỗi buổi i, duyệt tuyến tính tìm buổi gần nhất kết thúc trước khi i bắt đầu. O(N²).
Subtask 3 (Full): Sort + DP + Binary Search — kết hợp 3 kỹ thuật:

Bước 1 — Sort: Sắp xếp N buổi theo thời gian kết thúc e tăng dần.

Bước 2 — DP: Gọi dp[i] = tổng hấp dẫn lớn nhất khi xét i buổi đầu tiên (sau khi sort).
Với buổi thứ i, có 2 lựa chọn:
  • Bỏ qua buổi i: dp[i] = dp[i−1]
  • Chọn buổi i: tìm buổi gần nhất p kết thúc ≤ thời điểm bắt đầu s[i]. dp[i] = dp[p] + w[i]
  → dp[i] = max(dp[i−1], dp[p] + w[i])

Bước 3 — Binary Search: Vì mảng đã sort theo e, dùng tìm kiếm nhị phân O(log N) để tìm p thay vì duyệt tuyến tính O(N).

Tại sao đúng? Đây là bài toán Weighted Activity Selection kinh điển. Tính chất con tối ưu: lời giải tối ưu cho i buổi đầu hoặc bỏ qua buổi i (→ dùng dp[i−1]) hoặc chọn buổi i rồi dùng lời giải tối ưu cho các buổi kết thúc trước s[i].
Tổng: O(N log N) sort + O(N log N) DP×BS = O(N log N).
