Bài 1: Thời tiết (THOITIET)
📝 Code C++ — Monotonic Stack O(N) 
// Bai 1: THOITIET — Monotonic Stack O(N)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("THOITIET.INP");
    ofstream out("THOITIET.OUT");

    int n;
    in >> n;
    vector<int> a(n);
    for (int i = 0; i < n; i++) in >> a[i];

    vector<int> ans(n, 0);
    stack<int> st;
    for (int i = 0; i < n; i++) {
        while (!st.empty() && a[i] > a[st.top()]) {
            ans[st.top()] = i - st.top();
            st.pop();
        }
        st.push(i);
    }

    for (int i = 0; i < n; i++) {
        if (i) out << " ";
        out << ans[i];
    }

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 1000): Với mỗi ngày i, duyệt j = i+1, i+2, ... tìm ngày đầu tiên a[j] > a[i]. Tổng O(N²) ≈ 10⁶.
Subtask 2 (Full): Ngăn xếp đơn điệu (Monotonic Stack):

Ý tưởng: Duy trì stack chứa chỉ số các ngày chưa tìm được câu trả lời, sắp xếp sao cho nhiệt độ tương ứng giảm dần từ đáy lên đỉnh.

Khi duyệt đến ngày i:
  • Nếu a[i] > a[st.top()]: ngày st.top() đã tìm được câu trả lời → ans[st.top()] = i − st.top(), pop ra.
  • Lặp lại cho đến khi stack rỗng hoặc a[st.top()] ≥ a[i].
  • Đẩy i vào stack.

Tại sao đúng? Mỗi phần tử được push/pop đúng 1 lần, nên tổng thao tác = O(N). Stack luôn giảm dần → khi a[i] lớn hơn đỉnh, tức i là ngày nóng hơn GẦN NHẤT cho đỉnh stack.

Các ngày còn lại trong stack khi duyệt xong: không có ngày nóng hơn → ans = 0 (đã khởi tạo).
Độ phức tạp: O(N) thời gian, O(N) bộ nhớ.
Bài 2: Thu hoạch (THUHOACH)
📝 Code C++ — Binary Search on Answer O(N log S) 
// Bai 2: THUHOACH — Binary Search on Answer O(N log S)
#include <bits/stdc++.h>
using namespace std;

int n, k;
long long a[100001];

bool check(long long mid) {
    int cnt = 1;
    long long sum = 0;
    for (int i = 0; i < n; i++) {
        if (sum + a[i] > mid) {
            cnt++;
            sum = a[i];
        } else {
            sum += a[i];
        }
    }
    return cnt <= k;
}

int main() {
    ifstream in("THUHOACH.INP");
    ofstream out("THUHOACH.OUT");

    in >> n >> k;
    long long lo = 0, hi = 0;
    for (int i = 0; i < n; i++) {
        in >> a[i];
        if (a[i] > lo) lo = a[i];
        hi += a[i];
    }

    while (lo < hi) {
        long long mid = (lo + hi) / 2;
        if (check(mid)) hi = mid;
        else lo = mid + 1;
    }

    out << lo;

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 1000): Thử tất cả cách chia bằng QHĐ/brute, hoặc binary search + check O(N). Tổng O(N × K) hoặc O(N log S).
Subtask 2 (Full): Tìm kiếm nhị phân trên đáp án (Binary Search on Answer):

Nhận xét: Nếu max cho phép mỗi người thu tối đa M tạ, ta kiểm tra "M có đủ để chia N ruộng cho K người không?" bằng tham lam: duyệt từ trái sang, cứ gom liên tiếp cho đến khi thêm ruộng kế sẽ vượt M → chuyển người mới.

Hàm check(M): Đếm số người cần thiết. Nếu ≤ K → khả thi (M đủ lớn).

Binary search:
  • lo = max(ai) (không thể nhỏ hơn ruộng lớn nhất)
  • hi = tổng(ai) (1 người thu hết)
  • Nếu check(mid) đúng → thu hẹp hi = mid
  • Nếu sai → lo = mid + 1

Tại sao tham lam đúng? Nếu M là max cho phép, gom càng nhiều ruộng cho 1 người càng tốt → tiết kiệm người cho các ruộng sau. Nếu tham lam cần > K người → M quá nhỏ.

Lưu ý: Dùng long long vì tổng có thể đạt 10⁵ × 10⁹ = 10¹⁴.
Độ phức tạp: O(N × log(sum)) ≈ O(N × 47).
Bài 3: Đoạn phủ (DOANPHU)
📝 Code C++ — Nén tọa độ + Sweep O(N log N) 
// Bai 3: DOANPHU — Coordinate Compression + Sweep O(N log N)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("DOANPHU.INP");
    ofstream out("DOANPHU.OUT");

    int n;
    in >> n;
    vector<int> l(n), r(n);
    vector<int> coords;
    for (int i = 0; i < n; i++) {
        in >> l[i] >> r[i];
        coords.push_back(l[i]);
        coords.push_back(r[i]);
    }

    // Nen toa do
    sort(coords.begin(), coords.end());
    coords.erase(unique(coords.begin(), coords.end()), coords.end());
    int K = coords.size();

    // Diff array tren toa do nen
    vector<int> diff(K + 1, 0);
    for (int i = 0; i < n; i++) {
        int cl = lower_bound(coords.begin(), coords.end(), l[i]) - coords.begin();
        int cr = lower_bound(coords.begin(), coords.end(), r[i]) - coords.begin();
        diff[cl]++;
        diff[cr]--;
    }

    // Prefix sum, tinh tong chieu dai
    long long ans = 0;
    int cnt = 0;
    for (int i = 0; i < K - 1; i++) {
        cnt += diff[i];
        if (cnt > 0)
            ans += coords[i + 1] - coords[i];
    }

    out << ans;

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 1000, r ≤ 10⁶): Sắp xếp đoạn theo l, gộp các đoạn chồng nhau, tính tổng chiều dài. O(N²) hoặc O(N log N).
Subtask 2 (Full): Nén tọa độ (Coordinate Compression):

Vấn đề: Tọa độ lên đến 10⁹ → không thể tạo mảng kích thước 10⁹. Nhưng chỉ có 2N ≤ 2×10⁵ điểm đặc biệt (các đầu mút li, ri).

Ý tưởng nén:
  • Bước 1: Thu thập tất cả 2N tọa độ, sắp xếp, loại trùng → K giá trị phân biệt.
  • Bước 2: Ánh xạ mỗi tọa độ thực → chỉ số nén 0, 1, ..., K−1.
  • Bước 3: Dùng mảng hiệu trên chỉ số nén: diff[nén(l)]++, diff[nén(r)]−−.
  • Bước 4: Duyệt prefix sum. Với mỗi đoạn giữa 2 tọa độ nén liên tiếp [coords[i], coords[i+1]], nếu đang bị phủ (cnt > 0) → cộng chiều dài thực coords[i+1] − coords[i].

Tại sao đúng? Giữa hai tọa độ nén liên tiếp, không có đầu mút nào → trạng thái phủ là đồng nhất. Ta chỉ cần kiểm tra tại mỗi đoạn con.

Cách khác: Sort + Merge (sắp xếp đoạn theo l, gộp đoạn chồng) cũng O(N log N) và đơn giản hơn. Nhưng nén tọa độ là kỹ thuật tổng quát hơn, áp dụng được cho nhiều bài phức tạp hơn.
Độ phức tạp: O(N log N) thời gian, O(N) bộ nhớ.
