Bài 1: Chuyến xe liên tỉnh (CHUYENXE)
📝 Code C++ — Floyd-Warshall O(N³) 
// Bai 1: CHUYENXE — Floyd-Warshall O(N^3)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("CHUYENXE.INP");
    ofstream out("CHUYENXE.OUT");

    int n, m, q;
    in >> n >> m >> q;

    const long long INF = 1e18;
    vector<vector<long long> > d(n + 1, vector<long long>(n + 1, INF));
    for (int i = 1; i <= n; i++) d[i][i] = 0;
    for (int i = 0; i < m; i++) {
        int u, v;
        long long w;
        in >> u >> v >> w;
        d[u][v] = min(d[u][v], w);
    }

    for (int k = 1; k <= n; k++)
        for (int i = 1; i <= n; i++)
            for (int j = 1; j <= n; j++)
                if (d[i][k] < INF && d[k][j] < INF)
                    d[i][j] = min(d[i][j], d[i][k] + d[k][j]);

    for (int i = 0; i < q; i++) {
        int s, t;
        in >> s >> t;
        if (d[s][t] >= INF) out << -1;
        else out << d[s][t];
        out << "\n";
    }

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 50, Q ≤ 100): Chạy Dijkstra cho mỗi truy vấn. O(Q × (M + N log N)) ≈ 100 × (2500 + 300) ≈ 3 × 10⁵.
Subtask 2 (Full): Floyd-Warshall — Tìm đường ngắn nhất mọi cặp đỉnh:

Ý tưởng: Xây ma trận khoảng cách d[i][j] cho MỌI cặp (i, j). Sau đó mỗi truy vấn chỉ cần tra bảng O(1).

Thuật toán Floyd (3 vòng lặp):
  for k = 1 → N: for i = 1 → N: for j = 1 → N:
    d[i][j] = min(d[i][j], d[i][k] + d[k][j])

Ý nghĩa: Vòng k xét nút trung gian k. Nếu đi i → k → j rẻ hơn đường hiện tại i → j, thì cập nhật.

Tại sao k ở ngoài cùng? Đảm bảo khi xét k, tất cả đường qua {1, 2, …, k−1} đã được tính. Đây là bản chất QHĐ: dk[i][j] = đường ngắn nhất i→j chỉ qua trung gian ∈ {1,…,k}.

Kiểm tra INF: Trước khi cộng d[i][k] + d[k][j], kiểm tra cả hai < INF để tránh tràn số.

Tại sao không dùng Dijkstra? Dijkstra cho 1 nguồn → chạy N lần = O(N × (M + N log N)). Với N = 400: 400 × (160000 + 3500) ≈ 6.5 × 10⁷. Floyd: 400³ = 6.4 × 10⁷. Tương đương, nhưng Floyd code đơn giản hơn nhiều.

Tại sao không dùng BFS? Đồ thị có trọng số → BFS thuần không áp dụng được.
Độ phức tạp: Tiền xử lý O(N³), mỗi query O(1). Tổng O(N³ + Q).
Bài 2: Chia đôi (CHIADOI)
📝 Code C++ — Bipartite BFS O(N+M) 
// Bai 2: CHIADOI — Bipartite BFS O(N+M)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("CHIADOI.INP");
    ofstream out("CHIADOI.OUT");

    int n, m;
    in >> n >> m;
    vector<vector<int> > adj(n + 1);
    for (int i = 0; i < m; i++) {
        int u, v;
        in >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    vector<int> color(n + 1, -1);
    bool ok = true;
    for (int s = 1; s <= n && ok; s++) {
        if (color[s] != -1) continue;
        queue<int> q;
        color[s] = 1;
        q.push(s);
        while (!q.empty() && ok) {
            int u = q.front();
            q.pop();
            for (int j = 0; j < (int)adj[u].size(); j++) {
                int v = adj[u][j];
                if (color[v] == -1) {
                    color[v] = 3 - color[u];
                    q.push(v);
                } else if (color[v] == color[u]) {
                    ok = false;
                }
            }
        }
    }

    if (!ok) {
        out << "NO";
    } else {
        out << "YES" << "\n";
        for (int i = 1; i <= n; i++) {
            if (i > 1) out << " ";
            out << color[i];
        }
    }

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N, M ≤ 1000): BFS/DFS tô màu. Cùng thuật toán, subtask chỉ khác ràng buộc.
Subtask 2 (Full): Kiểm tra đồ thị 2 phía (Bipartite Check) bằng BFS tô 2 màu:

Định lý: Đồ thị chia được 2 nhóm thỏa mãn ⟺ đồ thị là đồ thị 2 phía (bipartite) ⟺ không có chu trình lẻ.

Thuật toán:
  1. Khởi tạo color[] = −1 (chưa tô)
  2. Với mỗi đỉnh chưa tô s: tô color[s] = 1, BFS mở rộng
  3. Khi xét cạnh u → v:
    • Nếu v chưa tô: color[v] = 3 − color[u] (đổi màu: 1↔2)
    • Nếu v cùng màu u: phát hiện chu trình lẻ → không thể chia → "NO"
  4. Nếu BFS hoàn tất không mâu thuẫn → "YES" + xuất color[]

Tại sao "3 − color[u]"? Nếu u màu 1 → v màu 2. Nếu u màu 2 → v màu 1. Phép 3 − x đổi giữa 1 và 2.

Xử lý đồ thị không liên thông: Vòng for s = 1 → N đảm bảo mọi thành phần liên thông đều được kiểm tra.

Ví dụ 1: Chu trình 4 đỉnh (1–2–3–4–1) là chu trình chẵn → bipartite: {1,3} vs {2,4}.
Ví dụ 2: Tam giác (1–2–3–1) là chu trình lẻ → không bipartite → "NO".

Độ phức tạp: O(N + M) — BFS chuẩn.
Bài 3: Giảm giá (GIAMGIA)
📝 Code C++ — Bellman-Ford O(N×M) 
// Bai 3: GIAMGIA — Bellman-Ford O(N*M)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("GIAMGIA.INP");
    ofstream out("GIAMGIA.OUT");

    int n, m;
    in >> n >> m;

    vector<int> eu(m), ev(m);
    vector<long long> ew(m);
    for (int i = 0; i < m; i++)
        in >> eu[i] >> ev[i] >> ew[i];

    const long long INF = 1e18;
    vector<long long> d(n + 1, INF);
    d[1] = 0;
    for (int i = 0; i < n - 1; i++) {
        bool updated = false;
        for (int j = 0; j < m; j++) {
            if (d[eu[j]] < INF && d[eu[j]] + ew[j] < d[ev[j]]) {
                d[ev[j]] = d[eu[j]] + ew[j];
                updated = true;
            }
        }
        if (!updated) break;
    }

    if (d[n] >= INF) out << -1;
    else out << d[n];

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (w ≥ 0, N ≤ 100): Dijkstra chuẩn. O(M + N log N) ≈ 5000 + 700 ≈ 6000. Hoặc thậm chí BFS nếu trọng số bằng nhau.
Subtask 2 (w có thể âm): Dijkstra KHÔNG hoạt động với cạnh âm! Phải dùng Bellman-Ford:

Tại sao Dijkstra sai với cạnh âm? Dijkstra greedy: đỉnh đã lấy ra khỏi PQ coi là "tối ưu". Nhưng cạnh âm có thể giảm khoảng cách sau khi đỉnh đã bị đánh dấu → sai.

Thuật toán Bellman-Ford:
  1. Khởi tạo: d[1] = 0, d[i] = +∞ ∀ i ≠ 1
  2. Lặp N−1 lần: duyệt toàn bộ M cạnh, relax mỗi cạnh (u, v, w):
    Nếu d[u] + w < d[v] → d[v] = d[u] + w
  3. Đáp án = d[N]

Tại sao N−1 lần? Đường ngắn nhất có tối đa N−1 cạnh. Sau lần lặp thứ k, d[v] chính xác cho mọi đường qua ≤ k cạnh. Sau N−1 lần → chính xác hoàn toàn.

Tối ưu early termination: Nếu một lần lặp không relax được cạnh nào (updated = false) → đã hội tụ → break sớm. Trường hợp tốt nhất chỉ 1 vòng lặp.

Ví dụ: 1→2 = 3, 2→3 = −2, 3→4 = 2. Đường 1→2→3→4 = 3 + (−2) + 2 = 3. Rẻ hơn 1→2→4 = 9.

Lưu ý: Kiểm tra d[eu[j]] < INF trước khi cộng, tránh tràn số (INF + w có thể overflow).
Độ phức tạp: O(N × M) = 1000 × 5000 = 5 × 10⁶ — đủ nhanh.
