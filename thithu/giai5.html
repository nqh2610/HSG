Bài 1: Cửa sổ (CUASO)
📝 Code C++ — Sliding Window O(N) 
// Bài 1: CUASO — Sliding Window O(N)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("CUASO.INP");
    ofstream out("CUASO.OUT");

    int n;
    long long s;
    in >> n >> s;

    vector<long long> a(n);
    for (int i = 0; i < n; i++) in >> a[i];

    int ans = n + 1;
    long long sum = 0;
    int left = 0;

    for (int right = 0; right < n; right++) {
        sum += a[right];
        while (sum >= s) {
            ans = min(ans, right - left + 1);
            sum -= a[left];
            left++;
        }
    }

    out << (ans == n + 1 ? 0 : ans);

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 1000): Duyệt tất cả đoạn con O(N²): với mỗi left, tăng right cho đến khi sum ≥ S, cập nhật min length.
Subtask 2 (Full): Kỹ thuật Sliding Window (cửa sổ trượt):

Dùng 2 con trỏ left và right. Mở rộng right để thêm phần tử. Khi sum ≥ S, thu hẹp bên trái (tăng left) để tìm đoạn ngắn nhất.

Tại sao đúng? Vì tất cả a[i] > 0, nên khi thu hẹp bên trái, tổng giảm. Khi mở rộng bên phải, tổng tăng. Do đó left chỉ đi sang phải (không quay lại), mỗi phần tử được thêm/bỏ đúng 1 lần → O(N).

Lưu ý: S có thể lên đến 1015 → cần dùng long long cho sum và S.
Bài 2: Đếm ước (DEMUOC)
📝 Code C++ — Sàng SPF + Đếm ước O(N log log N) 
// Bài 2: DEMUOC — Sieve (SPF) + Divisor Count
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("DEMUOC.INP");
    ofstream out("DEMUOC.OUT");

    int n, k;
    in >> n >> k;

    // Sieve of smallest prime factor
    vector<int> spf(n + 1, 0);
    for (int i = 2; i <= n; i++) {
        if (spf[i] == 0) {
            for (int j = i; j <= n; j += i) {
                if (spf[j] == 0) spf[j] = i;
            }
        }
    }

    int cnt = 0;
    for (int x = 1; x <= n; x++) {
        // Compute number of divisors from prime factorization
        int d = 1;
        int tmp = x;
        while (tmp > 1) {
            int p = spf[tmp];
            int e = 0;
            while (tmp % p == 0) {
                tmp /= p;
                e++;
            }
            d *= (e + 1);
        }
        if (d == k) cnt++;
    }

    out << cnt;

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 104): Với mỗi x, đếm ước bằng cách duyệt từ 1 đến √x. Tổng: O(N√N) ≈ 106.
Subtask 2 (N ≤ 107): O(N√N) ≈ 3×1010 — quá chậm! Cần Sàng ước số nguyên tố nhỏ nhất (SPF Sieve):

Bước 1 — Sàng SPF: Với mỗi số nguyên tố p (phát hiện khi spf[p] = 0), đánh dấu spf[j] = p cho mọi bội j của p chưa được đánh dấu. Giống sàng Eratosthenes nhưng lưu ước nguyên tố nhỏ nhất thay vì chỉ đánh dấu.

Bước 2 — Phân tích nhanh: Với mỗi x, chia liên tục cho spf[x] để lấy phân tích thừa số nguyên tố: x = p₁e₁ × p₂e₂ × … Mỗi bước chia mất O(1) tra bảng → phân tích x mất O(log x).

Bước 3 — Công thức ước: Nếu x = p₁e₁ × p₂e₂ × … × pkek thì d(x) = (e₁+1)(e₂+1)…(ek+1). So sánh d(x) với K.

Tổng: O(N log log N) cho sàng + O(N log N) cho phân tích tất cả số.
Bài 3: Chọn cây (CHONCAY)
📝 Code C++ — DFS + QHĐ trên cây O(N) 
// Bài 3: CHONCAY — DFS (BFS order) + Tree DP
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("CHONCAY.INP");
    ofstream out("CHONCAY.OUT");

    int n;
    in >> n;

    vector<long long> w(n + 1);
    for (int i = 1; i <= n; i++) in >> w[i];

    vector<vector<int>> adj(n + 1);
    for (int i = 0; i < n - 1; i++) {
        int u, v;
        in >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    // BFS to get traversal order + parent
    vector<int> order, par(n + 1, 0);
    vector<bool> visited(n + 1, false);
    queue<int> q;
    q.push(1);
    visited[1] = true;
    while (!q.empty()) {
        int u = q.front(); q.pop();
        order.push_back(u);
        for (int i = 0; i < (int)adj[u].size(); i++) {
            int v = adj[u][i];
            if (!visited[v]) {
                visited[v] = true;
                par[v] = u;
                q.push(v);
            }
        }
    }

    // Process in reverse BFS order (leaves first)
    vector<long long> dp0(n + 1, 0), dp1(n + 1);
    for (int i = 1; i <= n; i++) dp1[i] = w[i];

    for (int i = (int)order.size() - 1; i >= 0; i--) {
        int u = order[i];
        for (int j = 0; j < (int)adj[u].size(); j++) {
            int v = adj[u][j];
            if (v == par[u]) continue;
            dp0[u] += max(dp0[v], dp1[v]);
            dp1[u] += dp0[v];
        }
    }

    out << max(dp0[1], dp1[1]);

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 20): Duyệt tất cả 2N tập con, kiểm tra không có cặp kề nhau, lấy max tổng trọng số.
Subtask 2 (Full): Bài này cần kết hợp 2 kỹ thuật:

1. DFS/BFS duyệt cây: Chọn gốc (đỉnh 1), BFS lấy thứ tự duyệt + xác định cha mỗi đỉnh. Duyệt ngược (lá trước, gốc sau) để đảm bảo khi xử lý đỉnh u, tất cả con đã được xử lý.

2. QHĐ trên cây (Tree DP): Với mỗi đỉnh u, lưu 2 trạng thái:
  • dp0[u] = tổng max khi không chọn u → mỗi con v được chọn hoặc không: dp0[u] = Σ max(dp0[v], dp1[v])
  • dp1[u] = tổng max khi chọn u → tất cả con v không được chọn: dp1[u] = w[u] + Σ dp0[v]

Đáp án = max(dp0[1], dp1[1]).

Tại sao dùng BFS thay DFS đệ quy? Với N ≤ 105, cây có thể là đường thẳng (sâu 105 mức) → DFS đệ quy tràn stack. BFS + duyệt ngược an toàn hơn, vẫn O(N).
Tổng: O(N) — BFS O(N) + DP O(N).
