Bài 1: Lợi nhuận (DOANHSO)
📝 Code C++ — Kadane's Algorithm O(N) 
// Bai 1: DOANHSO — Kadane Max Subarray Sum O(N)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("DOANHSO.INP");
    ofstream out("DOANHSO.OUT");

    int n;
    in >> n;
    long long x, maxEnd, maxSoFar;
    in >> x;
    maxEnd = maxSoFar = x;

    for (int i = 1; i < n; i++) {
        in >> x;
        maxEnd = max(x, maxEnd + x);
        maxSoFar = max(maxSoFar, maxEnd);
    }

    out << maxSoFar;

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 1000): Duyệt mọi cặp (l, r): tính tổng a[l..r], cập nhật max. Dùng prefix sum có thể tính tổng O(1). Tổng O(N²) ≈ 10⁶.
Subtask 2 (Full): Thuật toán Kadane O(N):

Ý tưởng: Duy trì 2 biến:
  • maxEnd = tổng lớn nhất của đoạn con kết thúc tại vị trí hiện tại
  • maxSoFar = tổng lớn nhất toàn bộ (đáp án)

Chuyển trạng thái: Tại vị trí i, đoạn con kết thúc tại i hoặc:
  • Nối tiếp đoạn trước: maxEnd + a[i]
  • Bắt đầu đoạn mới từ a[i]: a[i]
→ maxEnd = max(a[i], maxEnd + a[i])

Tại sao đúng? Nếu maxEnd < 0, đoạn cũ chỉ "kéo giảm" tổng → bắt đầu lại từ a[i] tốt hơn. Ngược lại, maxEnd ≥ 0 → nối thêm a[i] chỉ có lợi hoặc không thay đổi.

Lưu ý: Dùng long long vì tổng có thể đạt 10⁵ × 10⁹ = 10¹⁴.
Độ phức tạp: O(N) thời gian, O(1) bộ nhớ phụ.
Bài 2: Cân bằng (XEPSO)
📝 Code C++ — Subset Sum DP O(N×S) 
// Bai 2: XEPSO — Subset Sum DP O(N*S)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("XEPSO.INP");
    ofstream out("XEPSO.OUT");

    int n;
    in >> n;
    vector<int> a(n);
    int S = 0;
    for (int i = 0; i < n; i++) { in >> a[i]; S += a[i]; }

    int half = S / 2;
    vector<bool> dp(half + 1, false);
    dp[0] = true;

    for (int i = 0; i < n; i++) {
        for (int j = half; j >= a[i]; j--) {
            if (dp[j - a[i]]) dp[j] = true;
        }
    }

    for (int j = half; j >= 0; j--) {
        if (dp[j]) {
            out << S - 2 * j;
            break;
        }
    }

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N ≤ 20): Duyệt toàn bộ 2N ≈ 10⁶ tập con, tính tổng mỗi tập, tìm độ chênh lệch nhỏ nhất.
Subtask 2 (Full): QHĐ Tổng tập con (Subset Sum DP):

Nhận xét: Gọi S = tổng tất cả. Nếu một nhóm có tổng = j thì nhóm còn lại có tổng = S − j. Chênh lệch = |S − 2j|. Để minimize chênh lệch → cần j gần S/2 nhất có thể.

Bài toán quy về: Tìm giá trị j lớn nhất ≤ S/2 sao cho tồn tại tập con có tổng đúng bằng j.

Công thức QHĐ:
  • dp[j] = true/false — có tồn tại tập con nào có tổng đúng bằng j?
  • Cơ sở: dp[0] = true (tập rỗng có tổng 0)
  • Chuyển trạng thái: Với mỗi phần tử a[i], duyệt j giảm dần từ half xuống a[i]:
    Nếu dp[j − a[i]] = true → dp[j] = true

Tại sao duyệt giảm? Đảm bảo mỗi phần tử chỉ dùng tối đa 1 lần (0/1 Knapsack). Nếu duyệt tăng, một phần tử có thể bị "cộng nhiều lần" (Unbounded Knapsack).

Đáp án: Tìm j lớn nhất ≤ S/2 mà dp[j] = true. Chênh lệch = S − 2j.
Độ phức tạp: O(N × S) với S ≤ 100 × 1000 = 10⁵. Tổng ≈ 10⁷ — nhanh.
Bài 3: Đếm đường đi (DEMDUONG)
📝 Code C++ — Grid DP + Big Number Addition O(N×M×digits) 
// Bai 3: DEMDUONG — Grid DP + Big Number Addition O(N*M*digits)
#include <bits/stdc++.h>
using namespace std;

string addBig(const string& a, const string& b) {
    if (a == "0") return b;
    if (b == "0") return a;
    string res;
    int carry = 0, i = a.size() - 1, j = b.size() - 1;
    while (i >= 0 || j >= 0 || carry) {
        int s = carry;
        if (i >= 0) s += a[i--] - '0';
        if (j >= 0) s += b[j--] - '0';
        res += (char)('0' + s % 10);
        carry = s / 10;
    }
    reverse(res.begin(), res.end());
    return res;
}

int main() {
    ifstream in("DEMDUONG.INP");
    ofstream out("DEMDUONG.OUT");

    int n, m, k;
    in >> n >> m >> k;

    vector<vector<bool>> blk(n + 1, vector<bool>(m + 1, false));
    for (int i = 0; i < k; i++) {
        int r, c; in >> r >> c;
        blk[r][c] = true;
    }

    vector<string> prev(m + 1, "0"), curr(m + 1, "0");

    if (!blk[1][1]) prev[1] = "1";
    for (int j = 2; j <= m; j++) {
        if (!blk[1][j]) prev[j] = prev[j - 1];
    }

    for (int i = 2; i <= n; i++) {
        for (int j = 0; j <= m; j++) curr[j] = "0";
        for (int j = 1; j <= m; j++) {
            if (blk[i][j]) continue;
            curr[j] = prev[j];
            if (j > 1) curr[j] = addBig(curr[j], curr[j - 1]);
        }
        swap(prev, curr);
    }

    out << prev[m];

    in.close();
    out.close();
    return 0;
}
💡 Phân tích:
Subtask 1 (N, M ≤ 20): DP trên lưới bình thường, dùng long long (C(38, 19) ≈ 3.5 × 10¹⁰ — vẫn vừa long long). Không cần số lớn.
Subtask 2 (Full): Cần kết hợp 2 kĩ thuật: QHĐ trên lưới + Cộng xâu số lớn:

Kĩ thuật 1 — QHĐ trên lưới:
  • dp[i][j] = số đường đi từ (1,1) đến (i,j)
  • Cơ sở: dp[1][1] = 1 (nếu không bị chặn)
  • Chuyển trạng thái: dp[i][j] = dp[i−1][j] + dp[i][j−1] (nếu ô không bị chặn)
  • Nếu ô (i,j) bị chặn: dp[i][j] = 0
  • Tối ưu bộ nhớ: chỉ cần 2 hàng (prev, curr) → O(M) space

Kĩ thuật 2 — Cộng xâu số lớn:
Với N, M = 500, kết quả là C(998, 499) ≈ 10²⁹⁹ (khoảng 300 chữ số) — vượt xa long long (≈ 9.2 × 10¹⁸). Mỗi ô dp cần lưu dưới dạng string và phép cộng thực hiện bằng cộng xâu số lớn.

Hàm addBig(a, b):
  • Duyệt từ cuối cả 2 xâu about cùng lúc, cộng từng chữ số + carry
  • Xây chuỗi kết quả ngược, cuối cùng reverse
  • Tối ưu: nếu a hoặc b là "0", trả ngay xâu kia (tránh copy thừa)

Tại sao khó? Phải nắm vững DP lưới, xử lý ô bị chặn, rolling array, và cài đặt cộng xâu số lớn chính xác. Sai bất kỳ phần nào cũng cho kết quả sai.

Độ phức tạp: O(N × M × D) với D ≈ 300 chữ số. Tổng ≈ 500 × 500 × 300 = 7.5 × 10⁷ — chạy tốt trong 2 giây.
