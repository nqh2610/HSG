Bài 1: LEO CẦU THANG
CAUTHANG.INP / CAUTHANG.OUT

Một cầu thang có N bậc, đánh số từ 1 đến N. Mỗi bậc thứ i có điểm thưởng Ai (có thể âm). Một học sinh đứng ở mặt đất (bậc 0) và muốn leo lên bậc N. Tại mỗi bước, học sinh có thể bước lên 1 hoặc 2 bậc.

Hãy tìm tổng điểm thưởng lớn nhất mà học sinh có thể thu được khi leo từ bậc 0 lên bậc N.

Dữ liệu vào (CAUTHANG.INP):
  • Dòng 1: Số nguyên N (1 ≤ N ≤ 105)
  • Dòng 2: N số nguyên A1, A2, ..., AN (|Ai| ≤ 109)
Kết quả ra (CAUTHANG.OUT):
  • Một số nguyên duy nhất là tổng điểm thưởng lớn nhất

5
3 -1 7 2 -5

CAUTHANG.OUT

7
📝 Code C++ 
// Bài 1: CAUTHANG - DP cơ bản
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("CAUTHANG.INP");
    ofstream out("CAUTHANG.OUT");
    
    int n;
    in >> n;
    
    vector<long long> a(n + 1), dp(n + 1);
    for(int i = 1; i <= n; i++) in >> a[i];

    // dp[i] = điểm lớn nhất khi leo đến bậc i
    dp[0] = 0;
    dp[1] = a[1];
    for(int i = 2; i <= n; i++) {
        dp[i] = max(dp[i-1], dp[i-2]) + a[i];
    }
    
    out << dp[n];
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: dp[i] = điểm lớn nhất đến bậc i. dp[i] = max(dp[i-1], dp[i-2]) + a[i]. Độ phức tạp: O(N).
Bài 2: ĐỔI TIỀN
DOITIEN.INP / DOITIEN.OUT

Ngân hàng có N loại tiền xu, loại thứ i có mệnh giá Ci đồng (mỗi loại có số lượng không giới hạn). Khách hàng cần đổi đúng S đồng.

Hãy tìm số đồng xu ít nhất cần dùng để đổi đúng S đồng. Nếu không thể đổi được, in -1.

Dữ liệu vào (DOITIEN.INP):
  • Dòng 1: Hai số nguyên N và S (1 ≤ N ≤ 100; 1 ≤ S ≤ 106)
  • Dòng 2: N số nguyên C1, C2, ..., CN (1 ≤ Ci ≤ 106)
Kết quả ra (DOITIEN.OUT):
  • Một số nguyên duy nhất là số đồng xu ít nhất, hoặc -1 nếu không đổi được

// Bài 2: DOITIEN - DP đổi tiền (Unbounded Knapsack)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("DOITIEN.INP");
    ofstream out("DOITIEN.OUT");
    
    int n, s;
    in >> n >> s;
    
    vector<int> c(n);
    for(int i = 0; i < n; i++) in >> c[i];
    
    // dp[j] = số xu ít nhất để đổi j đồng
    vector<int> dp(s + 1, s + 1);
    dp[0] = 0;
    
    for(int j = 1; j <= s; j++) {
        for(int i = 0; i < n; i++) {
            if(c[i] <= j && dp[j - c[i]] + 1 < dp[j])
                dp[j] = dp[j - c[i]] + 1;
        }
    }
    
    out << (dp[s] > s ? -1 : dp[s]);
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: dp[j] = số xu ít nhất để đổi j đồng. Duyệt mỗi mệnh giá, cập nhật dp[j] = min(dp[j], dp[j-c[i]] + 1). Dạng Unbounded Knapsack. Độ phức tạp: O(N × S).

// Bài 3: DAYCON - LIS bằng chặt nhị phân O(NlogN)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("DAYCON.INP");
    ofstream out("DAYCON.OUT");
    
    int n;
    in >> n;
    
    vector<int> a(n);
    for(int i = 0; i < n; i++) in >> a[i];
    
    // tail[i] = phần tử nhỏ nhất kết thúc LIS độ dài i+1
    vector<int> tail;
    for(int i = 0; i < n; i++) {
        int pos = lower_bound(tail.begin(), tail.end(), a[i]) - tail.begin();
        if(pos == (int)tail.size())
            tail.push_back(a[i]);
        else
            tail[pos] = a[i];
    }
    
    out << (int)tail.size();
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: Dùng thuật toán LIS (Longest Increasing Subsequence) với chặt nhị phân. Duy trì mảng tail chứa phần tử nhỏ nhất kết thúc LIS mỗi độ dài. Độ phức tạp: O(N log N).

// Bài 4: BALO - 0/1 Knapsack
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("BALO.INP");
    ofstream out("BALO.OUT");
    
    int n, W;
    in >> n >> W;
    
    vector<int> w(n), v(n);
    for(int i = 0; i < n; i++) in >> w[i] >> v[i];
    
    // dp[j] = giá trị lớn nhất với trọng lượng <= j
    vector<long long> dp(W + 1, 0);
    
    for(int i = 0; i < n; i++) {
        for(int j = W; j >= w[i]; j--) {
            dp[j] = max(dp[j], dp[j - w[i]] + v[i]);
        }
    }
    
    out << dp[W];
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: Bài toán ba lô 0/1 (0/1 Knapsack) kinh điển. Duyệt ngược j để đảm bảo mỗi vật chỉ dùng 1 lần. Độ phức tạp: O(N × W).
Bài 5: XÂU CON CHUNG DÀI NHẤT

Cho hai xâu ký tự S và T chỉ gồm các chữ cái in thường. Xâu con chung dài nhất (LCS) của S và T là xâu dài nhất mà vừa là dãy con của S vừa là dãy con của T.

  • Dòng 2: Xâu T (1 ≤ |T| ≤ 1000)
Kết quả ra (XAUCHUNG.OUT):
  • Một số nguyên duy nhất là độ dài LCS

📝 Code C++ 
// Bài 5: XAUCHUNG - LCS (Longest Common Subsequence)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("XAUCHUNG.INP");
    ofstream out("XAUCHUNG.OUT");
    
    string s, t;
    in >> s >> t;
    
    int m = s.size(), n = t.size();
    // dp[i][j] = độ dài LCS của s[1..i] và t[1..j]
    vector<vector<int>> dp(m + 1, vector<int>(n + 1, 0));
    
    for(int i = 1; i <= m; i++) {
        for(int j = 1; j <= n; j++) {
            if(s[i-1] == t[j-1])
                dp[i][j] = dp[i-1][j-1] + 1;
            else
                dp[i][j] = max(dp[i-1][j], dp[i][j-1]);
        }
    }
    
    out << dp[m][n];
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: Bài toán kinh điển LCS. Nếu s[i] == t[j] thì dp[i][j] = dp[i-1][j-1] + 1, ngược lại dp[i][j] = max(dp[i-1][j], dp[i][j-1]). Độ phức tạp: O(M × N).

// Bài 6: LUOI - DP trên lưới
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("LUOI.INP");
    ofstream out("LUOI.OUT");
    
    int m, n;
    in >> m >> n;
    
    vector<vector<long long>> a(m + 1, vector<long long>(n + 1));
    // dp[i][j] = tổng lớn nhất đi từ (1,1) đến (i,j)
    vector<vector<long long>> dp(m + 1, vector<long long>(n + 1, -1e18));
    
    for(int i = 1; i <= m; i++)
        for(int j = 1; j <= n; j++)
            in >> a[i][j];
    
    dp[1][1] = a[1][1];
    for(int i = 1; i <= m; i++) {
        for(int j = 1; j <= n; j++) {
            if(i == 1 && j == 1) continue;
            long long best = -1e18;
            if(i > 1) best = max(best, dp[i-1][j]);
            if(j > 1) best = max(best, dp[i][j-1]);
            dp[i][j] = best + a[i][j];
        }
    }
    
    out << dp[m][n];
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: dp[i][j] = tổng lớn nhất đi từ (1,1) đến (i,j). dp[i][j] = max(dp[i-1][j], dp[i][j-1]) + a[i][j]. Độ phức tạp: O(M × N).

// Bài 7: CHIAKEO - Subset Sum (DP)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("CHIAKEO.INP");
    ofstream out("CHIAKEO.OUT");
    
    int n;
    in >> n;
    
    vector<int> a(n);
    int sum = 0;
    for(int i = 0; i < n; i++) {
        in >> a[i];
        sum += a[i];
    }
    
    if(sum % 2 != 0) {
        out << "NO";
    } else {
        int half = sum / 2;
        // dp[j] = có thể chọn tổng đúng j không?
        vector<bool> dp(half + 1, false);
        dp[0] = true;
        
        for(int i = 0; i < n; i++) {
            for(int j = half; j >= a[i]; j--) {
                if(dp[j - a[i]]) dp[j] = true;
            }
        }
        
        out << (dp[half] ? "YES" : "NO");
    }
    
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: Quy về bài toán Subset Sum: kiểm tra có tập con nào có tổng = sum/2 không. Dùng DP boolean. Độ phức tạp: O(N × sum).

// Bài 8: SUACHUOI - Edit Distance
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("SUACHUOI.INP");
    ofstream out("SUACHUOI.OUT");
    
    string s, t;
    in >> s >> t;
    
    int m = s.size(), n = t.size();
    // dp[i][j] = chi phí nhỏ nhất biến s[1..i] thành t[1..j]
    vector<vector<int>> dp(m + 1, vector<int>(n + 1, 0));
    
    for(int i = 0; i <= m; i++) dp[i][0] = i;
    for(int j = 0; j <= n; j++) dp[0][j] = j;
    
    for(int i = 1; i <= m; i++) {
        for(int j = 1; j <= n; j++) {
            if(s[i-1] == t[j-1])
                dp[i][j] = dp[i-1][j-1];
            else
                dp[i][j] = min({dp[i-1][j], dp[i][j-1], dp[i-1][j-1]}) + 1;
        }
    }
    
    out << dp[m][n];
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: Bài toán Edit Distance (Levenshtein). dp[i][j] = chi phí nhỏ nhất biến s[1..i] thành t[1..j]. Nếu s[i]==t[j]: dp[i][j]=dp[i-1][j-1], ngược lại lấy min 3 thao tác. Độ phức tạp: O(M × N).

// Bài 9: PALIN - LPS = LCS(S, reverse(S))
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("PALIN.INP");
    ofstream out("PALIN.OUT");
    
    string s;
    in >> s;
    
    int n = s.size();
    string t = s;
    reverse(t.begin(), t.end());
    
    // LPS = LCS(s, reverse(s))
    // dp[i][j] = LCS của s[1..i] và s_ngược[1..j]
    vector<vector<int>> dp(n + 1, vector<int>(n + 1, 0));
    for(int i = 1; i <= n; i++) {
        for(int j = 1; j <= n; j++) {
            if(s[i-1] == t[j-1])
                dp[i][j] = dp[i-1][j-1] + 1;
            else
                dp[i][j] = max(dp[i-1][j], dp[i][j-1]);
        }
    }
    
    out << dp[n][n];
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: Xâu con đối xứng dài nhất (LPS) = LCS của S và reverse(S). Áp dụng LCS kinh điển. Độ phức tạp: O(N²).

// Bài 10: MAXSUM - Kadane + truy vết
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("MAXSUM.INP");
    ofstream out("MAXSUM.OUT");
    
    int n;
    in >> n;
    
    vector<long long> a(n + 1);
    for(int i = 1; i <= n; i++) in >> a[i];
    
    // curSum = tổng lớn nhất dãy con kết thúc tại vị trí hiện tại (Kadane)
    long long curSum = a[1], bestSum = a[1];
    int curL = 1, bestL = 1, bestR = 1;
    
    for(int i = 2; i <= n; i++) {
        if(curSum + a[i] >= a[i]) {
            curSum += a[i];
        } else {
            curSum = a[i];
            curL = i;
        }
        if(curSum > bestSum) {
            bestSum = curSum;
            bestL = curL;
            bestR = i;
        }
    }
    
    out << bestSum << "\n";
    out << bestL << " " << bestR;
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: Thuật toán Kadane: curSum = max(a[i], curSum + a[i]). Lưu thêm vị trí bắt đầu/kết thúc để truy vết. Độ phức tạp: O(N).

// Bài 11: MATRIX - DP đoạn (Matrix Chain Multiplication)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("MATRIX.INP");
    ofstream out("MATRIX.OUT");
    
    int n;
    in >> n;
    
    vector<long long> p(n + 1);
    for(int i = 0; i <= n; i++) in >> p[i];
    
    // dp[i][j] = chi phí nhân ma trận i đến j
    vector<vector<long long>> dp(n + 1, vector<long long>(n + 1, 0));
    
    for(int len = 2; len <= n; len++) {
        for(int i = 1; i <= n - len + 1; i++) {
            int j = i + len - 1;
            dp[i][j] = LLONG_MAX;
            for(int k = i; k < j; k++) {
                long long cost = dp[i][k] + dp[k+1][j] + p[i-1]*p[k]*p[j];
                dp[i][j] = min(dp[i][j], cost);
            }
        }
    }
    
    out << dp[1][n];
    in.close();
    out.close();
    return 0;
}

// Bài 12: CATSAT - Rod Cutting (DP)
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("CATSAT.INP");
    ofstream out("CATSAT.OUT");
    
    int L;
    in >> L;
    
    vector<long long> p(L + 1);
    for(int i = 1; i <= L; i++) in >> p[i];
    
    // dp[j] = doanh thu lớn nhất từ thanh dài j
    vector<long long> dp(L + 1, 0);
    
    for(int j = 1; j <= L; j++) {
        for(int i = 1; i <= j; i++) {
            dp[j] = max(dp[j], p[i] + dp[j - i]);
        }
    }
    
    out << dp[L];
    in.close();
    out.close();
    return 0;
}

#include <bits/stdc++.h>

int main() {

#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("XUONGNUI.INP");
    ofstream out("XUONGNUI.OUT");

// Bài 16: COPHIEU - DP mua bán cổ phiếu K lần
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("COPHIEU.INP");
    ofstream out("COPHIEU.OUT");
    
    int n, K;
    in >> n >> K;
    
    vector<int> p(n);
    for(int i = 0; i < n; i++) in >> p[i];
    
    if(2 * K >= n) {
        // Không giới hạn giao dịch
        long long ans = 0;
        for(int i = 1; i < n; i++)
            if(p[i] > p[i-1]) ans += p[i] - p[i-1];
        out << ans;
    } else {
        // dp[k][0] = lợi nhuận max sau k giao dịch, không giữ CP
        // dp[k][1] = lợi nhuận max sau k giao dịch, đang giữ CP
        vector<vector<long long>> dp(K + 1, vector<long long>(2, 0));
        for(int k = 0; k <= K; k++) dp[k][1] = -1e18;
        
        for(int i = 0; i < n; i++) {
            for(int k = K; k >= 1; k--) {
                dp[k][0] = max(dp[k][0], dp[k][1] + p[i]);
                dp[k][1] = max(dp[k][1], dp[k-1][0] - p[i]);
            }
        }
        
        out << dp[K][0];
    }
    
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: DP trạng thái với dp[k][0/1] = lợi nhuận max sau tối đa k giao dịch, trạng thái 0=không giữ, 1=đang giữ cổ phiếu. Độ phức tạp: O(N × K).

// Bài 17: TRAM - DP chia nhóm
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("TRAM.INP");
    ofstream out("TRAM.OUT");
    
    int n, k;
    in >> n >> k;
    
    vector<long long> x(n + 1);
    for(int i = 1; i <= n; i++) in >> x[i];
    
    // cost[i][j] = chi phí đặt 1 trạm phục vụ nhà i..j (đặt ở median)
    vector<vector<long long>> cost(n + 1, vector<long long>(n + 1, 0));
    for(int i = 1; i <= n; i++) {
        for(int j = i; j <= n; j++) {
            int mid = (i + j) / 2;
            for(int t = i; t <= j; t++)
                cost[i][j] += abs(x[t] - x[mid]);
        }
    }
    
    // dp[g][i] = chi phí min dùng g trạm phục vụ nhà 1..i
    vector<vector<long long>> dp(k + 1, vector<long long>(n + 1, 1e18));
    dp[0][0] = 0;
    
    for(int g = 1; g <= k; g++) {
        for(int i = 1; i <= n; i++) {
            for(int j = 0; j < i; j++) {
                dp[g][i] = min(dp[g][i], dp[g-1][j] + cost[j+1][i]);
            }
        }
    }
    
    out << dp[k][n];
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: Tính trước cost[i][j] = chi phí 1 trạm phục vụ nhà i..j (đặt ở vị trí trung vị). dp[g][i] = chi phí tối ưu dùng g trạm cho nhà 1..i. Độ phức tạp: O(K × N²).

// Bài 18: ROBOTVANG - DP 2 robot trên lưới
#include <bits/stdc++.h>
using namespace std;

int a[55][55];
int dp[55][55][55]; // dp[i][j1][j2]: hàng i, robot A cột j1, robot B cột j2

int main() {
    ifstream in("ROBOTVANG.INP");
    ofstream out("ROBOTVANG.OUT");
    
    int m, n;
    in >> m >> n;
    for(int i = 1; i <= m; i++)
        for(int j = 1; j <= n; j++)
            in >> a[i][j];
    
    memset(dp, -1, sizeof(dp));
    dp[1][1][n] = a[1][1] + a[1][n];
    
    int dx[] = {-1, 0, 1};
    
    for(int i = 2; i <= m; i++) {
        for(int j1 = 1; j1 <= n; j1++) {
            for(int j2 = j1; j2 <= n; j2++) {
                for(int d1 = 0; d1 < 3; d1++) {
                    for(int d2 = 0; d2 < 3; d2++) {
                        int pj1 = j1 + dx[d1];
                        int pj2 = j2 + dx[d2];
                        if(pj1 < 1 || pj1 > n || pj2 < 1 || pj2 > n) continue;
                        int a1 = min(pj1, pj2), a2 = max(pj1, pj2);
                        if(dp[i-1][a1][a2] < 0) continue;
                        int val = a[i][j1] + (j1 != j2 ? a[i][j2] : 0);
                        dp[i][j1][j2] = max(dp[i][j1][j2], dp[i-1][a1][a2] + val);
                    }
                }
            }
        }
    }
    
    int ans = 0;
    for(int j1 = 1; j1 <= n; j1++)
        for(int j2 = j1; j2 <= n; j2++)
            if(dp[m][j1][j2] > ans) ans = dp[m][j1][j2];
    
    out << ans;
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: DP 2 robot: dp[i][j1][j2] = tổng vàng max khi ở hàng i, robot A cột j1, B cột j2. Vì j1 ≤ j2, giảm số trạng thái. Độ phức tạp: O(M × N² × 9).

// Bài 19: CONGVIEC - Weighted Job Scheduling
#include <bits/stdc++.h>
using namespace std;

struct Job {
    int s, e;
    long long v;
};

bool cmpEnd(const Job &a, const Job &b) {
    return a.e < b.e;
}

int main() {
    ifstream in("CONGVIEC.INP");
    ofstream out("CONGVIEC.OUT");
    
    int n;
    in >> n;
    
    vector<Job> jobs(n);
    for(int i = 0; i < n; i++)
        in >> jobs[i].s >> jobs[i].e >> jobs[i].v;
    
    sort(jobs.begin(), jobs.end(), cmpEnd);
    
    // dp[i] = lợi nhuận max xét đến công việc i
    vector<long long> dp(n, 0);
    dp[0] = jobs[0].v;
    
    for(int i = 1; i < n; i++) {
        dp[i] = max(dp[i-1], jobs[i].v);
        // Tìm công việc cuối không chồng chéo bằng chặt nhị phân
        int lo = 0, hi = i - 1, best = -1;
        while(lo <= hi) {
            int mid = (lo + hi) / 2;
            if(jobs[mid].e <= jobs[i].s) {
                best = mid;
                lo = mid + 1;
            } else {
                hi = mid - 1;
            }
        }
        if(best >= 0)
            dp[i] = max(dp[i], dp[best] + jobs[i].v);
    }
    
    out << dp[n-1];
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: Sắp xếp theo thời gian kết thúc. dp[i] = max(dp[i-1], dp[last] + v[i]) với last = công việc gần nhất không chồng chéo (tìm bằng chặt nhị phân). Độ phức tạp: O(N log N).

using namespace std;

// dp[i] = giá trị lớn nhất lấy được từ nhà l..i (tối ưu bộ nhớ bằng prev1, prev2)
long long robLine(vector<long long> &a, int l, int r) {
    long long prev2 = 0, prev1 = 0;
    for(int i = l; i <= r; i++) {
        long long cur = max(prev1, prev2 + a[i]);
        prev2 = prev1;
        prev1 = cur;
    }
    return prev1;
}

int main() {
    ifstream in("TROMNHA.INP");
    ofstream out("TROMNHA.OUT");
    
    int n;
    in >> n;
    
    vector<long long> a(n);
    for(int i = 0; i < n; i++) in >> a[i];
    
    // Vòng tròn: max(rob nhà 0..n-2, rob nhà 1..n-1)
    out << max(robLine(a, 0, n - 2), robLine(a, 1, n - 1));
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: Nhà xếp vòng tròn → chia thành 2 trường hợp: lấy nhà 1 (không lấy nhà N) hoặc không lấy nhà 1 (có thể lấy nhà N). Mỗi trường hợp giải bằng House Robber trên dãy. Độ phức tạp: O(N).

📝 Code C++ 
// Bài 21: CHIANHOM - DP chia đoạn K nhóm
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("CHIANHOM.INP");
    ofstream out("CHIANHOM.OUT");
    
    int n, k;
    in >> n >> k;
    
    vector<int> a(n + 1);
    for(int i = 1; i <= n; i++) in >> a[i];
    
    // Tính trước cost[i][j] = max - min trong đoạn i..j
    vector<vector<long long>> cost(n + 1, vector<long long>(n + 1, 0));
    for(int i = 1; i <= n; i++) {
        int mn = a[i], mx = a[i];
        for(int j = i; j <= n; j++) {
            mn = min(mn, a[j]);
            mx = max(mx, a[j]);
            cost[i][j] = mx - mn;
        }
    }
    
    // dp[g][i] = chi phí min chia i người đầu thành g nhóm
    vector<vector<long long>> dp(k + 1, vector<long long>(n + 1, 1e18));
    dp[0][0] = 0;
    
    for(int g = 1; g <= k; g++) {
        for(int i = g; i <= n; i++) {
            for(int j = g - 1; j < i; j++) {
                dp[g][i] = min(dp[g][i], dp[g-1][j] + cost[j+1][i]);
            }
        }
    }
    
    out << dp[k][n];
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: Tính trước cost[i][j]. dp[g][i] = min(dp[g-1][j] + cost[j+1][i]) cho mọi j. Dạng DP chia đoạn kinh điển. Độ phức tạp: O(K × N²).

// Bài 22: NENNHA - Paint House DP
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("NENNHA.INP");
    ofstream out("NENNHA.OUT");
    
    int n;
    in >> n;
    
    // dp[i][c] = chi phí min sơn nhà 1..i, nhà i màu c (tối ưu bằng pr, pg, pb)
    long long r, g, b;
    long long pr = 0, pg = 0, pb = 0;
    
    for(int i = 0; i < n; i++) {
        in >> r >> g >> b;
        long long nr = min(pg, pb) + r;
        long long ng = min(pr, pb) + g;
        long long nb = min(pr, pg) + b;
        pr = nr; pg = ng; pb = nb;
    }
    
    out << min({pr, pg, pb});
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: dp[i][c] = chi phí min sơn nhà 1..i, nhà i sơn màu c. Chuyển tiếp: dp[i][c] = min(dp[i-1][c']) + cost[i][c] với c' ≠ c. Tối ưu không gian bằng 3 biến. Độ phức tạp: O(N).

// Bài 23: HOAHOP - Đếm số hoà hợp
#include <bits/stdc++.h>
using namespace std;

const int MOD = 1e9 + 7;

int main() {
    ifstream in("HOAHOP.INP");
    ofstream out("HOAHOP.OUT");
    
    int n;
    in >> n;
    
    // dp[d] = số cách kết thúc bằng chữ số d
    vector<long long> dp(10, 0);
    for(int d = 1; d <= 9; d++) dp[d] = 1;
    
    for(int i = 2; i <= n; i++) {
        vector<long long> ndp(10, 0);
        for(int d = 0; d <= 9; d++) {
            for(int p = max(0, d - 2); p <= min(9, d + 2); p++) {
                ndp[d] = (ndp[d] + dp[p]) % MOD;
            }
        }
        dp = ndp;
    }
    
    long long ans = 0;
    for(int d = 0; d <= 9; d++) ans = (ans + dp[d]) % MOD;
    
    out << ans;
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: dp[d] = số số kết thúc bằng chữ số d. Chữ số tiếp chỉ nằm trong [d-2, d+2]. Duyệt N-1 bước chuyển tiếp. Độ phức tạp: O(N × 10).

// Bài 24: NHAYDA - Frog Jump K bước
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("NHAYDA.INP");
    ofstream out("NHAYDA.OUT");
    
    int n, k;
    in >> n >> k;
    
    vector<long long> h(n + 1);
    for(int i = 1; i <= n; i++) in >> h[i];
    
    // dp[i] = chi phí nhỏ nhất nhảy đến đá thứ i
    vector<long long> dp(n + 1, 1e18);
    dp[1] = 0;
    
    for(int i = 2; i <= n; i++) {
        for(int j = 1; j <= k && i - j >= 1; j++) {
            dp[i] = min(dp[i], dp[i-j] + abs(h[i] - h[i-j]));
        }
    }
    
    out << dp[n];
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: dp[i] = chi phí nhỏ nhất đến đá i. dp[i] = min(dp[i-j] + |h[i]-h[i-j]|) với j = 1..K. Mở rộng bài leo cầu thang cơ bản. Độ phức tạp: O(N × K).

// Bài 25: MUAVE - DP ghép cặp liền kề
#include <bits/stdc++.h>
using namespace std;

int main() {
    ifstream in("MUAVE.INP");
    ofstream out("MUAVE.OUT");
    
    int n;
    in >> n;
    
    vector<long long> t(n + 1);
    for(int i = 1; i <= n; i++) in >> t[i];
    
    // dp[i] = thời gian nhỏ nhất để i người đầu mua xong vé
    vector<long long> dp(n + 1, 1e18);
    dp[0] = 0;
    dp[1] = t[1];
    
    for(int i = 2; i <= n; i++) {
        // Người i mua riêng, hoặc ghép với người i-1
        dp[i] = min(dp[i-1] + t[i], dp[i-2] + max(t[i-1], t[i]));
    }
    
    out << dp[n];
    in.close();
    out.close();
    return 0;
}
💡 Ý tưởng: dp[i] = thời gian min cho i người đầu. dp[i] = min(dp[i-1] + t[i], dp[i-2] + max(t[i-1], t[i])). Giống leo cầu thang nhưng có chi phí ghép cặp. Độ phức tạp: O(N).
