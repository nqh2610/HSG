3.1. N bao nhiêu thì dùng thuật gì?

QUY TẮC: Máy tính chạy được khoảng 10⁸ phép tính/giây. Nhân N với số vòng lặp, nếu > 10⁸ thì TLE!
Đề cho N ≤Bạn được dùngVí dụ bài toán
10Thử tất cả (vét cạn)Tìm thứ tự tối ưu → sinh hoán vị
20Duyệt 2^n tập conChọn tập con tổng = k → bitmask
10² (100)3 vòng for lồng nhauTìm bộ 3 số tổng = k → 3 vòng for
10³ - 5×10³2 vòng for lồng nhauDuyệt mọi cặp → 2 vòng for
10⁵1 vòng for + sortGhép cặp tối ưu → sort + greedy
10⁶Chỉ 1 vòng forTính tổng đoạn → prefix sum
MẸO NHỚ: N = 10⁵ (100.000) → cần sort + 1 vòng for. N = 10⁶ → chỉ được 1 vòng for. 2 vòng for lồng với N=10⁵ → TLE chắc chắn!

4.2. Two Pointers

KHI NÀO DÙNG:
  • Đề hỏi: "đoạn con dài nhất có tổng ≤ K"
  • Đề hỏi: "số cặp (i, j) thỏa a[i] + a[j] = k" (mảng đã sort)
  • Đề hỏi: "đoạn ngắn nhất chứa đủ k ký tự khác nhau"
  • Khi có 2 đầu mút (L, R) di chuyển được
// Tìm đoạn con dài nhất có tổng ≤ K
int L = 0, maxLen = 0;
long long sum = 0;

for (int R = 0; R < n; R++) {
    sum += a[R];
    while (sum > K && L <= R) {
        sum -= a[L];
        L++;
    }
    maxLen = max(maxLen, R - L + 1);
}

long long power(long long a, long long n, long long mod) {
    long long res = 1;
    a %= mod;
    while (n > 0) {
        if (n % 2 == 1)           // n lẻ
            res = res * a % mod;
        a = a * a % mod;
        n /= 2;
    }
    return res;
}

6.10. Kadane - Tìm đoạn con tổng lớn nhất

KHI NÀO DÙNG:
  • Đề hỏi: "đoạn con có tổng lớn nhất" (Max subarray sum)
  • Đề hỏi: "tổng max của đoạn liên tiếp"
  • Biến thể: max hiệu 2 tổng tiền tố, max tích
// Tìm đoạn con có tổng lớn nhất (có thể âm)
long long maxSum = a[0], cur = a[0];
for (int i = 1; i < n; i++) {
    cur = max((long long)a[i], cur + a[i]);  // Tiếp tục hoặc bắt đầu mới
    maxSum = max(maxSum, cur);
}
cout << maxSum;

// Nếu cần vị trí đoạn
int L = 0, R = 0, tempL = 0;
long long maxSum = a[0], cur = a[0];
for (int i = 1; i < n; i++) {
    if (cur + a[i] >= a[i]) {
        cur += a[i];
    } else {
        cur = a[i];
        tempL = i;
    }
    if (cur > maxSum) {
        maxSum = cur;
        L = tempL;
        R = i;
    }
}

6.11. Bitmask - Duyệt tập con

KHI NÀO DÙNG:
  • Đề hỏi: "tập con có tổng = k", "tập con thỏa điều kiện X"

// Duyệt tất cả 2^n tập con

    // mask biểu diễn tập con

        }
    }
    cout << "\n";
}

6.13. Phân tích thừa số nguyên tố

// Phân tích n thành thừa số nguyên tố
vector<pair<int, int>> phanTich(int n) {
    vector<pair<int, int>> res;  // {số nguyên tố, số mũ}
    for (int i = 2; i * i <= n; i++) {
        if (n % i == 0) {
            int cnt = 0;
            while (n % i == 0) {
                n /= i;
                cnt++;
            }
            res.push_back({i, cnt});
        }
    }
    if (n > 1) res.push_back({n, 1});
    return res;
}

// Đếm số ước của n
int demUoc(int n) {
    int cnt = 0;
    for (int i = 1; i * i <= n; i++) {
        if (n % i == 0) {
            cnt++;
            if (i != n / i) cnt++;
        }
    }
    return cnt;
}

7.5. Multiset

  • Đề hỏi: "tìm túi nhỏ nhất đủ chứa mẫu kích thước x" (ghép cặp greedy)

multiset<int> ms;

if (it != ms.end()) {

    ms.erase(it);    // XÓA 1 phần tử (không phải tất cả)
}

// Ứng dụng: Ghép cặp greedy
// Tìm túi nhỏ nhất đủ chứa mẫu có kích thước w
for (int i = 0; i < n; i++) {
    auto it = ms.lower_bound(mau[i]);

7.6. DSU (Disjoint Set Union)

KHI NÀO DÙNG:
  • Đề hỏi: "2 phần tử có cùng nhóm không"
  • Đề hỏi: "có bao nhiêu nhóm/thành phần liên thông"
  • Bài ghép các cặp lại với nhau (union)
  • Thuật toán Kruskal tìm cây khung nhỏ nhất
int cha[100005], hang[100005];

void khoiTao(int n) {
    for (int i = 1; i <= n; i++) {
        cha[i] = i;
        hang[i] = 0;
    }
}

int tim(int u) {
    if (cha[u] != u) cha[u] = tim(cha[u]);  // Nén đường
    return cha[u];
}

void hopNhat(int u, int v) {
    int pu = tim(u), pv = tim(v);
    if (pu == pv) return;  // Đã cùng nhóm
    
    // Union by rank
    if (hang[pu] < hang[pv]) swap(pu, pv);
    cha[pv] = pu;
    if (hang[pu] == hang[pv]) hang[pu]++;
}

bool cungNhom(int u, int v) {
    return tim(u) == tim(v);
}

7.7. BIT (Binary Indexed Tree / Fenwick Tree)

KHI NÀO DÙNG:
  • Đề hỏi: "tổng đoạn [L, R]" + "cập nhật giá trị a[i]" (nhiều lần)
  • Đề hỏi: "đếm số nghịch thế trong mảng"
  • Prefix Sum nhưng CÓ CẬP NHẬT (Prefix Sum thường không cập nhật được)
long long bit[100005];
int n;

void capNhat(int i, long long delta) {
    for (; i <= n; i += i & (-i))
        bit[i] += delta;
}

long long tong(int i) {
    long long s = 0;
    for (; i > 0; i -= i & (-i))
        s += bit[i];
    return s;
}

// Tổng đoạn [l, r]
long long tongDoan(int l, int r) {
    return tong(r) - tong(l - 1);
}

// Khởi tạo từ mảng a[]
void xayDung(int mang[], int n) {
    for (int i = 1; i <= n; i++)
        capNhat(i, mang[i]);
}

7.8. Sliding Window với Deque

KHI NÀO DÙNG:
  • Đề hỏi: "max/min trong mỗi cửa sổ kích thước k"
  • Đề hỏi: "với mỗi vị trí, tìm max trong k phần tử gần nhất"
  • Bài cần duyệt qua mảng, mỗi bước cần max/min của đoạn
// Tìm MAX trong mỗi cửa sổ kích thước k
deque<int> dq;  // Lưu CHỈ SỐ, không phải giá trị

for (int i = 0; i < n; i++) {
    // Xóa các phần tử ngoài cửa sổ
    while (!dq.empty() && dq.front() <= i - k)
        dq.pop_front();
    
    // Xóa các phần tử nhỏ hơn a[i] (không bao giờ là max)
    while (!dq.empty() && a[dq.back()] <= a[i])
        dq.pop_back();
    
    dq.push_back(i);
    
    // Từ i = k-1 trở đi, in max của cửa sổ
    if (i >= k - 1)
        cout << a[dq.front()] << " ";  // Max của cửa sổ
}
