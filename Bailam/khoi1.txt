#include <bits/stdc++.h>
using namespace std;
ifstream in("DOANEM.INP");
ofstream out("DOANEM.OUT");
const int maxn = 1e6 + 5;
int a[maxn];
int main()
{
    ios_base::sync_with_stdio(false);
    in.tie(NULL);
    int n, k; in >> n >> k;
    for(int i = 0; i < n; i++) in >> a[i];
    multiset<int> ms;
    int l = 0;
    long long result = 0;
    for(int r = 0; r < n; r++)
    {
        ms.insert(a[r]);
        while(*ms.rbegin() - *ms.begin() > k)
        {
            ms.erase(ms.find(a[l]));
            l++;
        }
        result += r - l + 1;
    }
    out << result;
    in.close();
    out.close();
    return 0;
}

#include <bits/stdc++.h>
using namespace std;
ifstream in("CHIAGD.INP");
ofstream out("CHIAGD.OUT");
using ll = long long;
const int maxn = 5005;
const ll INF = 1e18;
int a[maxn];
ll prefix[maxn];
ll  dp[maxn]; // chi phí ít nhất cho i công việc 
int main()
{
    ios_base::sync_with_stdio(false);
    in.tie(NULL);
    int n;
    ll S; 
    in >> n >> S;
    prefix[0] = 0;
    for(int i = 1; i <= n; i++)
    {
        in >> a[i];
        prefix[i] = prefix[i - 1] + a[i];
        dp[i] = INF;
    }
    dp[0] = 0; // chưa có công việc nào 
    for(ll i = 1; i <= n; i++)
    {
        for(ll j = 0; j < i; j++)
        {
            ll time = (prefix[i] - prefix[j]);
            dp[i] = min(dp[i], dp[j] + time * time + S); // chi phí tốt nhất trước đó + chi phí đoạn còn lại 
        }
    }
    out << dp[n] - S;
    in.close();
    out.close();
    return 0;
}
